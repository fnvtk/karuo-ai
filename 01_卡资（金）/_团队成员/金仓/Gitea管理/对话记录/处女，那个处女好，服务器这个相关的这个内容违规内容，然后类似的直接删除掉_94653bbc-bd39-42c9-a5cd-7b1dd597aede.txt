def 生成签名(api_key: str) -> tuple:
    now_time = int(time.time())
    sign_str = str(now_time) + hashlib.md5(api_key.encode("utf-8")).hexdigest()
    request_token = hashlib.md5(sign_str.encode("utf-8")).hexdigest()
    return now_time, request_token

        host, port = host_port, 3306

        "port": port,
        "user": user.group(1).strip(),
        "password": pw.group(1).strip(),
        "database": DB_NAME,
    }

def 从账号索引读腾讯云数据库():
    if not os.path.isfile(IDX):
        return None
    with open(IDX, "r", encoding="utf-8") as f:
        c = f.read()
    m = re.search(r"###\s*腾讯云数据库\s*\n(.*?)(?=###|\Z)", c, re.DOTALL)
    if not m:
        return None
    b = m.group(1)
    addr = re.search(r"地址\s*\|\s*`([^`]+)`", b)
    user = re.search(r"账号\s*\|\s*`([^`]+)`", b)
    pw = re.search(r"密码\s*\|\s*`([^`]+)`", b)
    if not (addr and user and pw):
        return None
    host_port = addr.group(1).strip()
    if ":" in host_port:
        host, port = host_port.rsplit(":", 1)
        port = int(port)
    else:
        host, port = host_port, 3306
    return {
        "host": host.strip(),
        "port": port,
        "user": user.group(1).strip(),
        "password": pw.group(1).strip(),
        "database": DB_NAME,
    }

def 获取版块列表(conn):
    """返回 fid -> name 及 名称含香港的 fid 列表。"""
    with conn.cursor() as cur:
        cur.execute(f"SELECT fid, name FROM {TABLE_PREFIX}forum_forum WHERE status='1'")
        rows = cur.fetchall()
    fid2name = {r[0]: (r[1] or "") for r in rows}
    hk_fids = [
        fid for fid, name in fid2name.items()
        if any(k in (name or "") for k in FID_KEYWORDS_香港)
    ]
    return fid2name, hk_fids


def 命中了(text: str) -> bool:
    if not text:

    text = (text or "").strip()

def 扫描并收集违规tid(conn, 仅香港: bool, fid2name: dict, hk_fids: list):
    """返回 [(tid, fid, subject, 版块名), ...]"""
    with conn.cursor() as cur:
        cur.execute(f"""
            SELECT t.tid, t.fid, t.subject,
                   (SELECT p.message FROM {TABLE_PREFIX}forum_post p WHERE p.tid = t.tid AND p.first=1 LIMIT 1) AS first_msg
            FROM {TABLE_PREFIX}forum_thread t
            WHERE t.displayorder >= 0
        """)
        rows = cur.fetchall()
    result = []
    for tid, fid, subject, first_msg in rows:
        if 仅香港 and fid not in hk_fids:
            continue
        subject = (subject or "")[:200]
        msg = (first_msg or "")[:2000]  # 首楼前 2000 字
        if 命中了(subject) or 命中了(msg):
            result.append((tid, fid, subject, fid2name.get(fid, "")))
    return result
