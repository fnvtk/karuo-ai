  contents: #!/bin/bash
set -e
cd "$(dirname "$0")"

APPLICATIONS_WECHAT="/Applications/WeChat.app"
# 用到的临时备份名（若已存在则覆盖）
BACKUP_NAME="/Applications/WeChat_backup_$(date +%Y%m%d_%H%M%S).app"

# 从运行中的进程得到 WeChat.app 的路径
get_running_wechat_app_path() {
    local pid
    pid=$(osascript -e 'tell application "System Events" to get the unix id of (first process whose name contains "WeChat")' 2>/dev/null || true)
    if [[ -z "$pid" || "$pid" == "NOTRUNNING" ]]; then
        return 1
    fi
    # 通过 PID 取主程序路径，格式多为 .../WeChat.app/Contents/MacOS/WeChat
    local exe
    exe=$(ps -p "$pid" -o command= 2>/dev/null | sed 's/ .*//')
    if [[ -z "$exe" || "$exe" == *"grep"* ]]; then
        return 1
    fi
    # 去掉 /Contents/MacOS/WeChat，得到 .app 路径
    if [[ "$exe" == *".app/Contents/MacOS/"* ]]; then
        echo "${exe%/*/Contents/MacOS/*}"
        return 0
    fi
    return 1
}

# 在已挂载卷里找 WeChat.app（例如从 DMG 安装包）
find_wechat_in_volumes() {
    local v
    for v in /Volumes/WeChat* /Volumes/微信*; do
        if [[ -d "$v/WeChat.app" ]]; then
            echo "$v/WeChat.app"
            return 0
        fi
    done
    return 1
}

SOURCE_APP=""
# 1) 若微信正在运行，用当前运行的这个
if SOURCE_APP=$(get_running_wechat_app_path 2>/dev/null) && [[ -d "$SOURCE_APP" ]]; then
    echo "已找到正在运行的微信: $SOURCE_APP"
# 2) 若在挂载的安装盘里（如 DMG）
elif SOURCE_APP=$(find_wechat_in_volumes 2>/dev/null) && [[ -d "$SOURCE_APP" ]]; then
    echo "已找到安装包中的微信: $SOURCE_APP"
else
    echo "未找到「安装包里的」微信（请保持弹窗时运行本脚本，或先挂载微信安装 DMG）。"
    echo "将直接从应用程序启动已有微信。"
    open -a "WeChat"
    exit 0
fi

# 若来源就是应用程序里的，直接启动
if [[ "$(cd "$SOURCE_APP" && pwd)" == "$(cd "$APPLICATIONS_WECHAT" 2>/dev/null && pwd)" ]]; then
    echo "微信已在应用程序中，正在启动。"
    open -a "WeChat"
    exit 0
fi

# 先退出正在运行的微信
osascript -e 'tell application "WeChat" to quit' 2>/dev/null || true
sleep 1

# 若 /Applications/WeChat.app 已存在，先备份
if [[ -d "$APPLICATIONS_WECHAT" ]]; then
    echo "正在备份现有: $APPLICATIONS_WECHAT -> $BACKUP_NAME"
    sudo -v
    sudo rm -rf "$BACKUP_NAME"
    sudo mv "$APPLICATIONS_WECHAT" "$BACKUP_NAME"
fi

# 把「当前这个」微信移到应用程序
echo "正在复制到: $APPLICATIONS_WECHAT"
sudo cp -R "$SOURCE_APP" "$APPLICATIONS_WECHAT"
# 修复权限，避免以后要 sudo 才能开
sudo chown -R "$USER:admin" "$APPLICATIONS_WECHAT"

echo "已放入应用程序，正在启动微信。"
open -a "WeChat"
