"""
腾讯云 账单拉取与财务报表
========================
通过 Billing API 拉取最近 18 个月账单（按产品汇总），按年汇总并生成财务报表 Markdown。
凭证：环境变量或 金盾/账号与API索引 § 腾讯云。
依赖：pip install tencentcloud-sdk-python-billing
"""
import os
import re
import sys
from collections import defaultdict
from datetime import datetime, timedelta

def _load_creds():
    sid = os.environ.get("TENCENTCLOUD_SECRET_ID")
    skey = os.environ.get("TENCENTCLOUD_SECRET_KEY")
    if sid and skey:
        return sid, skey
    base = os.path.dirname(os.path.abspath(__file__))
    team_root = os.path.abspath(os.path.join(base, "..", "..", ".."))
    p = os.path.join(team_root, "金盾", "账号密码与资料管理", "账号与API索引.md")
    if not os.path.isfile(p):
        return None, None
    try:
        with open(p, "r", encoding="utf-8") as f:
            c = f.read()
        m = re.search(r"###\s*腾讯云\s*\n(.*?)(?=###|\Z)", c, re.DOTALL)
        if not m:
            return None, None
        b = m.group(1)
        s1 = re.search(r"SecretId\s*\|\s*`([^`]+)`", b)
        s2 = re.search(r"SecretKey\s*\|\s*`([^`]+)`", b)
        return (s1.group(1).strip(), s2.group(1).strip()) if s1 and s2 else (None, None)
    except Exception:
        return None, None


def main():
    sid, skey = _load_creds()
    if not sid or not skey:
        print("未找到腾讯云凭证。→ 找 @金盾 配置 账号与API索引 § 腾讯云 或设 TENCENTCLOUD_SECRET_ID/SECRET_KEY", file=sys.stderr)
        sys.exit(1)
    try:
        from tencentcloud.common import credential
        from tencentcloud.billing.v20180709 import billing_client, models
    except ImportError:
        print("请安装: pip install tencentcloud-sdk-python-billing", file=sys.stderr)
        sys.exit(1)

    cred = credential.Credential(sid, skey)
    client = billing_client.BillingClient(cred, "ap-guangzhou")

    # 拉取最近 18 个月，按月请求（API 单次仅支持单月）
    now = datetime.now()
    month_list = []
    for i in range(18):
        d = now - timedelta(days=30 * i)
        month_list.append(d.strftime("%Y-%m"))

    # 去重并按时间正序
    seen = set()
    ordered = []
    for m in reversed(month_list):
        if m not in seen:
            seen.add(m)
            ordered.append(m)

    # 按月拉取按产品汇总
    monthly_total = {}       # month -> 总费用(元)
    monthly_by_product = {}  # month -> { product_name: amount }

    for month in ordered:
        req = models.DescribeBillSummaryByProductRequest()
        req.BeginTime = month
        req.EndTime = month
        req.PayType = "consume"  # 消费
        try:
            resp = client.DescribeBillSummaryByProduct(req)
        except Exception as e:
            print(f"  [跳过] {month}: {e}", file=sys.stderr)
            continue
        if getattr(resp, "Ready", 0) != 1:
            print(f"  [未就绪] {month} 数据准备中", file=sys.stderr)
            continue
        total = 0.0
        by_product = {}
        if getattr(resp, "SummaryTotal", None):
            st = resp.SummaryTotal
            total = float(getattr(st, "RealTotalCost", 0) or 0)
        monthly_total[month] = total
        for item in getattr(resp, "SummaryOverview", None) or []:
            name = getattr(item, "BusinessCodeName", None) or getattr(item, "BusinessCode", "") or "其他"
            cost = float(getattr(item, "RealTotalCost", 0) or 0)
            by_product[name] = by_product.get(name, 0) + cost
        monthly_by_product[month] = by_product

    # 按年汇总
    year_total = defaultdict(float)
    year_by_product = defaultdict(lambda: defaultdict(float))
    for month, total in monthly_total.items():
        year = month[:4]
        year_total[year] += total
        for prod, amount in monthly_by_product.get(month, {}).items():
            year_by_product[year][prod] += amount

    # 生成财务报表 Markdown
    lines = []
    lines.append("# 腾讯云服务器 消费财务报表")
    lines.append("")
    lines.append("> 数据来源：腾讯云 Billing API（DescribeBillSummaryByProduct），按**消费**汇总。")
    lines.append("> 说明：API 最多拉取**近 18 个月**数据，更早年份需在控制台导出。")
    lines.append("")
    lines.append("## 一、按年汇总（总消费）")
    lines.append("")
    lines.append("| 年份 | 总消费（元） | 同比变化 | 占全部比例 |")
    lines.append("|------|-------------|----------|------------|")

    years = sorted(year_total.keys())
    grand_total = sum(year_total.values())
    prev = None
    for year in years:
        t = year_total[year]
        pct = (t / grand_total * 100) if grand_total else 0
        if prev is not None and prev != 0:
            change = (t - prev) / prev * 100
            change_str = f"{change:+.1f}%"
        else:
            change_str = "-"
        lines.append(f"| {year} | {t:,.2f} | {change_str} | {pct:.1f}% |")
        prev = t

    lines.append("")
    lines.append("**合计（API 可查区间）**：" + f" **¥{grand_total:,.2f}**")
    lines.append("")
    lines.append("## 二、按年 × 产品分布")
    lines.append("")
    all_products = set()
    for y, prods in year_by_product.items():
        all_products.update(prods.keys())
    all_products = sorted(all_products)

    header = "| 年份 | " + " | ".join(all_products) + " | **合计** |"
    lines.append(header)
    lines.append("|" + "------|" * (len(all_products) + 2))

    for year in years:
        row = [year]
        for p in all_products:
            v = year_by_product[year].get(p, 0)
            row.append(f"{v:,.2f}" if v else "-")
        row.append(f"**{year_total[year]:,.2f}**")
        lines.append("| " + " | ".join(row) + " |")

    lines.append("")
    lines.append("## 三、按月明细（近 18 个月）")
    lines.append("")
    lines.append("| 月份 | 总消费（元） | 环比 |")
    lines.append("|------|-------------|------|")

    prev_m = None
    for month in ordered:
        t = monthly_total.get(month, 0)
        if prev_m is not None and prev_m != 0:
            change = (t - prev_m) / prev_m * 100
            change_str = f"{change:+.1f}%"
        else:
            change_str = "-"
        lines.append(f"| {month} | {t:,.2f} | {change_str} |")
        prev_m = t

    lines.append("")
    lines.append("## 四、年度对比与占比（简要）")
    lines.append("")
    if years:
        min_y, max_y = min(years), max(years)
        lines.append(f"- 统计区间（API）：**{min_y}** 至 **{max_y}**（约 {len(ordered)} 个月）。")
    lines.append(f"- 总消费（上表合计）：**¥{grand_total:,.2f}**。")
    if len(years) >= 2:
        first, last = year_total[years[0]], year_total[years[-1]]
        if first and first != 0:
            lines.append(f"- 首年 vs 末年：{years[0]} 年 ¥{first:,.2f} → {years[-1]} 年 ¥{last:,.2f}（{(last/first-1)*100:+.1f}%）。")
    lines.append("")
    lines.append("---")
    lines.append("")
    lines.append("*生成时间：{}*".format(datetime.now().strftime("%Y-%m-%d %H:%M")))
    lines.append("")

    out = "\n".join(lines)
    print(out)

    # 可选：写入 references
    out_path = os.path.join(base, "..", "references", "腾讯云_服务器消费财务报表.md")
    out_path = os.path.abspath(out_path)
    if os.path.isdir(os.path.dirname(out_path)):
        with open(out_path, "w", encoding="utf-8") as f:
            f.write(out)
        print("", file=sys.stderr)
        print(f"已写入：{out_path}", file=sys.stderr)


if __name__ == "__main__":
    main()

    client = billing_client.BillingClient(cred, "ap-guangzhou")
    req = models.DescribeBillSummaryByProductRequest()
    req.BeginTime = month
    req.EndTime = month
    req.PayType = "consume"
    try:
        resp = client.DescribeBillSummaryByProduct(req)
    except Exception as e:
        print(f"拉取失败 {month}: {e}", file=sys.stderr)
        sys.exit(1)
    if getattr(resp, "Ready", 0) != 1:
        print(f"[未就绪] {month} 数据准备中", file=sys.stderr)
        sys.exit(1)

    total = 0.0
    by_product = []
    if getattr(resp, "SummaryTotal", None):
        total = float(getattr(resp.SummaryTotal, "RealTotalCost", 0) or 0)
    for item in getattr(resp, "SummaryOverview", None) or []:
        name = getattr(item, "BusinessCodeName", None) or getattr(item, "BusinessCode", "") or "其他"
        cost = float(getattr(item, "RealTotalCost", 0) or 0)
        if cost != 0:
            by_product.append((name, cost))

    by_product.sort(key=lambda x: -x[1])

    # 输出单月片段（Markdown 表格，供月度报表引用）
    lines = [
        f"## 腾讯云消费（{month}）",
        "",
        f"- **当月合计**：¥{total:,.2f}",
        "",
        "| 产品 | 金额（元） |",
        "|------|------------|",
    ]
    for name, cost in by_product:
        lines.append(f"| {name} | {cost:,.2f} |")
    lines.append("")
    print("\n".join(lines))
