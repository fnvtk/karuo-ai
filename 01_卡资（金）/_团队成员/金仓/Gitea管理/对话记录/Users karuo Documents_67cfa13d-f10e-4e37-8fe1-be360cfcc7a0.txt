  contents: #!/bin/bash
# iCloud下载进度实时监控脚本

STUCK_COUNT=0

    echo "[$(date '+%H:%M:%S')] Documents: $DOCS_HR | iCloud: $ICLOUD_HR | bird CPU: ${BIRD_CPU:-停止}%"
    
    # 检测是否卡住
    if [ "$TOTAL" = "$PREV_SIZE" ]; then
        STUCK_COUNT=$((STUCK_COUNT + 1))

            STUCK_COUNT=0
        fi
    else
        STUCK_COUNT=0
    fi
    
    PREV_SIZE=$TOTAL
    sleep 30
done

# Google 全托管 RAG 来了！Gemini File Search 深度解析

这是谷歌最近推出的一项新功能，File Search in Gemini API。它把原本复杂的 RAG 系统直接集成到了 Gemini API 中，不需要搭建向量数据库，甚至不用配置检索流程，只需要简单的 API 调用，就能实现一个完整的 RAG 系统。无论是搭建知识库、打造专属的 AI 客服，还是企业内部文档问答，都能快速落地。而且它支持多种文件格式，解析速度快，所有的检索能力都由 Gemini 完全托管，可以说这是目前最省力、性价比最高的 RAG 方案之一。那这期视频，我们就来详细介绍这个 Gemini File Search 以及它的具体使用。

首先我们还是先来介绍一下到底什么是 RAG，为什么现在大家都在用 RAG。假设我们刚刚研发了一款全新的运动手环，我们想要了解它的详细参数，比如续航时长，如果直接去问 AI，可能得到模糊甚至是错误的答案。为什么会这样呢？第一个原因是大模型的知识是有截止日期的，只包含在某个时间点之前的内容，那之后出现的新产品、新技术它就不知道了。当然现在很多大模型已经具备了联网搜索的能力，如果这款手环的资料已经公开了，那 AI 可以去搜索，但如果这个资料没有公开，那 AI 没有办法搜索到，也就无能为力了。即便信息在网上存在，AI 在专业知识上也可能理解不够精准，特别是在面对像技术参数、规格细节等这些专业的问题的时候，AI 可能无法精确的回答。那这种情况下我们该怎么办呢？这时候就需要外置的知识库来补充这个模型的能力，也就是我们常说的 RAG。比如我们提前准备好这款运动手环的参数文档，用它来构建一个专属的知识库。当用户提问的时候，这个 AI 就不再盲猜，而是先从知识库里检索到最相关的内容，拿到正确的上下文之后，再用自己的语言生成答案，最后给出的回复就会又准确，又清晰，又可靠。那这个流程就是 RAG，RAG 的中文叫做检索增强生成，意思就是通过检索外部的知识库来增强大模型的回答，让结果更加准确。

一个 RAG 系统到底是怎么工作的，它的完整流程是怎么样的，它的每一步分别在做什么？首先，我们需要把要使用的这个文档去做一个切分，因为一个 PDF 文档可能会非常的长，模型没有办法去一次性处理，我们需要把它切分成一段一段的小块，然后使用一个嵌入模型把这些文本块转成向量，这些向量会存入到向量数据库，这样这个专属的知识库它的存储就搭建完成了。后面当用户提出一个问题的时候，这个问题同样会经过嵌入模型被转成向量，那系统会用这个向量，去向量数据库里检索出最相关的 K 条内容。这里通常还会有一步排序，由模型把这些候选片段按照相关性重新排一下顺序，确保最终提交给大模型的内容是最有价值的。把这些检索到的文本块作为上下文，让大模型在真实资料的基础上回答用户的问题，这样生成的回答就既准确又可靠。那这整套流程就是 RAG 的工作方式。

那可以看到，要想自己从 0 搭建一个 RAG 系统还是比较复杂的，不仅需要去维护向量数据库，还要自己控制文档切片的大小、嵌入式模型的选择，还有这个 TOP K 的选取，这些参数都会直接影响到系统的性能，决定最后回答的准确性。而 Gemini File Search 它的一个最主要的亮点，就是把这些所有的复杂的操作全部都集成到了 Gemini API 里面，我们只需要通过简单的 API 调用，就能去快速搭建一个 RAG 系统。代码里第一步先创建了一个数据存储，也就是你的专属知识库，然后把文档上传到刚才创建的数据存储里面，Gemini 会自动帮我们完成切片、向量化等所有的处理。第二步当用户提问的时候，直接调用这个检索接口，从数据存储里面查找出最相关的内容，然后让大模型根据这些信息去生成最终的答案。它不仅能给出答案，还能附带对原文的引用，也就是说，它会告诉用户为什么要这么回答，引用的是文档里面的哪一部分，这样就可以很好的提升这个回答的可信度，让用户更加放心。整个流程加起来也就几十行代码，一共执行了 3 次 API 调用，不用调参数，不用自己写检索的逻辑，几分钟就能把一个可用的 RAG 系统快速的跑起来。

我们再来介绍一下这个 Gemini File Search 它的内部是如何工作的。当我们上传一个文件时，会提取出一个文件对象，然后对数据进行切分，使用嵌入模型转为向量，并且存入数据库。当用户查询的时候，Gemini 会先判断这个用户的问题是否需要使用外置的知识库。如果用户的问题是一些普遍性的问题，比如法国的首都是哪里，这时候 Gemini 就会直接生成答案并返回给用户。如果判断问题需要使用外置知识库，会先生成一些优化后的查询，然后使用这个嵌入式模型把这些查询转为向量，去数据库检索出最相关的 K 条内容，作为上下文返回给 Gemini。Gemini 会结合用户的问题和查询到的这个上下文去进行判断，如果认为当前获取到的数据还不足够去生成答案，就会去继续生成新的查询，然后继续从数据库中获取相关的上下文，直到 Gemini 认为获取到了足够的内容，才会去使用这些内容生成最终的答案并返回给用户。

可以看到 Gemini File Search 和传统的 RAG 系统非常相似，但是它自己也做了一些优化，比如判断是否需要使用知识库、判断是否获取到了足够的知识等等，而且关键是这些操作全部都集成到了 Gemini API 中，使用非常方便。

const fs = require("fs");
const path = require("path");

const API_KEY = process.env.GEMINI_API_KEY || "你的API_KEY";

/**
 * 1. 创建数据存储（知识库）
 */
async function createDataStore(displayName = STORE_NAME) {
  console.log(`📦 创建数据存储: ${displayName}`);
  
  const response = await genAI.createCachedContent({
    displayName: displayName,
    // 可选：设置文档切片参数
    // chunkSize: 1024,
    // chunkOverlap: 200,
  });
  
  console.log(`✅ 数据存储创建成功: ${response.name}`);
  return response;
}

/**
 * 2. 上传单个文档
 */
async function uploadDocument(filePath, dataStoreName) {
  const fileName = path.basename(filePath);
  console.log(`📄 上传文档: ${fileName}`);
  
  const fileContent = fs.readFileSync(filePath, "utf-8");
  
  // 上传到数据存储
  const response = await genAI.uploadFile({
    file: {
      displayName: fileName,
      mimeType: getMimeType(filePath),
      content: fileContent,
    },
    parent: dataStoreName,
  });
  
  console.log(`✅ 上传成功: ${response.name}`);
  return response;
}

/**
 * 3. 批量上传文档
 */
async function uploadDocuments(directory, dataStoreName) {
  const files = fs.readdirSync(directory);
  const results = [];
  
  for (const file of files) {
    const filePath = path.join(directory, file);
    const stat = fs.statSync(filePath);
    
    if (stat.isFile() && isSupported(filePath)) {
      try {
        const result = await uploadDocument(filePath, dataStoreName);
        results.push(result);
      } catch (error) {
        console.error(`❌ 上传失败: ${file}`, error.message);
      }
    }
  }
  
  console.log(`\n📊 上传完成: ${results.length}/${files.length} 个文件`);
  return results;
}

async function askQuestion(question, dataStoreName) {
  console.log(`\n🔍 问题: ${question}`);
  console.log("─".repeat(50));
  
  const model = genAI.getGenerativeModel({
    model: "gemini-2.0-flash",
    // 启用 File Search
    tools: [{
      fileSearch: {
        dataStoreId: dataStoreName,
      }
    }],
  });
  
  const result = await model.generateContent(question);

def load_documents():
    """加载聊天记录文件"""
    from langchain_community.document_loaders import DirectoryLoader, TextLoader
    
    console.print("\n📂 扫描聊天记录目录...", style="bold blue")
    
    # 加载 txt 文件
    txt_loader = DirectoryLoader(
        CHAT_RECORDS_DIR,
        glob="**/*.txt",
        loader_cls=TextLoader,
        loader_kwargs={"encoding": "utf-8"},
        show_progress=True,
        use_multithreading=True,
    )
    
    # 加载 md 文件
    md_loader = DirectoryLoader(
        CHAT_RECORDS_DIR,
        glob="**/*.md",
        loader_cls=TextLoader,
        loader_kwargs={"encoding": "utf-8"},
        show_progress=True,
    )
    
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        console=console,
    ) as progress:
        task = progress.add_task("加载文档...", total=None)
        
        try:
            txt_docs = txt_loader.load()
            progress.update(task, description=f"已加载 {len(txt_docs)} 个 TXT 文件")
        except Exception as e:
            console.print(f"⚠️ 加载TXT时出错: {e}", style="yellow")
            txt_docs = []
        
        try:
            md_docs = md_loader.load()
            progress.update(task, description=f"已加载 {len(md_docs)} 个 MD 文件")
        except Exception as e:
            console.print(f"⚠️ 加载MD时出错: {e}", style="yellow")
            md_docs = []
    
    all_docs = txt_docs + md_docs
    console.print(f"✅ 总共加载 {len(all_docs)} 个文档", style="bold green")
    
    return all_docs

# ============ 文档切分 ============

def split_documents(documents):
    """将文档切分成小块"""
    from langchain_text_splitters import RecursiveCharacterTextSplitter
    
    console.print("\n✂️ 切分文档...", style="bold blue")
    
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=1000,  # 每块1000字符
        chunk_overlap=200,  # 重叠200字符
        length_function=len,
        separators=["\n\n", "\n", "。", "！", "？", "，", " ", ""],
    )
    
    chunks = text_splitter.split_documents(documents)
    console.print(f"✅ 切分成 {len(chunks)} 个文本块", style="bold green")
    
    return chunks

def create_vector_store(chunks):
    """创建本地向量数据库"""
    from langchain_community.embeddings import OllamaEmbeddings
    from langchain_community.vectorstores import Chroma
    
    console.print("\n🔢 创建向量索引（完全本地）...", style="bold blue")
    
    # 使用 Ollama 本地嵌入模型
    embeddings = OllamaEmbeddings(
        model=EMBEDDING_MODEL,
        base_url="http://localhost:11434",
    )
    
    # 先下载嵌入模型（如果没有）
    console.print("  检查嵌入模型...", style="dim")
    os.system(f"ollama pull {EMBEDDING_MODEL} 2>/dev/null")
    
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        console=console,
    ) as progress:
        task = progress.add_task("创建向量索引...", total=None)
        
        # 创建 Chroma 向量数据库
        vectorstore = Chroma.from_documents(
            documents=chunks,
            embedding=embeddings,
            persist_directory=CHROMA_DB_PATH,
        )
        
        progress.update(task, description="向量索引创建完成")
    
    console.print(f"✅ 向量数据库已保存到: {CHROMA_DB_PATH}", style="bold green")
    
    return vectorstore

def load_vector_store():
    """加载已有的向量数据库"""
    from langchain_community.embeddings import OllamaEmbeddings
    from langchain_community.vectorstores import Chroma
    
    if not os.path.exists(CHROMA_DB_PATH):
        console.print("❌ 向量数据库不存在，请先运行: python karuo_local_rag.py index", style="bold red")
        sys.exit(1)
    
    embeddings = OllamaEmbeddings(
        model=EMBEDDING_MODEL,
        base_url="http://localhost:11434",
    )
    
    vectorstore = Chroma(
        persist_directory=CHROMA_DB_PATH,
        embedding_function=embeddings,
    )
    
    return vectorstore

# ============ RAG 问答 ============

def create_rag_chain(vectorstore):
    """创建 RAG 问答链"""
    from langchain_ollama import OllamaLLM
    from langchain.chains import RetrievalQA
    from langchain.prompts import PromptTemplate
    
    # 本地大模型
    llm = OllamaLLM(
        model=OLLAMA_MODEL,
        base_url="http://localhost:11434",
        temperature=0.3,
    )
    
    # 自定义提示词
    prompt_template = """你是卡若的私人AI助手。根据以下聊天记录内容回答问题。

聊天记录内容:
{context}

问题: {question}

请用中文回答，要求:
1. 基于聊天记录内容回答，不要编造
2. 如果记录中没有相关信息，请明确说明
3. 引用具体的对话内容作为依据
4. 回答简洁、准确、有条理

回答:"""

    PROMPT = PromptTemplate(
        template=prompt_template,
        input_variables=["context", "question"]
    )
    
    # 创建检索问答链
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=vectorstore.as_retriever(search_kwargs={"k": 5}),
        return_source_documents=True,
        chain_type_kwargs={"prompt": PROMPT},
    )
    
    return qa_chain

def ask_question(qa_chain, question):
    """提问并获取回答"""
    console.print(f"\n🔍 问题: {question}", style="bold cyan")
    console.print("─" * 50)
    
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        console=console,
    ) as progress:
        task = progress.add_task("思考中...", total=None)
        result = qa_chain.invoke({"query": question})
        progress.update(task, description="完成")
    
    # 显示回答
    console.print("\n💬 回答:", style="bold green")
    console.print(Panel(result["result"], border_style="green"))
    
    # 显示来源
    if result.get("source_documents"):
        console.print("\n📚 引用来源:", style="bold yellow")
        for i, doc in enumerate(result["source_documents"][:3], 1):
            source = doc.metadata.get("source", "未知")
            # 只显示相对路径
            source = source.replace(CHAT_RECORDS_DIR + "/", "")
            content_preview = doc.page_content[:100].replace("\n", " ")
            console.print(f"  [{i}] {source}")
            console.print(f"      {content_preview}...", style="dim")
    
    return result

def interactive_search():
    """交互式搜索"""

            break

"""
卡若AI 本地知识库 - 完全本地运行的 RAG 系统
所有数据都在本地处理，不上传任何内容到云端

使用方法:

    python karuo_local_rag.py ask "问题"  # 单次提问
"""

import os
import sys
from pathlib import Path
from rich.console import Console
from rich.panel import Panel
from rich.progress import Progress, SpinnerColumn, TextColumn, BarColumn, TaskProgressColumn
from rich.markdown import Markdown

# 配置
CHAT_RECORDS_DIR = "/Users/karuo/Documents/聊天记录"
CHROMA_DB_PATH = "/Users/karuo/Documents/个人/卡若AI/06_AI知识库/local_rag/chroma_db"
OLLAMA_MODEL = "qwen2.5:1.5b"  # 中文优化模型（轻量版）
EMBEDDING_MODEL = "nomic-embed-text"  # 本地嵌入模型

# 性能配置
CHUNK_SIZE = 800  # 减小块大小
CHUNK_OVERLAP = 100  # 减小重叠
BATCH_SIZE = 50  # 批量处理大小

console = Console()

# ============ 文档加载 ============

def load_documents():
    """加载聊天记录文件"""
    from langchain_community.document_loaders import DirectoryLoader, TextLoader
    
    console.print("\n📂 扫描聊天记录目录...", style="bold blue")
    
    # 加载 txt 文件
    txt_loader = DirectoryLoader(
        CHAT_RECORDS_DIR,
        glob="**/*.txt",
        loader_cls=TextLoader,
        loader_kwargs={"encoding": "utf-8"},
        show_progress=True,
        use_multithreading=True,
        silent_errors=True,  # 忽略错误文件
    )
    
    # 加载 md 文件
    md_loader = DirectoryLoader(
        CHAT_RECORDS_DIR,
        glob="**/*.md",
        loader_cls=TextLoader,
        loader_kwargs={"encoding": "utf-8"},
        show_progress=True,
        silent_errors=True,
    )
    
    try:
        txt_docs = txt_loader.load()
        console.print(f"  已加载 {len(txt_docs)} 个 TXT 文件", style="dim")
    except Exception as e:
        console.print(f"⚠️ 加载TXT时出错: {e}", style="yellow")
        txt_docs = []
    
    try:
        md_docs = md_loader.load()
        console.print(f"  已加载 {len(md_docs)} 个 MD 文件", style="dim")
    except Exception as e:
        console.print(f"⚠️ 加载MD时出错: {e}", style="yellow")
        md_docs = []
    
    all_docs = txt_docs + md_docs
    console.print(f"✅ 总共加载 {len(all_docs)} 个文档", style="bold green")
    
    return all_docs

# ============ 文档切分 ============

def split_documents(documents):
    """将文档切分成小块"""
    from langchain_text_splitters import RecursiveCharacterTextSplitter
    
    console.print("\n✂️ 切分文档...", style="bold blue")
    
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=CHUNK_SIZE,
        chunk_overlap=CHUNK_OVERLAP,
        length_function=len,
        separators=["\n\n", "\n", "。", "！", "？", "，", " ", ""],
    )
    
    chunks = text_splitter.split_documents(documents)
    console.print(f"✅ 切分成 {len(chunks)} 个文本块", style="bold green")
    
    return chunks

# ============ 向量存储 ============

def create_vector_store(chunks):
    """创建本地向量数据库（批量处理）"""
    from langchain_ollama import OllamaEmbeddings
    from langchain_community.vectorstores import Chroma
    
    console.print(f"\n🔢 创建向量索引（批量处理，每批 {BATCH_SIZE} 个）...", style="bold blue")
    
    # 使用新版 OllamaEmbeddings
    embeddings = OllamaEmbeddings(
        model=EMBEDDING_MODEL,
        base_url="http://localhost:11434",
    )
    
    total_chunks = len(chunks)
    vectorstore = None
    
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        BarColumn(),
        TaskProgressColumn(),
        console=console,
    ) as progress:
        task = progress.add_task("索引中...", total=total_chunks)
        
        # 分批处理
        for i in range(0, total_chunks, BATCH_SIZE):
            batch = chunks[i:i + BATCH_SIZE]
            
            if vectorstore is None:
                # 第一批：创建数据库
                vectorstore = Chroma.from_documents(
                    documents=batch,
                    embedding=embeddings,
                    persist_directory=CHROMA_DB_PATH,
                )
            else:
                # 后续批次：追加到数据库
                vectorstore.add_documents(batch)
            
            progress.update(task, advance=len(batch))
            progress.update(task, description=f"已处理 {min(i + BATCH_SIZE, total_chunks)}/{total_chunks}")
    
    console.print(f"✅ 向量数据库已保存到: {CHROMA_DB_PATH}", style="bold green")
    
    return vectorstore

def load_vector_store():
    """加载已有的向量数据库"""
    from langchain_ollama import OllamaEmbeddings
    from langchain_community.vectorstores import Chroma
    
    if not os.path.exists(CHROMA_DB_PATH):
        console.print("❌ 向量数据库不存在，请先运行: python karuo_local_rag.py index", style="bold red")
        sys.exit(1)
    
    embeddings = OllamaEmbeddings(
        model=EMBEDDING_MODEL,
        base_url="http://localhost:11434",
    )
    
    vectorstore = Chroma(
        persist_directory=CHROMA_DB_PATH,
        embedding_function=embeddings,
    )
    
    return vectorstore

def create_rag_chain(vectorstore):
    """创建 RAG 问答链"""
    from langchain_ollama import OllamaLLM
    from langchain.chains import RetrievalQA
    from langchain.prompts import PromptTemplate
    
    # 本地大模型
    llm = OllamaLLM(
        model=OLLAMA_MODEL,
        base_url="http://localhost:11434",
        temperature=0.3,
    )
    
    # 自定义提示词
    prompt_template = """你是卡若的私人AI助手。根据以下聊天记录内容回答问题。

聊天记录内容:
{context}

问题: {question}

请用中文回答，要求:
1. 基于聊天记录内容回答，不要编造
2. 如果记录中没有相关信息，请明确说明
3. 引用具体的对话内容作为依据
4. 回答简洁、准确、有条理

回答:"""

    PROMPT = PromptTemplate(
        template=prompt_template,
        input_variables=["context", "question"]
    )
    
    # 创建检索问答链
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=vectorstore.as_retriever(search_kwargs={"k": 5}),
        return_source_documents=True,
        chain_type_kwargs={"prompt": PROMPT},
    )
    
    return qa_chain

def ask_question(qa_chain, question):

            content_preview = doc.page_content[:100].replace("\n", " ")
            console.print(f"  [{i}] {source}")
            console.print(f"      {content_preview}...", style="dim")
    
    return result

"""
快速索引 - 只索引 Soul 目录（约67个文件）
用于快速验证系统可用性
"""

import os
import sys
from pathlib import Path
from rich.console import Console
from rich.panel import Panel
from rich.progress import Progress, SpinnerColumn, TextColumn, BarColumn, TaskProgressColumn

# 配置
SOUL_DIR = "/Users/karuo/Documents/聊天记录/soul"
CHROMA_DB_PATH = "/Users/karuo/Documents/个人/卡若AI/06_AI知识库/local_rag/chroma_db"
EMBEDDING_MODEL = "nomic-embed-text"

CHUNK_SIZE = 1000
CHUNK_OVERLAP = 100
BATCH_SIZE = 20

console = Console()

def main():
    console.print(Panel.fit(
        "🚀 快速索引 - Soul派对记录\n约67个文件，预计5-10分钟",
        border_style="blue"
    ))
    
    from langchain_community.document_loaders import DirectoryLoader, TextLoader
    from langchain_text_splitters import RecursiveCharacterTextSplitter
    from langchain_ollama import OllamaEmbeddings
    from langchain_community.vectorstores import Chroma
    
    # 1. 加载文档
    console.print("\n📂 加载 Soul 目录...", style="bold blue")
    
    loader = DirectoryLoader(
        SOUL_DIR,
        glob="**/*.txt",
        loader_cls=TextLoader,
        loader_kwargs={"encoding": "utf-8"},
        show_progress=True,
        silent_errors=True,
    )
    
    documents = loader.load()
    console.print(f"✅ 加载了 {len(documents)} 个文件", style="bold green")
    
    # 2. 切分文档
    console.print("\n✂️ 切分文档...", style="bold blue")
    
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=CHUNK_SIZE,
        chunk_overlap=CHUNK_OVERLAP,
        separators=["\n\n", "\n", "。", "！", "？", "，", " "],
    )
    
    chunks = splitter.split_documents(documents)
    console.print(f"✅ 切分成 {len(chunks)} 个文本块", style="bold green")
    
    # 3. 创建向量索引
    console.print(f"\n🔢 创建向量索引...", style="bold blue")
    
    embeddings = OllamaEmbeddings(
        model=EMBEDDING_MODEL,
        base_url="http://localhost:11434",
    )
    
    total = len(chunks)
    vectorstore = None
    
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        BarColumn(),
        TaskProgressColumn(),
        console=console,
    ) as progress:
        task = progress.add_task("索引中...", total=total)
        
        for i in range(0, total, BATCH_SIZE):
            batch = chunks[i:i + BATCH_SIZE]
            
            if vectorstore is None:
                vectorstore = Chroma.from_documents(
                    documents=batch,
                    embedding=embeddings,
                    persist_directory=CHROMA_DB_PATH,
                )
            else:
                vectorstore.add_documents(batch)
            
            progress.update(task, advance=len(batch))
            progress.update(task, description=f"已处理 {min(i + BATCH_SIZE, total)}/{total}")
    
    console.print("\n" + "=" * 50)
    console.print("✅ 索引完成！", style="bold green")
    console.print(f"   文件数: {len(documents)}")
    console.print(f"   文本块: {len(chunks)}")
    console.print(f"   数据库: {CHROMA_DB_PATH}")
    console.print("\n现在可以运行:")
    console.print("  python karuo_local_rag.py search")

if __name__ == "__main__":
    main()
