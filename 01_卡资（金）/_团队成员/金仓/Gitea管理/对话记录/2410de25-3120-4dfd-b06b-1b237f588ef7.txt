-- 用途：股票、高手、后台配置、用户/账号密码、AI 向量化存储，便于 AI 查询与分析
-- 字符集：utf8mb4 | 引擎：InnoDB

-- -------------------------------
-- 1. 系统配置（后台参数、策略参数）
-- -------------------------------
DROP TABLE IF EXISTS `sys_config`;
CREATE TABLE `sys_config` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `config_key` varchar(128) NOT NULL COMMENT '配置键，如 strategy.holiday.buy_before_days',
  `config_value` text COMMENT '配置值（可 JSON）',
  `config_group` varchar(64) DEFAULT 'default' COMMENT '分组：strategy/auth/crawl/nlp/risk',
  `value_type` varchar(32) DEFAULT 'string' COMMENT 'string/number/json/boolean',
  `remark` varchar(255) DEFAULT NULL,
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_config_key` (`config_key`),
  KEY `idx_config_group` (`config_group`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='系统/策略/后台配置';

-- 2. 用户与认证
-- -------------------------------
DROP TABLE IF EXISTS `users`;
CREATE TABLE `users` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `user_id` varchar(64) NOT NULL COMMENT '业务用户ID',
  `username` varchar(64) NOT NULL,
  `password_hash` varchar(255) NOT NULL,
  `role` varchar(32) DEFAULT 'user' COMMENT 'admin/user',
  `profile` json DEFAULT NULL COMMENT '昵称、头像等',
  `status` tinyint NOT NULL DEFAULT 1 COMMENT '1正常 0禁用',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_id` (`user_id`),
  UNIQUE KEY `uk_username` (`username`),
  KEY `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='用户表';

-- 第三方/交易账户绑定（含账号密码加密存储）
DROP TABLE IF EXISTS `app_accounts`;
CREATE TABLE `app_accounts` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `user_id` varchar(64) NOT NULL COMMENT '关联 users.user_id',
  `account_id` varchar(64) NOT NULL COMMENT '业务账户ID',
  `broker` varchar(64) NOT NULL COMMENT '券商/平台：雪球/模拟券商等',

  `account_no` varchar(128) DEFAULT NULL COMMENT '账号（可加密）',
  `password_encrypted` varchar(512) DEFAULT NULL COMMENT '密码加密存储',
  `is_default` tinyint NOT NULL DEFAULT 0,
  `extra` json DEFAULT NULL,
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_account` (`user_id`,`account_id`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='用户绑定账户（含账号密码）';

DROP TABLE IF EXISTS `stocks`;
CREATE TABLE `stocks` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `code` varchar(16) NOT NULL COMMENT '6位股票代码',
  `name` varchar(64) NOT NULL,
  `market` varchar(16) DEFAULT 'SH' COMMENT 'SH/SZ/BJ',
  `sector` varchar(64) DEFAULT NULL COMMENT '行业/板块',
  `price` decimal(12,4) DEFAULT NULL,
  `change_pct` decimal(8,4) DEFAULT NULL,
  `stats` json DEFAULT NULL COMMENT 'hot_score、成交量等扩展',
  `extra` json DEFAULT NULL COMMENT '扩展字段',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_code` (`code`),
  KEY `idx_name` (`name`),
  KEY `idx_sector` (`sector`),
  FULLTEXT KEY `ft_name_sector` (`name`,`sector`) WITH PARSER ngram COMMENT 'AI/全文检索'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='股票基础信息';

async def mysql_get_config_group(group: str) -> Dict[str, Any]:
    out = {}
    async with mysql_connection() as conn:
        if conn is None:
            return out
        async with conn.cursor() as cur:
            await cur.execute(
                "SELECT config_key, config_value, value_type FROM sys_config WHERE config_group = %s",
                (group,),
            )
            for r in await cur.fetchall():
                k, v, vt = r[0], r[1], (r[2] or "string")
                if vt == "number" and v is not None:
                    try:
                        v = float(v) if "." in str(v) else int(v)
                    except ValueError:
                        pass
                elif vt == "json" and v:
                    try:
                        v = json.loads(v)
                    except Exception:
                        pass
                elif vt == "boolean":
                    v = str(v).lower() in ("1", "true", "yes")
                out[k] = v
    return out

    """批量插入论坛帖子（采集结果），重复 post_id 则忽略。返回插入条数。"""

            sql = """
                INSERT IGNORE INTO forum_posts (
                    post_id, platform, url, author_user_id, author_username, author_avatar,
                    author_followers, author_verified, author_is_expert, title, body,
                    stocks_mentioned, sentiment, keywords, views, likes, replies, shares,

                    %s, %s, %s, %s, %s
                )
            """
            for doc in posts:
                row = _post_doc_to_row(doc)
                try:
                    await cur.execute(sql, row)
                    if cur.rowcount:
                        inserted += 1
                except Exception:
                    pass
            conn.commit()
    return inserted

def _ensure_hot_score(post: dict) -> dict:
    if "hot_score" in post:
        return post
    meta = post.get("meta") or {}
    v = meta.get("views", 0) or 0
    lk = meta.get("likes", 0) or 0
    rp = meta.get("replies", 0) or 0
    post["hot_score"] = round(v * 0.3 + lk * 0.5 + rp * 0.2, 2)
    post.setdefault("crawled_at", datetime.utcnow())
    return post
