def check_ollama():
    """检查Ollama服务是否运行"""
    try:
        import urllib.request
        req = urllib.request.Request(f"{OLLAMA_URL}/api/tags")
        with urllib.request.urlopen(req, timeout=3) as resp:
            return resp.status == 200
    except:
        return False

def get_available_models():
    """获取可用模型列表"""
    try:
        import urllib.request
        req = urllib.request.Request(f"{OLLAMA_URL}/api/tags")
        with urllib.request.urlopen(req, timeout=5) as resp:
            data = json.loads(resp.read().decode())
            return [m["name"] for m in data.get("models", [])]
    except:
        return []

def chat_with_ollama(prompt, model, context=None):
    """调用Ollama生成回复（流式输出）"""
    import urllib.request
    
    payload = {
        "model": model,
        "prompt": prompt,
        "stream": True,
    }
    if context:
        payload["context"] = context
    
    data = json.dumps(payload).encode("utf-8")
    req = urllib.request.Request(
        f"{OLLAMA_URL}/api/generate",
        data=data,
        headers={"Content-Type": "application/json"}
    )
    
    full_response = ""
    new_context = None
    
    try:
        with urllib.request.urlopen(req, timeout=120) as resp:
            for line in resp:
                if line:
                    chunk = json.loads(line.decode())
                    text = chunk.get("response", "")
                    full_response += text
                    print(text, end="", flush=True)
                    
                    if chunk.get("done"):
                        new_context = chunk.get("context")
                        break
    except KeyboardInterrupt:
        print(f"\n{Colors.YELLOW}[已中断]{Colors.END}")
    
    print()  # 换行
    return full_response, new_context

def write_memory(content, category="记录"):
    """写入记忆"""
    try:
        now = datetime.datetime.now()
        date_str = now.strftime("%Y-%m-%d")
        time_str = now.strftime("%H:%M")
        
        # 读取现有内容
        existing = read_memory()
        
        # 查找或创建日期段落
        date_header = f"### {date_str}"

        if date_header in existing:
            # 在日期段落后追加
            lines = existing.split('\n')
            new_lines = []
            inserted = False
            for i, line in enumerate(lines):
                new_lines.append(line)
                if line.strip() == date_header and not inserted:
                    new_lines.append(entry)
                    inserted = True
            existing = '\n'.join(new_lines)
        else:
            # 创建新日期段落
            existing += f"\n\n{date_header}\n{entry}"
        
        with open(MEMORY_FILE, 'w', encoding='utf-8') as f:
            f.write(existing)
        
        return True
    except Exception as e:
        print(f"写入记忆失败: {e}")
        return False

class KaruoHandler(SimpleHTTPRequestHandler):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, directory=os.path.join(os.path.dirname(__file__), 'static'), **kwargs)
    
    def do_GET(self):
        parsed = urlparse(self.path)
        
        if parsed.path == '/':
            self.send_response(200)
            self.send_header('Content-Type', 'text/html; charset=utf-8')
            self.end_headers()
            with open(os.path.join(os.path.dirname(__file__), 'static', 'index.html'), 'rb') as f:
                self.wfile.write(f.read())
        
        elif parsed.path == '/api/skills':
            skills = get_skills()
            self.send_json(skills)
        
        elif parsed.path == '/api/memory':
            memories = get_recent_memories(7)
            self.send_json({"memories": memories})
        
        elif parsed.path == '/api/status':
            # 检查Ollama状态
            try:
                req = urllib.request.Request(f"{OLLAMA_URL}/api/tags")
                with urllib.request.urlopen(req, timeout=2) as resp:
                    status = "online"
            except:
                status = "offline"
            self.send_json({"status": status, "model": current_model})
        
        elif parsed.path.startswith('/api/siri'):
            # Siri快捷指令接口
            params = parse_qs(parsed.query)
            query = params.get('q', [''])[0]
            if query:
                response = call_ollama(query)
                speak(response)  # 语音播报
                self.send_json({"response": response})
            else:
                self.send_json({"error": "缺少参数q"})
        
        else:
            super().do_GET()
    
    def do_POST(self):
        parsed = urlparse(self.path)
        content_length = int(self.headers.get('Content-Length', 0))
        body = self.rfile.read(content_length).decode('utf-8')
        
        try:
            data = json.loads(body) if body else {}
        except:
            data = {}
        
        if parsed.path == '/api/chat':
            message = data.get('message', '')
            voice = data.get('voice', False)
            
            if message:
                response = call_ollama(message)
                if voice:
                    speak(response)
                self.send_json({"response": response})
            else:
                self.send_json({"error": "缺少消息内容"})
        
        elif parsed.path == '/api/memory':
            content = data.get('content', '')
            category = data.get('category', '用户记录')
            if content:
                success = write_memory(content, category)
                self.send_json({"success": success})
            else:
                self.send_json({"error": "缺少内容"})
        
        elif parsed.path == '/api/clear':
            global conversation_history
            conversation_history = []
            self.send_json({"success": True})
        
        elif parsed.path == '/api/model':
            global current_model
            model = data.get('model', 'large')
            current_model = MODELS.get(model, MODELS['large'])
            self.send_json({"model": current_model})
        
        else:
            self.send_json({"error": "未知接口"})
    
    def send_json(self, data):
        self.send_response(200)
        self.send_header('Content-Type', 'application/json; charset=utf-8')
        self.send_header('Access-Control-Allow-Origin', '*')
        self.end_headers()
        self.wfile.write(json.dumps(data, ensure_ascii=False).encode('utf-8'))
    
    def do_OPTIONS(self):
        self.send_response(200)
        self.send_header('Access-Control-Allow-Origin', '*')
        self.send_header('Access-Control-Allow-Methods', 'GET, POST, OPTIONS')
        self.send_header('Access-Control-Allow-Headers', 'Content-Type')
        self.end_headers()
    
    def log_message(self, format, *args):
        pass  # 静默日志

"""
🔥 语音网关：Fn 双击听写 → 本地模型（卡若AI）→ 输出 + 朗读

使用方式：
  1. 运行本脚本，弹出小窗口
  2. 光标放在输入框内，按「双击 Fn」用系统听写说话
  3. 回车或点「发送」→ 内容发给本地助手，回复显示并朗读

依赖：Python 3 + tkinter（系统自带），无需额外安装
"""

import tkinter as tk
from tkinter import ttk, scrolledtext, messagebox
import json
import urllib.request
import subprocess
import re
import threading

API_URL = "http://localhost:5888"
APP_DIR = "/Users/karuo/Documents/个人/卡若AI/_共享模块/local_llm/app"


def ensure_server():
    """若服务未运行则尝试启动"""
    try:
        req = urllib.request.Request(f"{API_URL}/api/status")
        urllib.request.urlopen(req, timeout=2)
        return True
    except Exception:
        try:
            subprocess.Popen(
                ["python3", f"{APP_DIR}/server.py"],
                cwd=APP_DIR,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
            import time
            time.sleep(3)
            req = urllib.request.Request(f"{API_URL}/api/status")
            urllib.request.urlopen(req, timeout=5)
            return True
        except Exception:
            return False


def call_karuo(message: str, voice: bool = True) -> str:
    """调用本地助手 API，可选朗读"""
    try:
        data = json.dumps({"message": message, "voice": voice}).encode("utf-8")
        req = urllib.request.Request(
            f"{API_URL}/api/chat",
            data=data,
            headers={"Content-Type": "application/json"},
            method="POST",
        )
        with urllib.request.urlopen(req, timeout=60) as r:
            out = json.loads(r.read().decode())
            response = out.get("response", out.get("error", "抱歉，出错了"))
            if voice and response:
                text = re.sub(r"[*#`\[\]]", "", response)[:500]
                subprocess.Popen(["say", "-v", "Tingting", text])
            return response
    except Exception as e:
        return f"请求失败：{str(e)}"


class VoiceGatewayApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("卡若AI · 语音网关")
        self.root.geometry("520x420")
        self.root.minsize(400, 320)

        # 提示
        tip = ttk.Label(
            self.root,
            text="在下方输入框内点击聚焦后，按「双击 Fn」听写，再回车或点发送",
            font=("", 10),
            wraplength=480,
        )
        tip.pack(pady=(10, 4), padx=10, anchor="w")

        # 输入框（听写目标）
        self.input_frame = ttk.Frame(self.root)
        self.input_frame.pack(fill="x", padx=10, pady=4)
        self.input_text = tk.Text(
            self.input_frame,
            height=3,
            font=("", 14),
            wrap="word",
            relief="solid",
            borderwidth=1,
        )
        self.input_text.pack(side="left", fill="both", expand=True)
        scroll_in = ttk.Scrollbar(self.input_frame, command=self.input_text.yview)
        scroll_in.pack(side="right", fill="y")
        self.input_text.config(yscrollcommand=scroll_in.set)

        # 发送区
        btn_frame = ttk.Frame(self.root)
        btn_frame.pack(fill="x", padx=10, pady=6)
        self.send_btn = ttk.Button(btn_frame, text="发送到卡若AI", command=self.on_send)
        self.send_btn.pack(side="left", padx=(0, 8))
        self.status = ttk.Label(btn_frame, text="", foreground="gray")
        self.status.pack(side="left")

        # 回复区
        ttk.Label(self.root, text="回复：", font=("", 10)).pack(anchor="w", padx=10, pady=(8, 0))
        self.output_text = scrolledtext.ScrolledText(
            self.root,
            height=12,
            font=("", 12),
            wrap="word",
            state="disabled",
            relief="solid",
            borderwidth=1,
        )
        self.output_text.pack(fill="both", expand=True, padx=10, pady=6)

        self.input_text.bind("<Return>", self._on_enter)
        self.input_text.bind("<Control-Return>", lambda e: self.input_text.insert("insert", "\n"))
        self.root.protocol("WM_DELETE_WINDOW", self.on_close)

    def _on_enter(self, event):
        """回车发送，Ctrl+回车换行由上面绑定"""
        if event.state & 0x4:  # Ctrl
            return
        self.on_send()
        return "break"

    def on_send(self):
        content = self.input_text.get("1.0", "end").strip()
        if not content:
            self.status.config(text="请输入或听写内容")
            return
        self.send_btn.config(state="disabled")
        self.status.config(text="正在请求本地模型…")

        def do_request():
            try:
                resp = call_karuo(content, voice=True)
                self.root.after(0, lambda: self._show_response(resp, None))
            except Exception as e:
                self.root.after(0, lambda: self._show_response("", str(e)))

        threading.Thread(target=do_request, daemon=True).start()

    def _show_response(self, response: str, error: str = None):
        self.send_btn.config(state="normal")
        if error:
            self.status.config(text="请求失败")
            self.output_text.config(state="normal")
            self.output_text.insert("end", f"\n[错误] {error}\n")
            self.output_text.see("end")
            self.output_text.config(state="disabled")
            return
        self.status.config(text="已回复并朗读")
        self.output_text.config(state="normal")
        self.output_text.insert("end", f"\n你：{self.input_text.get('1.0', 'end').strip()}\n")
        self.output_text.insert("end", f"卡若AI：{response}\n")
        self.output_text.see("end")
        self.output_text.config(state="disabled")
        self.input_text.delete("1.0", "end")
        self.input_text.focus_set()

    def on_close(self):
        self.root.destroy()

    def run(self):
        if not ensure_server():
            messagebox.showerror("错误", "无法连接本地助手服务，请先运行 server.py 或双击「卡若AI本地助手.app」")
            return
        self.input_text.focus_set()
        self.root.mainloop()


if __name__ == "__main__":
    VoiceGatewayApp().run()

"""
卡若AI 语音：点一下 → 双击 Fn 说话 → 直接听回复（中间过程隐藏）

import tkinter as tk
from tkinter import font as tkfont
import json
import urllib.request
import subprocess
import re
import threading

API_URL = "http://localhost:5888"
APP_DIR = "/Users/karuo/Documents/个人/卡若AI/_共享模块/local_llm/app"
AUTO_SEND_DELAY_MS = 1200  # 听写停顿多久后自动发送

def ensure_server():
    try:
        urllib.request.urlopen(urllib.request.Request(f"{API_URL}/api/status"), timeout=2)
        return True
    except Exception:
        try:
            subprocess.Popen(
                ["python3", f"{APP_DIR}/server.py"],
                cwd=APP_DIR, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL,
            )
            import time
            time.sleep(3)
            urllib.request.urlopen(urllib.request.Request(f"{API_URL}/api/status"), timeout=5)
            return True
        except Exception:
            return False

def call_karuo(message: str, voice: bool = True) -> str:
    try:
        data = json.dumps({"message": message, "voice": voice}).encode("utf-8")
        req = urllib.request.Request(
            f"{API_URL}/api/chat", data=data,
            headers={"Content-Type": "application/json"}, method="POST",
        )
        with urllib.request.urlopen(req, timeout=60) as r:
            out = json.loads(r.read().decode())
            response = out.get("response", out.get("error", "抱歉，出错了"))
            if voice and response:
                text = re.sub(r"[*#`\[\]]", "", response)[:500]
                subprocess.Popen(["say", "-v", "Tingting", text])
            return response
    except Exception as e:
        return f"请求失败：{str(e)}"


class VoiceGatewayApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("卡若AI")
        self.root.geometry("380x200")
        self.root.minsize(320, 160)
        self._auto_send_id = None
        self._sending = False

        # 上方：唯一操作区 = 点这里然后双击 Fn 说话（即输入框，但样式像一句提示）

            self.root,
            font=tkfont.Font(size=14),
            justify="center",
            width=28,
        )
        self.input_text.pack(pady=8, padx=20, fill="x")
        self.input_text.bind("<KeyRelease>", self._on_input_change)
        self.input_text.bind("<FocusIn>", lambda e: self.prompt.config(text="双击 Fn 说话"))
        self.input_text.bind("<FocusOut>", lambda e: self.prompt.config(text="点这里，然后双击 Fn 说话"))

        # 下方：只显示回复（无“你：”等中间过程）
        self.reply_label = tk.Label(
            self.root,
            text="",
            font=tkfont.Font(size=13),
            wraplength=340,
            justify="left",
            fg="#111",
        )
        self.reply_label.pack(pady=12, padx=20, fill="both", expand=True)
        self.root.protocol("WM_DELETE_WINDOW", self.root.destroy)

    def _on_input_change(self, event):
        if self._sending:
            return
        if self._auto_send_id:
            self.root.after_cancel(self._auto_send_id)
        content = self.input_text.get().strip()
        if not content:
            self._auto_send_id = None
            return
        self._auto_send_id = self.root.after(AUTO_SEND_DELAY_MS, self._do_send)

    def _do_send(self):
        self._auto_send_id = None
        content = self.input_text.get().strip()
        if not content:
            return
        self._sending = True
        self.input_text.delete(0, "end")
        self.reply_label.config(text="…", fg="#666")

        def run():
            try:
                resp = call_karuo(content, voice=True)
                self.root.after(0, lambda: self._show_reply(resp, None))
            except Exception as e:
                self.root.after(0, lambda: self._show_reply("", str(e)))

        threading.Thread(target=run, daemon=True).start()

    def _show_reply(self, response: str, error: str):
        self._sending = False
        if error:
            self.reply_label.config(text=f"请求失败：{error}", fg="#c00")
        else:
            self.reply_label.config(text=response or "（无回复）", fg="#111")
        self.input_text.focus_set()

    def run(self):
        if not ensure_server():
            self.root.withdraw()
            tk.messagebox.showerror("错误", "请先运行「卡若AI本地助手」或 server.py")
            self.root.destroy()
            return
        self.input_text.focus_set()
        self.root.mainloop()

# 卡若AI：打开 → 按两下 Fn 说话 → 自动出回复

import tkinter as tk
from tkinter import font as tkfont, messagebox
import json
import urllib.request
import subprocess
import re
import threading

API_URL = "http://localhost:5888"
APP_DIR = "/Users/karuo/Documents/个人/卡若AI/_共享模块/local_llm/app"
AUTO_MS = 1200

def ensure_server():
    try:
        urllib.request.urlopen(urllib.request.Request(f"{API_URL}/api/status"), timeout=2)
        return True
    except Exception:
        try:
            subprocess.Popen(["python3", f"{APP_DIR}/server.py"], cwd=APP_DIR, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
            import time
            time.sleep(3)
            urllib.request.urlopen(urllib.request.Request(f"{API_URL}/api/status"), timeout=5)
            return True
        except Exception:
            return False

def call_karuo(msg, voice=True):
    try:
        req = urllib.request.Request(
            f"{API_URL}/api/chat",
            data=json.dumps({"message": msg, "voice": voice}).encode("utf-8"),
            headers={"Content-Type": "application/json"}, method="POST",
        )
        with urllib.request.urlopen(req, timeout=60) as r:
            out = json.loads(r.read().decode())
            resp = out.get("response", out.get("error", "抱歉，出错了"))
            if voice and resp:
                subprocess.Popen(["say", "-v", "Tingting", re.sub(r"[*#`\[\]]", "", resp)[:500]])
            return resp
    except Exception as e:
        return f"请求失败：{str(e)}"


class App:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("卡若AI")
        self.root.geometry("360x180")
        self.root.minsize(300, 140)
        self._tid = None
        self._busy = False

        self.inp = tk.Entry(self.root, font=tkfont.Font(size=14), width=32)
        self.inp.pack(pady=14, padx=16, fill="x")
        self.inp.bind("<KeyRelease>", self._schedule_send)
        self.inp.focus_set()

        self.out = tk.Label(self.root, text="", font=tkfont.Font(size=12), wraplength=328, justify="left", fg="#111")
        self.out.pack(pady=6, padx=16, fill="both", expand=True)
        self.root.protocol("WM_DELETE_WINDOW", self.root.destroy)

    def _schedule_send(self, e=None):
        if self._busy:
            return
        if self._tid:
            self.root.after_cancel(self._tid)
        t = self.inp.get().strip()
        if not t:
            self._tid = None
            return
        self._tid = self.root.after(AUTO_MS, self._send)

    def _send(self):
        self._tid = None
        t = self.inp.get().strip()
        if not t or self._busy:
            return
        self._busy = True
        self.inp.delete(0, "end")
        self.out.config(text="…", fg="#666")

        def run():
            try:
                r = call_karuo(t, voice=True)
                self.root.after(0, lambda: self._done(r, None))
            except Exception as ex:
                self.root.after(0, lambda: self._done("", str(ex)))

        threading.Thread(target=run, daemon=True).start()

    def _done(self, resp, err):
        self._busy = False
        self.out.config(text=f"请求失败：{err}" if err else (resp or "（无回复）"), fg="#c00" if err else "#111")
        self.inp.focus_set()

    def run(self):
        if not ensure_server():
            messagebox.showerror("错误", "请先运行「卡若AI本地助手」")
            self.root.destroy()
            return
        self.inp.focus_set()
        self.root.mainloop()


if __name__ == "__main__":
    App().run()
