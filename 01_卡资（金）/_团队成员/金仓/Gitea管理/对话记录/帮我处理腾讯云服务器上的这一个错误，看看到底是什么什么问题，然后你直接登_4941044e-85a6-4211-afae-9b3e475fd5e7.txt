def scan_port(ip: str, port: int = 22, timeout: float = 2.0) -> Tuple[str, bool, Optional[str]]:
    """
    扫描单个IP的指定端口
    
    返回: (ip, is_open, banner)
    """
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex((ip, port))
        
        if result == 0:
            # 端口开放，尝试获取banner
            try:
                sock.send(b'\n')
                banner = sock.recv(1024).decode('utf-8', errors='ignore').strip()
            except:
                banner = "Unknown"
            sock.close()
            return (ip, True, banner)
        else:
            sock.close()
            return (ip, False, None)
    except socket.timeout:
        return (ip, False, None)
    except Exception as e:
        return (ip, False, None)

def try_login(target: str, port: int, username: str, password: str, timeout: int = 5) -> Tuple[bool, str]:
    """
