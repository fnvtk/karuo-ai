import os
import sys

TARGET_IP = "42.194.245.239"  # 存客宝公网IP


def main():
    secret_id = os.environ.get("TENCENTCLOUD_SECRET_ID") or "AKIDjc6yO3nPeOuK2OKsJPBBVbTiiz0aPNHl"
    secret_key = os.environ.get("TENCENTCLOUD_SECRET_KEY")
    if not secret_key:
        print("请设置 TENCENTCLOUD_SECRET_KEY 环境变量")
        print("腾讯云控制台: https://console.cloud.tencent.com/cam/capi")
        sys.exit(1)

    try:
        from tencentcloud.common import credential
        from tencentcloud.cvm.v20170312 import cvm_client, models as cvm_models
        from tencentcloud.vpc.v20170312 import vpc_client, models as vpc_models
    except ImportError:
        print("请安装: pip install tencentcloud-sdk-python-cvm tencentcloud-sdk-python-vpc")
        sys.exit(1)

    cred = credential.Credential(secret_id, secret_key)
    region = "ap-guangzhou"  # 华南广州，存客宝可能在此

    # 1. 查找实例（尝试多个地域）
    for reg in ["ap-guangzhou", "ap-shanghai", "ap-beijing"]:
        try:
            cvm = cvm_client.CvmClient(cred, reg)
            req = cvm_models.DescribeInstancesRequest()
            resp = cvm.DescribeInstances(req)
            for ins in (resp.InstanceSet or []):
                pub_ips = list(ins.PublicIpAddresses or [])
                if TARGET_IP in pub_ips:
                    sg_ids = list(ins.SecurityGroupIds or [])
                    print(f"找到实例: {ins.InstanceId} ({ins.InstanceName}) 地域:{reg}")
                    print(f"安全组: {sg_ids}")
                    if not sg_ids:
                        print("该实例无安全组，请检查")
                        sys.exit(1)
                    region = reg
                    break
            else:
                continue
            break
        except Exception as e:
            continue
    else:
        print(f"未找到公网IP为 {TARGET_IP} 的CVM实例")
        print("若为轻量应用服务器，需使用 Lighthouse API")
        sys.exit(1)

    # 2. 添加安全组入站规则（TCP 22）
    vpc = vpc_client.VpcClient(cred, region)
    for sg_id in sg_ids:
        req = vpc_models.CreateSecurityGroupPoliciesRequest()
        req.SecurityGroupId = sg_id
        req.SecurityGroupPolicySet = vpc_models.SecurityGroupPolicySet()
        req.SecurityGroupPolicySet.Ingress = [
            vpc_models.SecurityGroupPolicy(
                Protocol="TCP",
                Port="22",
                CidrBlock="0.0.0.0/0",
                Action="ACCEPT",
                PolicyDescription="SSH-存客宝开放"
            )
        ]
        try:
            vpc.CreateSecurityGroupPolicies(req)
            print(f"已为安全组 {sg_id} 添加 TCP:22 入站规则")
        except Exception as e:
            if "Duplicate" in str(e) or "already exists" in str(e).lower():
                print(f"安全组 {sg_id} 已存在22端口规则")
            else:
                print(f"添加规则失败: {e}")
                raise

    print("\n完成。请测试: ssh root@42.194.245.239")


if __name__ == "__main__":
    main()
