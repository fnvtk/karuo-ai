def 执行修复(info: dict, dry_run: bool = False):
    """修复：数据库内错误域名/协议替换；权限需 SSH 执行。"""
    dbname = info.get("dbname")
    tablepre = info.get("tablepre")
    if not dbname:
        print("无法获取数据库名，跳过数据库修复。")
        return

    # 常见错误形式：http://www.lkdie.com -> https://www.lkdie.com；http://lkdie.com -> https://www.lkdie.com
    replacements = [
        ("http://www.lkdie.com", "https://www.lkdie.com"),
        ("http://lkdie.com", "https://www.lkdie.com"),
        ("https://lkdie.com", "https://www.lkdie.com"),
    ]

    for old, new in replacements:
        # 更新 pre_forum_post.message
        sql = f"UPDATE {tablepre}forum_post SET message = REPLACE(message, '{old}', '{new}')"
        if dry_run:
            print(f"[dry-run] {sql[:80]}...")
            continue
        res = 宝塔POST("database?action=ExecuteSql", {"database": dbname, "sql": sql})
        if res.get("status") is True or res.get("data") is not None:
            print(f"  已替换帖子内容: {old} -> {new}")
        else:
            # 尝试 SqlExec
            res = 宝塔POST("database?action=SqlExec", {"database": dbname, "sql": sql})
            if res.get("status") is True or res.get("data") is not None:
                print(f"  已替换帖子内容: {old} -> {new}")
            else:
                print(f"  API 执行失败，请到宝塔 [数据库] 手动执行: {sql[:100]}...")

    # 若存在 pre_forum_attachment 等表中有 URL 字段也可同样 REPLACE（Discuz 附件表多为本地路径，可不改）
    print("\n[权限] 请在 kr宝塔 SSH 执行以下命令修正附件目录权限：")
    print("  chown -R www:www " + LKDIECOM_ROOT)
    print("  find " + LKDIECOM_ROOT + " -type d -exec chmod 755 {} \\;")
    print("  find " + LKDIECOM_ROOT + " -type f -exec chmod 644 {} \\;")

"""
lkdie.com 恢复站长账号「卡若」
================================
现象：UC 里还有卡若，论坛端被误删/顶掉，导致「该用户已被删除」、内容屏蔽、无法登录。
处理：查 UC 与论坛用户表，恢复 pre_common_member（及 status/count 等），保证帖子正常且可登录。
依赖：腾讯云数据库 lkdie_bbs 连接信息（账号与API索引 § 腾讯云数据库）。
"""
import os
import re
import sys

BASE = os.path.dirname(os.path.abspath(__file__))
IDX = os.path.normpath(os.path.join(BASE, "..", "..", "..", "金盾", "账号密码与资料管理", "账号与API索引.md"))
DB_NAME = "lkdie_bbs"
TABLE_PREFIX = "pre_"
目标用户名 = "卡若"


def 从账号索引读腾讯云数据库():
    if not os.path.isfile(IDX):
        return None
    with open(IDX, "r", encoding="utf-8") as f:
        c = f.read()
    m = re.search(r"###\s*腾讯云数据库\s*\n(.*?)(?=###|\Z)", c, re.DOTALL)
    if not m:
        return None
    b = m.group(1)
    addr = re.search(r"地址\s*\|\s*`([^`]+)`", b)
    user = re.search(r"账号\s*\|\s*`([^`]+)`", b)
    pw = re.search(r"密码\s*\|\s*`([^`]+)`", b)
    if not (addr and user and pw):
        return None
    host_port = addr.group(1).strip()
    if ":" in host_port:
        host, port = host_port.rsplit(":", 1)
        port = int(port)
    else:
        host, port = host_port, 3306
    return {
        "host": host.strip(),
        "port": port,
        "user": user.group(1).strip(),
        "password": pw.group(1).strip(),
        "database": DB_NAME,
    }


def 获取连接(conn_info):
    try:
        import pymysql
    except ImportError:
        print("请安装: pip install pymysql")
        return None
    return pymysql.connect(
        host=conn_info["host"],
        port=int(conn_info["port"]),
        user=conn_info["user"],
        password=conn_info["password"],
        database=conn_info.get("database") or DB_NAME,
        charset="utf8mb4",
        connect_timeout=10,
    )


def 查用户表名(conn):
    """返回 (uc_members 表名, common_member 表名)。"""
    with conn.cursor() as cur:
        cur.execute("SHOW TABLES")
        tables = [list(r)[0] for r in cur.fetchall()]
    uc_tbl = None
    cm_tbl = None
    for t in tables:
        if "ucenter" in t.lower() and "member" in t.lower():
            uc_tbl = t
        if t == TABLE_PREFIX + "common_member":
            cm_tbl = t
    return uc_tbl, cm_tbl or (TABLE_PREFIX + "common_member")


def 诊断(conn, dry_run: bool):
    uc_tbl, cm_tbl = 查用户表名(conn)
    print("UC 用户表:", uc_tbl or "未找到")
    print("论坛用户表:", cm_tbl)

    # 1) UC 里「卡若」的 uid
    uc_uid = None
    uc_row = None
    if uc_tbl:
        with conn.cursor() as cur:
            cur.execute(f"SELECT uid, username, email FROM `{uc_tbl}` WHERE username = %s", (目标用户名,))
            row = cur.fetchone()
            if row:
                uc_uid = row[0]
                uc_row = row
                print(f"\n[UC] 找到用户: uid={uc_uid}, username={row[1]}, email={row[2]}")
            else:
                cur.execute(f"SELECT uid, username FROM `{uc_tbl}` WHERE username LIKE %s", (f"%{目标用户名}%",))
                for r in cur.fetchall():
                    print(f"[UC] 相似: uid={r[0]}, username={r[1]}")
    if not uc_uid and uc_tbl:
        print("\n[UC] 未找到用户「卡若」，请确认用户名或是否在其它 UC 库。")
        return None

    # 2) 论坛 common_member 里该 uid 或 username=卡若
    with conn.cursor() as cur:
        cur.execute(f"SELECT uid, username, groupid FROM `{cm_tbl}` WHERE uid = %s OR username = %s", (uc_uid, 目标用户名))
        rows = cur.fetchall()
    if rows:
        for r in rows:
            print(f"[论坛 common_member] uid={r[0]}, username={r[1]}, groupid={r[2]}")
        # 若 uid 对应的是别人（username 不是卡若），说明被顶掉
        for r in rows:
            if r[0] == uc_uid and (r[1] or "").strip() != 目标用户名:
                print(f"  -> 被顶掉：uid {uc_uid} 当前 username 为「{r[1]}」，将恢复为「{目标用户名}`」")
    else:
        print(f"[论坛 common_member] 无 uid={uc_uid} 且无 username={目标用户名}，需从 UC 同步一条。")

    # 3) 帖子里 author=卡若 的 authorid
    with conn.cursor() as cur:
        cur.execute(
            f"SELECT DISTINCT authorid FROM {TABLE_PREFIX}forum_post WHERE author = %s LIMIT 5",
            (目标用户名,),
        )
        post_uids = [r[0] for r in cur.fetchall() if r[0]]
    if post_uids:
        print(f"\n[帖子] author=卡若 的 authorid: {post_uids}")
