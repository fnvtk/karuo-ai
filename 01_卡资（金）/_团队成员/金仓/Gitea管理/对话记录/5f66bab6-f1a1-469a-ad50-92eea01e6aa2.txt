### 为什么MySQL会启动失败？

1. **OOM Killer**：Linux内存不足时会杀死占用内存大的进程，MySQL可能被杀
2. **MySQL启动需要内存**：启动时需要分配缓冲池，当前可用内存不足
3. **恶性循环**：MySQL被杀 → 重启失败 → 业务受影响

def open_ssh_for_instance(cred, vpc_client_module, target_ip: str, label: str, regions: list):
    """根据公网 IP 查找 CVM 实例，为其安全组添加 TCP 22 入站规则。"""
    sg_ids = []
    region_used = None
    for reg in regions:
        try:
            from tencentcloud.cvm.v20170312 import cvm_client, models as cvm_models
            cvm = cvm_client.CvmClient(cred, reg)
            req = cvm_models.DescribeInstancesRequest()
            resp = cvm.DescribeInstances(req)
            for ins in (resp.InstanceSet or []):
                pub_ips = list(ins.PublicIpAddresses or [])
                if target_ip in pub_ips:
                    sg_ids = list(ins.SecurityGroupIds or [])
                    region_used = reg
                    print(f"  [{label}] 找到实例: {ins.InstanceId} ({ins.InstanceName}) 地域:{reg}, 安全组: {sg_ids}")
                    if not sg_ids:
                        print(f"  [{label}] 该实例无安全组，跳过")
                    break
            if sg_ids:
                break
        except Exception as e:
            continue
    if not sg_ids or not region_used:
        print(f"  [{label}] 未找到公网IP {target_ip} 的CVM实例（若为轻量需用 Lighthouse API）")
        return False

    vpc = vpc_client_module.VpcClient(cred, region_used)
    for sg_id in sg_ids:
        req = vpc_models.CreateSecurityGroupPoliciesRequest()
        req.SecurityGroupId = sg_id
        req.SecurityGroupPolicySet = vpc_models.SecurityGroupPolicySet()
        req.SecurityGroupPolicySet.Ingress = [
            vpc_models.SecurityGroupPolicy(
                Protocol="TCP",
                Port="22",
                CidrBlock="0.0.0.0/0",
                Action="ACCEPT",
                PolicyDescription=f"SSH-{label}"
            )
        ]
        try:
            vpc.CreateSecurityGroupPolicies(req)
            print(f"  [{label}] 已为安全组 {sg_id} 添加 TCP:22 入站规则")
        except Exception as e:
            if "Duplicate" in str(e) or "already exists" in str(e).lower():
                print(f"  [{label}] 安全组 {sg_id} 已存在 22 端口规则")
            else:
                print(f"  [{label}] 添加规则失败: {e}")
                return False
    return True
