 * 知己 · 分销系统核心 API

import { NextRequest, NextResponse } from "next/server"
import { connectToDatabase, COLLECTIONS } from "@/lib/mongodb"

// ============================================================
// 分销系统常量
// ============================================================

/** 默认分销配置（后台可配置覆盖） */
const DEFAULT_DISTRIBUTION_CONFIG = {
  enabled: true,                  // 分销功能开关
  commissionRate: 0.10,           // 一级分销佣金比例（10%）
  bindingDays: 30,                // 绑定有效期（天）
  minWithdrawAmount: 10,          // 最低提现金额（元）
  autoApprove: true,              // 是否自动审核分销员申请
  maxLevel: 1,                    // 最大分销层级（只做一级）
  settlementDays: 7,              // 结算周期（天）- 订单完成后N天可提现
}

/** 获取分销配置（从数据库读取，降级使用默认值） */
async function getDistributionConfig(db: any) {
  try {
    const settingsDoc = await db.collection(COLLECTIONS.SYSTEM_SETTINGS).findOne({ key: "distribution_config" })
    return settingsDoc?.config ? { ...DEFAULT_DISTRIBUTION_CONFIG, ...settingsDoc.config } : DEFAULT_DISTRIBUTION_CONFIG
  } catch {
    return DEFAULT_DISTRIBUTION_CONFIG
  }
}

// ============================================================
// GET - 查询分销信息
// ============================================================

export async function GET(req: NextRequest) {
  try {
    const { searchParams } = new URL(req.url)
    const action = searchParams.get("action") // info/stats/orders/team/config
    const userId = searchParams.get("userId")

    const { db } = await connectToDatabase()
    const config = await getDistributionConfig(db)

    // 获取分销配置（公开部分）
    if (action === "config") {
      return NextResponse.json({
        success: true,
        data: {
          enabled: config.enabled,
          commissionRate: config.commissionRate,
          bindingDays: config.bindingDays,
          minWithdrawAmount: config.minWithdrawAmount,
        },
      })
    }

    if (!userId) {
      return NextResponse.json({ success: false, error: "缺少 userId" }, { status: 400 })
    }

    const distributionCol = db.collection(COLLECTIONS.DISTRIBUTION_USERS)
    const ordersCol = db.collection(COLLECTIONS.DISTRIBUTION_ORDERS)
    const bindingsCol = db.collection(COLLECTIONS.REFERRAL_BINDINGS)

    // 获取分销员信息
    if (action === "info") {
      const distributor = await distributionCol.findOne({ userId })
      if (!distributor) {
        return NextResponse.json({ success: true, data: { isDistributor: false } })
      }
      return NextResponse.json({
        success: true,
        data: {
          isDistributor: true,
          distributor: {
            userId: distributor.userId,
            nickname: distributor.nickname,
            phone: distributor.phone,
            inviteCode: distributor.inviteCode,
            level: distributor.level,
            status: distributor.status,
            totalEarnings: distributor.totalEarnings || 0,
            availableBalance: distributor.availableBalance || 0,
            withdrawnAmount: distributor.withdrawnAmount || 0,
            totalTeam: distributor.totalTeam || 0,
            totalOrders: distributor.totalOrders || 0,
            createdAt: distributor.createdAt,
          },
        },
      })
    }

    // 获取分销统计
    if (action === "stats") {
      const distributor = await distributionCol.findOne({ userId })
      if (!distributor) {
        return NextResponse.json({ success: false, error: "非分销员" }, { status: 403 })
      }

      // 今日数据
      const todayStart = new Date()
      todayStart.setHours(0, 0, 0, 0)
      const todayStr = todayStart.toISOString()

      const todayOrders = await ordersCol.countDocuments({
        distributorId: userId,
        createdAt: { $gte: todayStr },
      })
      const todayEarningsResult = await ordersCol.aggregate([
        { $match: { distributorId: userId, createdAt: { $gte: todayStr }, status: { $in: ["settled", "pending"] } } },
        { $group: { _id: null, total: { $sum: "$commissionAmount" } } },
      ]).toArray()

      // 本月数据
      const monthStart = new Date()
      monthStart.setDate(1)
      monthStart.setHours(0, 0, 0, 0)
      const monthStr = monthStart.toISOString()

      const monthOrders = await ordersCol.countDocuments({
        distributorId: userId,
        createdAt: { $gte: monthStr },
      })
      const monthEarningsResult = await ordersCol.aggregate([
        { $match: { distributorId: userId, createdAt: { $gte: monthStr }, status: { $in: ["settled", "pending"] } } },
        { $group: { _id: null, total: { $sum: "$commissionAmount" } } },
      ]).toArray()

      // 绑定用户数（当前有效绑定）
      const activeBindings = await bindingsCol.countDocuments({
        referrerId: userId,
        status: "active",
        expiryDate: { $gt: new Date().toISOString() },
      })

      return NextResponse.json({
        success: true,
        data: {
          totalEarnings: distributor.totalEarnings || 0,
          availableBalance: distributor.availableBalance || 0,
          withdrawnAmount: distributor.withdrawnAmount || 0,
          todayEarnings: todayEarningsResult[0]?.total || 0,
          todayOrders,
          monthEarnings: monthEarningsResult[0]?.total || 0,
          monthOrders,
          totalTeam: distributor.totalTeam || 0,
          activeBindings,
          totalOrders: distributor.totalOrders || 0,
        },
      })
    }

    // 获取分销订单列表
    if (action === "orders") {
      const page = parseInt(searchParams.get("page") || "1")
      const limit = parseInt(searchParams.get("limit") || "20")
      const status = searchParams.get("status") // pending/settled/cancelled

      const query: Record<string, unknown> = { distributorId: userId }
      if (status) query.status = status

      const orders = await ordersCol
        .find(query)
        .sort({ createdAt: -1 })
        .skip((page - 1) * limit)
        .limit(limit)
        .toArray()

      const total = await ordersCol.countDocuments(query)

      return NextResponse.json({
        success: true,
        data: { list: orders, total, page, limit },
      })
    }

    // 获取团队列表（我绑定的用户）
    if (action === "team") {

    if (action === "register") {

          return NextResponse.json({ success: false, error: "已经是分销员了" })
        }
        if (existing.status === "pending") {
          return NextResponse.json({ success: false, error: "申请审核中" })
        }
      }

      // 生成唯一邀请码（6位大写字母+数字）
      const inviteCode = generateInviteCode()

      const now = new Date().toISOString()
      const newDistributor = {
        userId,
        nickname: nickname || "分销员",
        phone: phone || "",
        avatar: avatar || "",
        inviteCode,
        level: 1,                     // 分销等级（默认1级）
        status: config.autoApprove ? "active" : "pending",  // 自动审核或待审核
        parentId: null,               // 上级分销员ID（一级分销暂不使用）
        totalEarnings: 0,             // 累计收益
        availableBalance: 0,          // 可提现余额
        withdrawnAmount: 0,           // 已提现金额
        frozenAmount: 0,              // 冻结金额（结算中）
        totalTeam: 0,                 // 团队总人数
        totalOrders: 0,               // 总订单数
        commissionRate: config.commissionRate, // 佣金比例（可个性化配置）
        createdAt: now,
        updatedAt: now,
      }

      await distributionCol.insertOne(newDistributor)

      return NextResponse.json({
        success: true,
        data: {
          inviteCode,
          status: newDistributor.status,
          commissionRate: config.commissionRate,
          message: config.autoApprove ? "恭喜！已成为分销员" : "申请已提交，等待审核",
        },
      })
    }

    // ========== 记录佣金（被推荐人完成测评/购买后触发） ==========
    if (action === "record-commission") {
      const { refereeId, orderId, orderAmount, orderType } = body
      // orderType: "test_purchase" | "consultant_booking" | "vip_purchase"

      if (!refereeId || !orderId || !orderAmount) {
        return NextResponse.json({ success: false, error: "缺少必要参数" }, { status: 400 })
      }

      const bindingsCol = db.collection(COLLECTIONS.REFERRAL_BINDINGS)
      const distributionCol = db.collection(COLLECTIONS.DISTRIBUTION_USERS)
      const ordersCol = db.collection(COLLECTIONS.DISTRIBUTION_ORDERS)

      // 查找有效绑定关系
      const binding = await bindingsCol.findOne({

        status: "active",
        expiryDate: { $gt: new Date().toISOString() },
      })

      if (!binding) {
        return NextResponse.json({ success: true, data: { hasDistributor: false, message: "无有效推荐关系" } })
      }

      // 检查推荐人是否是活跃分销员
      const distributor = await distributionCol.findOne({
        userId: binding.referrerId,
        status: "active",
      })

      if (!distributor) {
        return NextResponse.json({ success: true, data: { hasDistributor: false, message: "推荐人非活跃分销员" } })
      }

      // 检查是否重复记录
      const existingOrder = await ordersCol.findOne({ orderId })
      if (existingOrder) {
        return NextResponse.json({ success: false, error: "佣金已记录" })
      }

      // 计算佣金
      const commissionRate = distributor.commissionRate || config.commissionRate
      const commissionAmount = Math.round(orderAmount * commissionRate * 100) / 100

      const now = new Date().toISOString()

      // 创建佣金记录
      await ordersCol.insertOne({
        distributorId: binding.referrerId,
        refereeId,
        orderId,
        orderAmount,
        orderType: orderType || "test_purchase",
        commissionRate,
        commissionAmount,
        status: "pending",       // pending → settled → withdrawn / cancelled

        settlementDate: new Date(Date.now() + config.settlementDays * 24 * 60 * 60 * 1000).toISOString(),
        createdAt: now,
        updatedAt: now,
      })

      await distributionCol.updateOne(
        { userId: binding.referrerId },
        {
          $inc: {
            totalEarnings: commissionAmount,
            frozenAmount: commissionAmount,  // 先冻结，结算后释放
            totalOrders: 1,
          },
          $set: { updatedAt: now },
        }
      )

      // 更新绑定关系的佣金累计
      await bindingsCol.updateOne(
        { _id: binding._id },
        {
          $inc: { commissionAmount },
          $set: { conversionDate: now, orderId, updatedAt: now },
        }
      )

      return NextResponse.json({
        success: true,
        data: {
          hasDistributor: true,
          distributorId: binding.referrerId,
          commissionAmount,
          commissionRate,
          message: `佣金 ¥${commissionAmount} 已记录`,
        },
      })
    }

    // ========== 申请提现 ==========
    if (action === "withdraw") {
      const { userId, amount, withdrawMethod, accountInfo } = body
      // withdrawMethod: "wechat" | "alipay" | "bank"
      // accountInfo: { name, account, bank? }

      if (!userId || !amount) {
        return NextResponse.json({ success: false, error: "缺少必要参数" }, { status: 400 })
      }

      const distributionCol = db.collection(COLLECTIONS.DISTRIBUTION_USERS)
      const withdrawalCol = db.collection(COLLECTIONS.DISTRIBUTION_WITHDRAWALS)

      const distributor = await distributionCol.findOne({ userId, status: "active" })
      if (!distributor) {
        return NextResponse.json({ success: false, error: "非活跃分销员" }, { status: 403 })
      }

      if (amount < config.minWithdrawAmount) {
        return NextResponse.json({ success: false, error: `最低提现金额 ¥${config.minWithdrawAmount}` }, { status: 400 })
      }

      if (amount > (distributor.availableBalance || 0)) {
        return NextResponse.json({ success: false, error: "可提现余额不足" }, { status: 400 })
      }

      const now = new Date().toISOString()

      // 创建提现申请
      await withdrawalCol.insertOne({
        distributorId: userId,
        amount,
        withdrawMethod: withdrawMethod || "wechat",
        accountInfo: accountInfo || {},
        status: "pending",  // pending → approved → paid / rejected
        createdAt: now,
        updatedAt: now,
      })

      // 扣减可提现余额
      await distributionCol.updateOne(
        { userId },
        {
          $inc: { availableBalance: -amount, frozenAmount: amount },
          $set: { updatedAt: now },
        }
      )

      return NextResponse.json({
        success: true,
        data: { message: `提现申请 ¥${amount} 已提交，等待审核` },
      })
    }

    return NextResponse.json({ success: false, error: "无效的 action" }, { status: 400 })
  } catch (error: unknown) {
    console.error("[知己] 分销操作失败:", error)
    return NextResponse.json({ success: false, error: "操作失败" }, { status: 500 })
  }
}

/** 生成6位邀请码 */
function generateInviteCode(): string {
  const chars = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"
  let code = ""
  for (let i = 0; i < 6; i++) {
    code += chars.charAt(Math.floor(Math.random() * chars.length))
  }
  return code
}

// ============================================================
// POST - 管理后台操作
// ============================================================

export async function POST(req: NextRequest) {
  try {
    const body = await req.json()
    const { action } = body

    const { db } = await connectToDatabase()
    const now = new Date().toISOString()

    // ========== 审核分销员 ==========
    if (action === "approve-distributor") {
      const { distributorId, approved } = body
      if (!distributorId) {
        return NextResponse.json({ success: false, error: "缺少分销员ID" }, { status: 400 })
      }

      await db.collection(COLLECTIONS.DISTRIBUTION_USERS).updateOne(
        { userId: distributorId },
        { $set: { status: approved ? "active" : "rejected", updatedAt: now } }
      )

      return NextResponse.json({ success: true, data: { message: approved ? "已通过" : "已拒绝" } })
    }

    // ========== 审核提现 ==========
    if (action === "approve-withdrawal") {
      const { withdrawalId, approved, remark } = body
      if (!withdrawalId) {
        return NextResponse.json({ success: false, error: "缺少提现ID" }, { status: 400 })
      }

      const { ObjectId } = await import("mongodb")
      const withdrawalCol = db.collection(COLLECTIONS.DISTRIBUTION_WITHDRAWALS)
      const distributorCol = db.collection(COLLECTIONS.DISTRIBUTION_USERS)

      const withdrawal = await withdrawalCol.findOne({ _id: new ObjectId(withdrawalId) })
      if (!withdrawal) {
        return NextResponse.json({ success: false, error: "提现记录不存在" }, { status: 404 })
      }

      if (approved) {
        // 审核通过 → 标记为已支付
        await withdrawalCol.updateOne(
          { _id: new ObjectId(withdrawalId) },
          { $set: { status: "paid", remark: remark || "", paidAt: now, updatedAt: now } }
        )
        // 更新分销员：冻结金额减少，已提现金额增加
        await distributorCol.updateOne(
          { userId: withdrawal.distributorId },
          {
            $inc: { frozenAmount: -withdrawal.amount, withdrawnAmount: withdrawal.amount },
            $set: { updatedAt: now },
          }
        )
      } else {
        // 拒绝 → 释放冻结金额回可提现
        await withdrawalCol.updateOne(
          { _id: new ObjectId(withdrawalId) },
          { $set: { status: "rejected", remark: remark || "", updatedAt: now } }
        )
        await distributorCol.updateOne(
          { userId: withdrawal.distributorId },
          {
            $inc: { frozenAmount: -withdrawal.amount, availableBalance: withdrawal.amount },
            $set: { updatedAt: now },
          }
        )
      }

      return NextResponse.json({ success: true, data: { message: approved ? "提现已审核通过" : "提现已拒绝" } })
    }

    // ========== 结算佣金（把pending的佣金结算为可提现） ==========
    if (action === "settle-commissions") {
      const ordersCol = db.collection(COLLECTIONS.DISTRIBUTION_ORDERS)
      const distributorCol = db.collection(COLLECTIONS.DISTRIBUTION_USERS)

      // 查找所有已到结算日期的pending订单
      const pendingOrders = await ordersCol.find({
        status: "pending",
        settlementDate: { $lte: now },
      }).toArray()

      if (pendingOrders.length === 0) {
        return NextResponse.json({ success: true, data: { settled: 0, message: "没有待结算的佣金" } })
      }

      // 按分销员分组结算
      const distributorSettlements: Record<string, number> = {}
      const orderIds: any[] = []

      for (const order of pendingOrders) {
        const amount = order.commissionAmount || 0
        distributorSettlements[order.distributorId] = (distributorSettlements[order.distributorId] || 0) + amount
        orderIds.push(order._id)
      }

      // 批量更新订单状态
      await ordersCol.updateMany(
        { _id: { $in: orderIds } },
        { $set: { status: "settled", settledAt: now, updatedAt: now } }
      )

      // 更新每个分销员的余额：冻结减少，可提现增加
      for (const [distributorId, amount] of Object.entries(distributorSettlements)) {
        await distributorCol.updateOne(
          { userId: distributorId },
          {
            $inc: { frozenAmount: -amount, availableBalance: amount },
            $set: { updatedAt: now },
          }
        )
      }

      return NextResponse.json({
        success: true,
        data: {
          settled: pendingOrders.length,
          totalAmount: Object.values(distributorSettlements).reduce((a, b) => a + b, 0),
          message: `已结算 ${pendingOrders.length} 笔佣金`,
        },
      })
    }

    // ========== 更新分销配置 ==========
    if (action === "update-config") {
      const { config } = body
      if (!config) {
        return NextResponse.json({ success: false, error: "缺少配置数据" }, { status: 400 })
      }

      await db.collection(COLLECTIONS.SYSTEM_SETTINGS).updateOne(
        { key: "distribution_config" },
        {
          $set: {
            key: "distribution_config",
            config: { ...DEFAULT_CONFIG, ...config },
            updatedAt: now,
          },
        },
        { upsert: true }
      )

      return NextResponse.json({ success: true, data: { message: "分销配置已更新" } })
    }

    return NextResponse.json({ success: false, error: "无效的 action" }, { status: 400 })
  } catch (error: unknown) {
    console.error("[知己Admin] 分销操作失败:", error)
    return NextResponse.json({ success: false, error: "操作失败" }, { status: 500 })
  }
}

function WithdrawSection({
  availableBalance,
  userId,
  onSuccess,
}: {
  availableBalance: number
  userId: string
  onSuccess: () => void
}) {
  const [amount, setAmount] = useState("")
  const [withdrawMethod, setWithdrawMethod] = useState<"wechat" | "alipay">("wechat")
  const [submitting, setSubmitting] = useState(false)

  const handleWithdraw = async () => {
    const num = parseFloat(amount)
    if (!num || num < 10) {
      alert("最低提现金额 ¥10")
      return
    }
    if (num > availableBalance) {
      alert("超过可提现余额")
      return
    }

    setSubmitting(true)
    try {
      const res = await fetch("/api/distribution", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          action: "withdraw",
          userId,
          amount: num,
          withdrawMethod,
        }),
      })
      const data = await res.json()
      if (data.success) {
        alert(data.data.message)
        setAmount("")
        onSuccess()
      } else {
        alert(data.error || "提现失败")
      }
    } catch {
      alert("网络错误")
    } finally {
      setSubmitting(false)
    }
  }

  return (
    <div className="space-y-3">
      {/* 可提现余额 */}
      <Card className="glass border-0 rounded-xl">
        <div className="p-5 text-center">
          <p className="text-sm text-muted-foreground mb-1">可提现余额</p>
          <p className="text-3xl font-bold text-foreground">¥{availableBalance.toFixed(2)}</p>
        </div>

              placeholder="请输入提现金额"
              className="w-full h-14 pl-10 pr-4 text-xl font-bold rounded-xl bg-white/50 border border-orange-100 focus:outline-none focus:border-coral/50 text-foreground"
            />
          </div>
          <div className="flex items-center justify-between mt-2">
            <span className="text-xs text-muted-foreground">最低提现 ¥10</span>
            <button onClick={() => setAmount(String(availableBalance))} className="text-xs text-coral font-medium">
              全部提现
            </button>
          </div>
        </div>
      </Card>

      {/* 提现方式 */}
      <Card className="glass border-0 rounded-xl">
        <div className="p-4">

            <Wallet className="w-4 h-4 text-coral" />
            提现方式
          </h3>
          <div className="space-y-2">
            {[
              { key: "wechat" as const, label: "微信钱包", desc: "提现到微信零钱" },
              { key: "alipay" as const, label: "支付宝", desc: "提现到支付宝账户" },
            ].map((method) => (

function ConfigTab() {
  const [config, setConfig] = useState<DistributionConfig | null>(null)
  const [loading, setLoading] = useState(true)
  const [saving, setSaving] = useState(false)
  const [saved, setSaved] = useState(false)

  useEffect(() => { fetchConfig() }, [])

  const fetchConfig = async () => {
    try {
      const res = await fetch("/api/admin/distribution?action=config")
      const result = await res.json()
      if (result.success) setConfig(result.data)
    } catch (err) { console.error("获取配置失败:", err) }
    finally { setLoading(false) }
  }

  const handleSave = async () => {
    if (!config) return
    setSaving(true)
    try {
      const res = await fetch("/api/admin/distribution", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ action: "update-config", config }),
      })
      const result = await res.json()
      if (result.success) {
        setSaved(true)
        setTimeout(() => setSaved(false), 2000)
      }
    } catch { alert("保存失败") }
    finally { setSaving(false) }
  }

  if (loading || !config) {
    return (
      <div className="space-y-4">
        {[1, 2, 3, 4].map((i) => (
          <div key={i} className="bg-white/80 rounded-2xl p-6 animate-pulse border border-orange-100">
            <div className="h-4 bg-orange-100 rounded w-32 mb-4" />
            <div className="h-10 bg-orange-100 rounded w-full" />
          </div>
        ))}
      </div>
    )
  }

  /** 配置项行 */
  const ConfigRow = ({ label, desc, children }: { label: string; desc?: string; children: React.ReactNode }) => (
    <div className="flex items-center justify-between py-4 border-b border-orange-50 last:border-0">
      <div className="flex-1 mr-6">
        <div className="text-sm font-medium text-foreground">{label}</div>
        {desc && <div className="text-xs text-muted-foreground mt-0.5">{desc}</div>}
      </div>
      <div className="flex-shrink-0">{children}</div>
    </div>
  )

  return (
    <div className="space-y-6">
      {/* 保存按钮 */}
      <div className="flex justify-end">
        <button
          onClick={handleSave}
          disabled={saving}
          className="flex items-center gap-2 px-4 py-2 bg-coral text-white rounded-xl hover:bg-coral/90 transition-colors text-sm disabled:opacity-50"
        >
          {saving ? <RefreshCw className="w-4 h-4 animate-spin" /> : <Save className="w-4 h-4" />}
          {saved ? "已保存 ✓" : "保存配置"}
        </button>
      </div>

      {/* 基础设置 */}
      <div className="bg-white/80 backdrop-blur rounded-2xl border border-orange-100 p-6">
        <h2 className="text-lg font-semibold text-foreground mb-4 flex items-center gap-2">
          <Settings className="w-5 h-5 text-coral" />
          基础设置
        </h2>
        <ConfigRow label="分销功能开关" desc="关闭后前端将隐藏分销入口">
          <button
            onClick={() => setConfig({ ...config, enabled: !config.enabled })}
            className={`relative w-12 h-6 rounded-full transition-colors ${config.enabled ? "bg-coral" : "bg-gray-300"}`}
          >
            <span className={`absolute top-0.5 w-5 h-5 rounded-full bg-white transition-transform shadow-sm ${config.enabled ? "left-6" : "left-0.5"}`} />
          </button>
        </ConfigRow>
        <ConfigRow label="自动审核" desc="分销员申请是否自动通过">
          <button
            onClick={() => setConfig({ ...config, autoApprove: !config.autoApprove })}
            className={`relative w-12 h-6 rounded-full transition-colors ${config.autoApprove ? "bg-coral" : "bg-gray-300"}`}

            <span className={`absolute top-0.5 w-5 h-5 rounded-full bg-white transition-transform shadow-sm ${config.autoApprove ? "left-6" : "left-0.5"}`} />
          </button>

      </div>

        <h2 className="text-lg font-semibold text-foreground mb-4 flex items-center gap-2">
          <Coins className="w-5 h-5 text-gold" />
          佣金设置
        </h2>
        <ConfigRow label="佣金比例" desc="分销员获得的佣金百分比">
