  private xmlToDict(xml: string): Record<string, string> {
    const result: Record<string, string> = {};
    const cdataRegex = /<(\w+)><!\[CDATA\[(.*?)\]\]><\/\1>/g;
    let match;
    while ((match = cdataRegex.exec(xml)) !== null) {
      result[match[1]] = match[2];
    }
    const simpleRegex = /<(\w+)>([^<]*)<\/\1>/g;
    while ((match = simpleRegex.exec(xml)) !== null) {
      if (!result[match[1]]) result[match[1]] = match[2];
    }
    return result;
  }

  private dictToXml(data: Record<string, string>): string {
    const xml = ['<xml>'];
    for (const [key, value] of Object.entries(data)) {
      xml.push(`<${key}><![CDATA[${value}]]></${key}>`);
    }
    xml.push('</xml>');
    return xml.join('');
  }

 * 前端轮询此接口检查支付是否成功
 */

import { NextRequest, NextResponse } from 'next/server';
import { getCollection, COLLECTIONS } from '@/lib/mongodb';
import { PaymentFactory } from '@/lib/payment';
import '@/lib/payment/wechat';
import '@/lib/payment/alipay';

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const orderSn = searchParams.get('orderSn');

    if (!orderSn) {
      return NextResponse.json(
        { code: 400, message: '缺少参数: orderSn', data: null },
        { status: 400 }
      );
    }

    // ============================================================
    // 1. 查询本地订单状态
    // ============================================================
    const ordersCol = await getCollection(COLLECTIONS.ORDERS);
    const order = await ordersCol.findOne({ orderSn });

    if (!order) {
      return NextResponse.json(
        { code: 404, message: '订单不存在', data: null },
        { status: 404 }
      );
    }

    // 如果订单已经是终态，直接返回
    if (['paid', 'closed', 'refunded', 'expired'].includes(order.status)) {
      return NextResponse.json({
        code: 200,
        message: '查询成功',
        data: {
          orderSn: order.orderSn,
          status: order.status,
          amount: order.payAmount,
          productType: order.productType,
          paidAt: order.paidAt,
          transactionId: order.transactionId,
        },
      });
    }

    // ============================================================
    // 2. 主动查询支付平台（补偿机制）
    // ============================================================
    if (order.status === 'paying' && order.paymentMethod) {
      try {
        // 查找对应的交易流水
        const tradesCol = await getCollection(COLLECTIONS.PAY_TRADES);
        const trade = await tradesCol.findOne({ orderSn, status: 'paying' });

        if (trade) {
          const gateway = PaymentFactory.create(order.paymentMethod);
          const queryResult = await gateway.queryTrade(trade.tradeSn);

          if (queryResult && queryResult.status === 'paid') {
            // 支付平台已确认支付，更新本地状态
            await ordersCol.updateOne(
              { orderSn },
              {
                $set: {
                  status: 'paid',
                  transactionId: queryResult.platformSn,
                  paidAt: queryResult.payTime,
                  updatedAt: new Date(),
                },
              }
            );

            await tradesCol.updateOne(
              { tradeSn: trade.tradeSn },
              {
                $set: {
                  status: 'paid',
                  platformSn: queryResult.platformSn,
                  payTime: queryResult.payTime,
                  updatedAt: new Date(),
                },
              }
            );

            return NextResponse.json({
              code: 200,
              message: '支付成功',
              data: {
                orderSn: order.orderSn,
                status: 'paid',
                amount: order.payAmount,
                productType: order.productType,
                paidAt: queryResult.payTime,
                transactionId: queryResult.platformSn,
              },
            });
          }
        }
      } catch (queryError) {
        console.log('[知己·支付查询] 主动查询失败:', queryError);
      }
    }

    // ============================================================
    // 3. 检查订单是否过期
    // ============================================================
    if (order.expiredAt && new Date() > new Date(order.expiredAt)) {
      await ordersCol.updateOne(
        { orderSn, status: { $in: ['created', 'paying'] } },
        { $set: { status: 'expired', updatedAt: new Date() } }
      );

      return NextResponse.json({
        code: 200,
        message: '订单已过期',
        data: {
          orderSn: order.orderSn,
          status: 'expired',
          amount: order.payAmount,
          productType: order.productType,
        },
      });
    }

    // 返回当前状态（仍在支付中）
    return NextResponse.json({
      code: 200,
      message: '等待支付',
      data: {
        orderSn: order.orderSn,
        status: order.status,
        amount: order.payAmount,
        productType: order.productType,
      },
    });
  } catch (error) {
    console.error('[知己·支付查询] 异常:', error);
    return NextResponse.json(
      { code: 500, message: '查询失败', data: null },
      { status: 500 }
    );
  }
}

/** 申请成为分销员 */
export async function applyDistributor(userId: string, parentInviteCode?: string): Promise<Distributor> {
  const col = await getCollection(COLLECTIONS.DISTRIBUTION_USERS);

  // 检查是否已申请
  const existing = await col.findOne({ userId });
  if (existing) {
    return existing as unknown as Distributor;
  }

  // 查找上级
  let parentId: string | undefined;
  if (parentInviteCode) {
    const parent = await col.findOne({ inviteCode: parentInviteCode, status: 'active' });
    if (parent) {
      parentId = parent.userId;
    }
  }

  const inviteCode = generateInviteCode();
  const now = new Date();

  const distributor: Distributor = {
    userId,
    inviteCode,
    level: 1,
    status: 'active', // 默认自动通过，可改为 'pending' 需审核
    parentId,
    totalEarnings: 0,
    pendingEarnings: 0,
    withdrawnAmount: 0,
    directCount: 0,
    indirectCount: 0,
    createdAt: now,
    updatedAt: now,
  };

  await col.insertOne(distributor);

  // 更新上级的团队统计
  if (parentId) {
    await col.updateOne(
      { userId: parentId },
      { $inc: { directCount: 1 }, $set: { updatedAt: now } }
    );

    // 更新上上级的间推统计
    const parent = await col.findOne({ userId: parentId });
    if (parent?.parentId) {
      await col.updateOne(
        { userId: parent.parentId },
        { $inc: { indirectCount: 1 }, $set: { updatedAt: now } }
      );
    }
  }

  console.log(`[知己·分销] 新增分销员: ${userId}, 邀请码: ${inviteCode}`);
  return distributor;
}

/**
 * 绑定推荐关系（30天有效期）
 * 规则：
 * 1. 首次点击绑定，30天内有效
 * 2. 已有有效绑定则不覆盖（除非过期）
 * 3. 不能自己推荐自己
 */
export async function bindReferral(refereeId: string, referrerInviteCode: string): Promise<boolean> {
  try {
    const config = getDistributionConfig();
    if (!config.enabled) return false;

    // 查找推荐人
    const referrer = await getDistributorByCode(referrerInviteCode);
    if (!referrer) return false;

    // 不能自己推荐自己
    if (referrer.userId === refereeId) return false;

    const bindingsCol = await getCollection(COLLECTIONS.REFERRAL_BINDINGS);
    const now = new Date();

    // 检查是否已有有效绑定
    const existingBinding = await bindingsCol.findOne({
      refereeId,
      status: 'active',
      expiryDate: { $gt: now },
    });

    if (existingBinding) {
      console.log(`[知己·分销] 已有有效绑定，跳过: ${refereeId}`);
      return false;
    }

    // 创建新绑定
    const expiryDate = new Date(now.getTime() + config.bindingDays * 24 * 60 * 60 * 1000);

    await bindingsCol.insertOne({
      referrerId: referrer.userId,
      refereeId,
      inviteCode: referrerInviteCode,
      status: 'active',
      bindingDate: now,
      expiryDate,
      createdAt: now,
      updatedAt: now,
    });

    console.log(`[知己·分销] 绑定成功: ${refereeId} → ${referrer.userId}, 到期: ${expiryDate}`);
    return true;
  } catch (error) {
    console.error('[知己·分销] 绑定失败:', error);
    return false;
  }
}

/**
 * 计算并创建佣金记录
 * 在支付回调中调用
 */
export async function calculateCommission(
  orderSn: string,
  buyerUserId: string,
  orderAmount: number, // 分
): Promise<void> {
  try {
    const config = getDistributionConfig();
    if (!config.enabled) return;

    // 查找有效推荐人
    const referrerId = await findActiveReferrer(buyerUserId);
    if (!referrerId) return;

    const distOrdersCol = await getCollection(COLLECTIONS.DISTRIBUTION_ORDERS);
    const distUsersCol = await getCollection(COLLECTIONS.DISTRIBUTION_USERS);
    const now = new Date();

    // 防重复
    const existing = await distOrdersCol.findOne({ orderSn, distributorId: referrerId });
    if (existing) return;

    // 一级佣金
    const level1Amount = Math.floor(orderAmount * config.level1Rate);
    if (level1Amount > 0) {
      const commission: CommissionRecord = {
        distributorId: referrerId,
        orderId: orderSn,
        orderSn,
        refereeId: buyerUserId,
        level: 1,
        orderAmount,
        commissionRate: config.level1Rate,
        commissionAmount: level1Amount,
        status: 'pending',
        createdAt: now,
        updatedAt: now,
      };
      await distOrdersCol.insertOne(commission);
      await distUsersCol.updateOne(
        { userId: referrerId },
        { $inc: { pendingEarnings: level1Amount, totalEarnings: level1Amount }, $set: { updatedAt: now } }
      );
      console.log(`[知己·分销] 一级佣金: ${referrerId} +${level1Amount}分`);
    }

    // 二级佣金
    const referrer = await distUsersCol.findOne({ userId: referrerId });
    if (referrer?.parentId) {
      const level2Amount = Math.floor(orderAmount * config.level2Rate);
      if (level2Amount > 0) {
        const commission: CommissionRecord = {
          distributorId: referrer.parentId,
          orderId: orderSn,
          orderSn,
          refereeId: buyerUserId,
          level: 2,
          orderAmount,
          commissionRate: config.level2Rate,
          commissionAmount: level2Amount,
          status: 'pending',
          createdAt: now,
          updatedAt: now,
        };
        await distOrdersCol.insertOne(commission);
        await distUsersCol.updateOne(
          { userId: referrer.parentId },
          { $inc: { pendingEarnings: level2Amount, totalEarnings: level2Amount }, $set: { updatedAt: now } }
        );
        console.log(`[知己·分销] 二级佣金: ${referrer.parentId} +${level2Amount}分`);
      }
    }

    // 更新绑定状态为已转化
    const bindingsCol = await getCollection(COLLECTIONS.REFERRAL_BINDINGS);
    await bindingsCol.updateOne(
      { refereeId: buyerUserId, referrerId, status: 'active' },
      { $set: { status: 'converted', conversionDate: now, orderId: orderSn, updatedAt: now } }
    );
  } catch (error) {
    console.error('[知己·分销] 佣金计算失败:', error);
  }
}

/**
 * 结算佣金（将pending改为settled）
 * 定时任务调用，结算7天前的佣金
 */
export async function settleCommissions(): Promise<number> {
  try {
    const config = getDistributionConfig();
    const settleDate = new Date(Date.now() - config.settleDelayDays * 24 * 60 * 60 * 1000);

    const distOrdersCol = await getCollection(COLLECTIONS.DISTRIBUTION_ORDERS);

    // 查找可结算的佣金
    const pendingCommissions = await distOrdersCol.find({
      status: 'pending',
      createdAt: { $lt: settleDate },
    }).toArray();

    let settledCount = 0;
    for (const commission of pendingCommissions) {
      await distOrdersCol.updateOne(
        { _id: commission._id },
        { $set: { status: 'settled', settledAt: new Date(), updatedAt: new Date() } }
      );

      // 待结算转为可提现
      const distUsersCol = await getCollection(COLLECTIONS.DISTRIBUTION_USERS);
      await distUsersCol.updateOne(
        { userId: commission.distributorId },
        { $inc: { pendingEarnings: -commission.commissionAmount }, $set: { updatedAt: new Date() } }
      );

      settledCount++;
    }

    if (settledCount > 0) {
      console.log(`[知己·分销] 结算完成: ${settledCount}条佣金`);
    }
    return settledCount;
  } catch (error) {
    console.error('[知己·分销] 结算失败:', error);
    return 0;
  }
}

/**
 * 申请提现
 */
export async function createWithdrawal(
  userId: string,
  amount: number, // 分
  accountType: 'wechat' | 'alipay' | 'bankcard',
  account: string,
  accountName: string,
  bankName?: string,
): Promise<WithdrawalRecord | null> {
  try {
    const config = getDistributionConfig();
    const distUsersCol = await getCollection(COLLECTIONS.DISTRIBUTION_USERS);

    // 检查分销员状态
    const distributor = await distUsersCol.findOne({ userId, status: 'active' });
    if (!distributor) {
      throw new Error('分销员不存在或未激活');
    }

    // 计算可提现余额 = 总收益 - 待结算 - 已提现
    const availableAmount = distributor.totalEarnings - distributor.pendingEarnings - distributor.withdrawnAmount;
    if (amount > availableAmount) {
      throw new Error(`可提现余额不足: 可提现${availableAmount}分，申请${amount}分`);
    }

    if (amount < config.minWithdrawAmount) {
      throw new Error(`最低提现金额: ${config.minWithdrawAmount / 100}元`);
    }

    // 计算手续费
    const fee = Math.floor(amount * config.withdrawFeeRate);
    const actualAmount = amount - fee;

    const now = new Date();
    const withdrawal: WithdrawalRecord = {
      distributorId: distributor.userId,
      userId,
      amount,
      fee,
      actualAmount,
      accountType,
      account,
      accountName,
      bankName,
      status: 'pending',
      createdAt: now,
      updatedAt: now,
    };

    const withdrawalsCol = await getCollection(COLLECTIONS.DISTRIBUTION_WITHDRAWALS);
    await withdrawalsCol.insertOne(withdrawal);

    // 冻结提现金额
    await distUsersCol.updateOne(
      { userId },
      { $inc: { withdrawnAmount: amount }, $set: { updatedAt: now } }
    );

    console.log(`[知己·分销] 提现申请: ${userId}, ${amount}分, ${accountType}`);
    return withdrawal;
  } catch (error) {
    console.error('[知己·分销] 提现申请失败:', error);
    return null;
  }
}

  pageSize = 20
): Promise<{ records: CommissionRecord[]; total: number }> {

  const total = await col.countDocuments({ distributorId: userId });
  const records = await col.find({ distributorId: userId })

 */
export async function getProfitSharingRule(productType: ProductType): Promise<ProfitSharingRule> {
  try {
    const rulesCol = await getCollection(COLLECTIONS.PROFIT_SHARING_RULES);

    // 先查商品类型专属规则
    let rule = await rulesCol.findOne({ productType, status: 'active' });
    if (rule) return rule as unknown as ProfitSharingRule;

    // 再查默认规则
    rule = await rulesCol.findOne({ productType: 'default', status: 'active' });
    if (rule) return rule as unknown as ProfitSharingRule;
  } catch {
    // 数据库不可用，使用默认配置
  }

  // 使用硬编码默认规则
  const defaultRule = DEFAULT_RULES[productType] || DEFAULT_RULES['default'];
  return {
    name: `${productType}分账规则`,
    productType,
    receivers: defaultRule.receivers,
    status: 'active',
    createdAt: new Date(),
    updatedAt: new Date(),
  };
}

/**
 * 保存/更新分账规则
 */
export async function saveProfitSharingRule(rule: Partial<ProfitSharingRule> & { productType: string }): Promise<boolean> {
  try {
    const rulesCol = await getCollection(COLLECTIONS.PROFIT_SHARING_RULES);
    const now = new Date();

    // 验证比例之和为1
    const totalRatio = (rule.receivers || []).reduce((sum, r) => sum + r.ratio, 0);
    if (Math.abs(totalRatio - 1.0) > 0.001) {
      console.error(`[知己·分账] 分账比例之和必须为1，当前: ${totalRatio}`);
      return false;
    }

    await rulesCol.updateOne(
      { productType: rule.productType },
      {
        $set: {
          ...rule,
          updatedAt: now,
        },
        $setOnInsert: { createdAt: now },
      },
      { upsert: true }
    );

    console.log(`[知己·分账] 规则已保存: ${rule.productType}`);
    return true;
  } catch (error) {
    console.error('[知己·分账] 规则保存失败:', error);
    return false;
  }
}

/**
 * 执行订单分账
 * 在支付成功回调中调用
 */
export async function executeProfitSharing(
  orderSn: string,
  productType: ProductType,
  totalAmount: number, // 分
  consultantId?: string, // 咨询师ID（咨询订单时传入）
): Promise<ProfitSharingRecord | null> {
  try {
    const recordsCol = await getCollection(COLLECTIONS.PROFIT_SHARING_RECORDS);
    const now = new Date();

    // 防重复
    const existing = await recordsCol.findOne({ orderSn });
    if (existing) {
      console.log(`[知己·分账] 订单已分账，跳过: ${orderSn}`);
      return existing as unknown as ProfitSharingRecord;
    }

    // 获取分账规则
    const rule = await getProfitSharingRule(productType);

    // 计算分账明细
    const details = rule.receivers.map((receiver) => {
      let amount = Math.floor(totalAmount * receiver.ratio);

      // 如果是咨询师类型，需要关联实际咨询师
      const actualReceiver = { ...receiver };
      if (receiver.type === 'consultant' && consultantId) {
        actualReceiver.account = consultantId;
      }

      return {
        receiverType: actualReceiver.type,
        receiverName: actualReceiver.name,
        amount,
        ratio: actualReceiver.ratio,
        account: actualReceiver.account,
        status: 'pending' as const,
      };
    });

    // 处理分账金额的尾差（给平台方）
    const totalShared = details.reduce((sum, d) => sum + d.amount, 0);
    if (totalShared !== totalAmount) {
      const platformDetail = details.find((d) => d.receiverType === 'platform');
      if (platformDetail) {
        platformDetail.amount += totalAmount - totalShared;
      }
    }

    // 创建分账记录
    const record: ProfitSharingRecord = {
      orderSn,
      orderId: orderSn,
      ruleId: rule._id || 'default',
      totalAmount,
      details,
      status: 'pending',
      createdAt: now,
      updatedAt: now,
    };

    await recordsCol.insertOne(record);
    console.log(`[知己·分账] 分账记录已创建: ${orderSn}, 总金额: ${fenToYuan(totalAmount)}元`);

    // 标记为处理中
    await recordsCol.updateOne(
      { orderSn },
      { $set: { status: 'processing', updatedAt: new Date() } }
    );

    // 执行各方分账（实际转账/记账）
    for (const detail of details) {
      try {
        await processReceiverPayment(detail);
        detail.status = 'success' as 'pending';
        // eslint-disable-next-line @typescript-eslint/no-explicit-any
        (detail as any).processedAt = new Date();
      } catch (error) {
        console.error(`[知己·分账] ${detail.receiverName}分账失败:`, error);
        detail.status = 'failed' as 'pending';
      }
    }

    // 更新分账记录状态
    const allSuccess = details.every((d) => d.status === ('success' as 'pending'));
    await recordsCol.updateOne(
      { orderSn },
      {
        $set: {
          details,
          status: allSuccess ? 'completed' : 'failed',
          processedAt: new Date(),
          updatedAt: new Date(),
        },
      }
    );

    console.log(`[知己·分账] 分账完成: ${orderSn}, 状态: ${allSuccess ? '成功' : '部分失败'}`);
    return record;
  } catch (error) {
    console.error('[知己·分账] 执行失败:', error);
    return null;
  }
}

/**
 * 处理单个接收方的分账
 * 实际业务中对接微信分账API / 支付宝分账API / 内部账户
 */
// eslint-disable-next-line @typescript-eslint/no-explicit-any
async function processReceiverPayment(detail: any): Promise<void> {
  switch (detail.receiverType) {
    case 'platform': {
      // 平台收入 → 记入平台账户（内部记账）
      console.log(`[知己·分账] 平台收入: ${fenToYuan(detail.amount)}元`);
      break;
    }
    case 'consultant': {
      // 咨询师分成 → 更新咨询师收益
      if (detail.account) {
        try {
          const consultantsCol = await getCollection(COLLECTIONS.CONSULTANTS);
          await consultantsCol.updateOne(
            { _id: detail.account },
            {
              $inc: { totalEarnings: detail.amount },
              $set: { updatedAt: new Date() },
            }
          );
          console.log(`[知己·分账] 咨询师收入: ${detail.account}, ${fenToYuan(detail.amount)}元`);
        } catch {
          console.log('[知己·分账] 咨询师收入记账失败');
        }
      }
      break;
    }
    case 'distributor': {
      // 分销佣金 → 已在分销系统处理
      console.log(`[知己·分账] 分销佣金: ${fenToYuan(detail.amount)}元`);
      break;
    }
    default: {
      console.log(`[知己·分账] 自定义接收方: ${detail.receiverName}, ${fenToYuan(detail.amount)}元`);
    }
  }
}

          const days = order.productType === 'vip_monthly' ? 30 : order.productType === 'vip_quarterly' ? 90 : 365;
          const expireDate = new Date(now.getTime() + days * 86400000);
          await usersCol.updateOne(
            { _id: order.userId },
            { $set: { vip: true, vipExpireDate: expireDate.toISOString(), vipType: order.productType, updatedAt: now } }
          );
          console.log(`[知己·开发模式] VIP开通: ${order.userId}, 到期: ${expireDate}`);
          break;
        }
        case 'points_pack': {
          const pointsMap: Record<string, number> = { '50': 50, '100': 110, '300': 350, '500': 600, '1000': 1300 };
          const points = pointsMap[order.productId || ''] || 0;
          if (points) {
            await usersCol.updateOne(
              { _id: order.userId },
              { $inc: { points }, $set: { updatedAt: now } }
            );
            console.log(`[知己·开发模式] 积分充值: ${order.userId}, +${points}`);
          }
          break;
        }
        case 'consultation': {
          const bookingsCol = await getCollection(COLLECTIONS.BOOKINGS);
          await bookingsCol.updateOne(
            { orderSn },
            { $set: { paymentStatus: 'paid', paidAt: now, updatedAt: now } }
          );
          console.log(`[知己·开发模式] 咨询订单已支付: ${orderSn}`);
          break;
        }
      }
    } catch (error) {
      console.error('[知己·开发模式] 权益发放失败:', error);
    }

    // 4. 处理分销佣金
    if (order.referrerId) {
      try {
        const distOrdersCol = await getCollection(COLLECTIONS.DISTRIBUTION_ORDERS);
        const distUsersCol = await getCollection(COLLECTIONS.DISTRIBUTION_USERS);

        const level1Rate = 0.2;
        const level1Commission = Math.floor(order.payAmount * level1Rate);

        if (level1Commission > 0) {
          await distOrdersCol.insertOne({
            distributorId: order.referrerId,
            orderId: orderSn, orderSn,
            refereeId: order.userId, level: 1,
            orderAmount: order.payAmount,
            commissionRate: level1Rate,
            commissionAmount: level1Commission,
            status: 'pending', createdAt: now, updatedAt: now,
          });
          await distUsersCol.updateOne(
            { userId: order.referrerId },
            { $inc: { pendingEarnings: level1Commission, totalEarnings: level1Commission }, $set: { updatedAt: now } }
          );
          console.log(`[知己·开发模式] 分销佣金: ${order.referrerId} +${level1Commission}分`);
        }
      } catch { /* 忽略 */ }
    }

    console.log(`[知己·开发模式] 支付确认完成: ${orderSn}`);

    return NextResponse.json({
      code: 200,
      message: '开发模式支付确认成功',
      data: {
        orderSn,
        status: 'paid',
        transactionId: mockTransactionId,
        paidAt: now,
        productType: order.productType,
        amount: order.payAmount,
      },
    });
  } catch (error) {
    console.error('[知己·开发模式] 确认失败:', error);
    return NextResponse.json(
      { code: 500, message: '确认失败', data: null },
      { status: 500 }
    );
  }
}
