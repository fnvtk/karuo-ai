/** 把 Mongo 风格 filter 转为 id 或 JSON 条件（仅支持简单等值、_id） */
function filterToWhere(filter: Record<string, unknown>): { id?: string; conditions: string[]; values: unknown[] } {
  const conditions: string[] = []
  const values: unknown[] = []
  let id: string | undefined

  for (const [key, value] of Object.entries(filter)) {
    if (value === undefined) continue
    if (key === "_id") {
      const v = value as { toString?: () => string }
      id = typeof v === "string" ? v : (v?.toString?.() ?? String(v))
      continue
    }
    if (key === "$or") {
      const orList = value as Record<string, unknown>[]
      const orParts = orList.map((sub) => {
        const { conditions: c, values: v } = filterToWhere(sub)
        if (c.length) {
          values.push(...v)
          return "(" + c.join(" AND ") + ")"
        }
        return "1=0"
      })
      if (orParts.length) conditions.push("(" + orParts.join(" OR ") + ")")
      continue
    }
    if (typeof value === "object" && value !== null && "$regex" in value) {
      const r = value as { $regex: string; $options?: string }
      const pattern = "%" + String(r.$regex).replace(/%/g, "\\%").replace(/_/g, "\\_") + "%"
      conditions.push("JSON_UNQUOTE(JSON_EXTRACT(doc, ?)) LIKE ?")
      values.push("$." + key, pattern)
      continue
    }
    conditions.push("JSON_UNQUOTE(JSON_EXTRACT(doc, ?)) = ?")
    values.push("$." + key, value === null ? null : String(value))
  }

  return { id, conditions, values }
}

/** 把 Mongo 的 update（$set, $inc 等）合并进 doc 对象 */
function applyUpdate(doc: Record<string, unknown>, update: Record<string, unknown>): Record<string, unknown> {
  const next = { ...doc }

  if (update.$set && typeof update.$set === "object") {
    for (const [k, v] of Object.entries(update.$set as Record<string, unknown>)) next[k] = v
  }
  if (update.$inc && typeof update.$inc === "object") {
    for (const [k, delta] of Object.entries(update.$inc as Record<string, number>)) {
      const cur = next[k] as number | undefined
      next[k] = (typeof cur === "number" ? cur : 0) + delta
    }
  }
  if (update.$setOnInsert && typeof update.$setOnInsert === "object") {
    for (const [k, v] of Object.entries(update.$setOnInsert as Record<string, unknown>)) {
      if (next[k] === undefined) next[k] = v
    }
  }

  return next
}

/** 日期/对象序列化为 JSON 友好格式 */
function serializeDoc(doc: Record<string, unknown>): Record<string, unknown> {
  const out: Record<string, unknown> = {}
  for (const [k, v] of Object.entries(doc)) {
    if (v instanceof Date) out[k] = v.toISOString()
    else if (v !== undefined && v !== null && typeof v === "object" && !Array.isArray(v) && Object.getPrototypeOf(v) === Object.prototype)
      out[k] = serializeDoc(v as Record<string, unknown>)
    else out[k] = v
  }
  return out
}

function serializeDoc(doc) {
  const out = {}

| 项 | 说明 |

## 2. 环境变量（MySQL 模式）

部署时必须设置 `DATABASE_TYPE=mysql`，并配置 MySQL 连接：

| 变量 | 必填 | 说明 |
|:---|:---:|:---|
| `DATABASE_TYPE` | ✅ | 固定为 `mysql`（NAS 部署） |
| `MYSQL_HOST` | ✅ | MySQL 地址（Docker 内为服务名 `mysql`） |
| `MYSQL_PORT` | | 默认 3306 |
| `MYSQL_DATABASE` | | 默认 `zhiji` |
| `MYSQL_USER` | | 默认见 docker-compose |
| `MYSQL_PASSWORD` | | 与 MySQL 容器一致 |

其他如 `AI_API_KEY`、`CUNKEBAO_*`、支付相关等按需在 `.env` 或 docker-compose 的 `environment` 中配置。

---

## 3. 一键部署（Docker Compose）

### 3.1 在 NAS 上操作

1. **上传项目**  
   将整个「知己开发」项目拷贝到 NAS 某目录，例如：`/docker/zhiji/`。

2. **创建环境变量文件（可选）**  
   在项目根目录创建 `.env`，覆盖默认密码等：
   ```bash
   MYSQL_ROOT_PASSWORD=你的root密码
   MYSQL_DATABASE=zhiji
   MYSQL_USER=zhiji
   MYSQL_PASSWORD=你的数据库密码
   ```

3. **构建并启动**
   ```bash
   cd /docker/zhiji
   docker compose up -d
   ```

4. **等待 MySQL 就绪**  
   首次启动 MySQL 会初始化，约 30 秒后再访问应用。  
   应用端口：**3050**（浏览器访问 `http://NAS的IP:3050`，管理后台 `http://NAS的IP:3050/admin`）。

### 3.2 本机测试（与 NAS 相同流程）

```bash
cd 知己开发
docker compose up -d
# 访问 http://localhost:3050
```

## 4. MongoDB 数据迁移到 MySQL（在 NAS 上执行）

若原有数据在 MongoDB，需先迁到 NAS 上的 MySQL，再切到 MySQL 模式运行。

### 4.1 保证 MySQL 已就绪

- 已用上面的 `docker compose up -d` 启动过，或 NAS 上已有 MySQL 服务并创建好数据库 `zhiji`。

### 4.2 执行迁移脚本

迁移脚本会从 MongoDB 读取各集合数据，写入 MySQL（表结构：`id` + `doc` JSON）。

**方式一：在 NAS 上安装 Node 后执行（推荐）**

```bash
cd /docker/zhiji
export MONGODB_URI="mongodb://你的Mongo地址:27017"
export MONGODB_DB=zhiji
export MYSQL_HOST=127.0.0.1
export MYSQL_PORT=3306
export MYSQL_DATABASE=zhiji
export MYSQL_USER=zhiji
export MYSQL_PASSWORD=你的MySQL密码
node scripts/migrate-mongo-to-mysql.mjs
```

**方式二：在已运行 MySQL 的 Docker 宿主机上执行**

- 若 MySQL 在 Docker 内，`MYSQL_HOST` 填 `127.0.0.1` 或 NAS 本机 IP，并确保 3306 已映射到主机。

**方式三：迁移脚本放进一次性容器内执行**

- 将 `scripts/migrate-mongo-to-mysql.mjs` 和依赖（或整个项目）挂载进一个带 Node 的容器，环境变量指向 NAS 上的 MySQL 和 Mongo，在容器内执行 `node scripts/migrate-mongo-to-mysql.mjs`。

### 4.3 迁移后

- 应用已使用 `DATABASE_TYPE=mysql` 且 `MYSQL_*` 指向同一 MySQL，则重启应用即可，无需再改代码。

## 5. 表结构说明（MySQL）

- 所有表均为：`id`（主键，对应 Mongo `_id` 字符串）+ `doc`（JSON，存整条文档）。
- 建表由应用在首次连接 MySQL 时自动执行（`lib/mysql.ts` 的 `initializeMySQLTables`），也可提前执行 `开发文档/7、数据库/MySQL表结构.sql`。

## 6. 常见问题

| 问题 | 处理 |
|:---|:---|
| 访问 3050 无响应 | 检查 NAS 防火墙是否放行 3050；`docker compose ps` 看 zhiji 容器是否运行。 |
| 应用报错连不上 MySQL | 确认 `MYSQL_HOST`（Docker 内为 `mysql`）、端口、账号密码与 MySQL 容器一致；等 MySQL 健康检查通过后再访问。 |
| 迁移脚本报连接失败 | 检查 `MONGODB_URI` 可从当前机器访问；`MYSQL_*` 是否指向 NAS 上已启动的 MySQL。 |
| 想继续用 MongoDB | 不设置 `DATABASE_TYPE` 或设为 `mongodb`，并配置 `MONGODB_URI` 即可。 |

---

## 7. 总结

- **NAS 部署**：用 `docker-compose.yml` 启动 MySQL + 知己应用，环境变量设 `DATABASE_TYPE=mysql` 和 `MYSQL_*`。
- **Mongo → MySQL**：在 NAS（或能同时访问 Mongo 与 MySQL 的机器）上执行 `node scripts/migrate-mongo-to-mysql.mjs`，配置好两边连接信息。
- 迁移完成后，整个项目在 NAS 上仅依赖 MySQL 即可运转，无需再部署 MongoDB。

## 你现在可以这样用

| 目标 | 操作 |
|:---|:---|
| **在 NAS 上一键跑起项目** | 把项目拷到 NAS，执行 `docker compose up -d`，访问 `http://NAS的IP:3050`。 |
| **用 MySQL 当数据库** | 在环境变量中设置 `DATABASE_TYPE=mysql` 和 `MYSQL_*`（docker-compose 已配好）。 |
| **把现有 Mongo 数据迁到 MySQL** | 在能同时访问 Mongo 和 MySQL 的机器上配置两边连接信息，执行 `node scripts/migrate-mongo-to-mysql.mjs`。 |
| **继续用 MongoDB** | 不设或设 `DATABASE_TYPE=mongodb`，并配置 `MONGODB_URI`。 |

关键文件一览：

- `lib/mysql.ts`：MySQL 连接与集合适配器（兼容现有 Mongo 风格 API）。  
- `lib/mongodb.ts`：按 `DATABASE_TYPE` 切换 Mongo/MySQL，并导出兼容的 `ObjectId`。  
- `scripts/migrate-mongo-to-mysql.mjs`：Mongo → MySQL 迁移脚本。  
- `docker-compose.yml` + `Dockerfile`：NAS/本机 Docker 部署。  
- `开发文档/8、部署/NAS部署说明.md`：NAS 部署与迁移的完整说明。
