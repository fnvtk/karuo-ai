  contents: #!/bin/bash
# NAS ↔ GitHub 双向自动同步脚本（在群晖 NAS 上执行）
# 用途：定时拉取 GitHub 更新并推送 NAS 本地修改，实现无需人工参与的双向同步
# 仓库：https://github.com/fnvtk/skills.git
# 建议：通过 DSM 计划任务每 5～10 分钟执行一次

set -e
REPO_DIR="${GITHUB_SYNC_REPO_DIR:-/volume1/SynologyDrive/存客宝AI}"
TOKEN_FILE="${GITHUB_SYNC_TOKEN_FILE:-$REPO_DIR/.github_sync_token}"
LOG_FILE="${REPO_DIR}/.sync.log"
BRANCH="${GITHUB_SYNC_BRANCH:-main}"
REMOTE="${GITHUB_SYNC_REMOTE:-origin}"

# 日志（带时间戳）
log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_FILE"; }

# 若目录不存在则退出
if [ ! -d "$REPO_DIR" ]; then
  log "ERROR: 目录不存在: $REPO_DIR"
  exit 1
fi

# 若未配置 token 则退出
if [ ! -f "$TOKEN_FILE" ]; then
  log "ERROR: Token 文件不存在: $TOKEN_FILE，请先运行部署脚本完成配置"
  exit 1
fi

GITHUB_TOKEN=$(cat "$TOKEN_FILE")
export GITHUB_TOKEN
if [ -z "$GITHUB_TOKEN" ]; then
  log "ERROR: Token 为空，请检查 $TOKEN_FILE"
  exit 1
fi

cd "$REPO_DIR"

# 若非 git 仓库则退出
if [ ! -d .git ]; then
  log "ERROR: 当前目录不是 Git 仓库: $REPO_DIR"
  exit 1
fi

# 使用 token 设置 remote URL（避免每次输入密码）
REMOTE_URL="https://${GITHUB_TOKEN}@github.com/fnvtk/skills.git"
git remote set-url "$REMOTE" "$REMOTE_URL" 2>/dev/null || true

# 配置提交者（用于 NAS 上的自动提交）
git config user.email "nas-sync@cunkebao.local" 2>/dev/null || true
git config user.name "NAS Auto Sync" 2>/dev/null || true

# 1. 先拉取远程更新（优先保证 GitHub → NAS）
log "Pull from $REMOTE/$BRANCH ..."
if ! git pull "$REMOTE" "$BRANCH" --rebase --autostash 2>>"$LOG_FILE"; then
  log "WARN: Pull 失败，尝试 merge..."
  git merge --abort 2>/dev/null || true
  git rebase --abort 2>/dev/null || true
  if ! git pull "$REMOTE" "$BRANCH" --no-rebase 2>>"$LOG_FILE"; then
    log "ERROR: Pull 仍失败，请检查冲突或网络"
    exit 1
  fi
fi
log "Pull 完成"

# 2. 若有本地修改则提交并推送（NAS → GitHub）
if [ -n "$(git status --porcelain)" ]; then
  log "检测到本地修改，提交并推送..."
  git add -A
  git commit -m "NAS auto-sync $(date '+%Y-%m-%d %H:%M')" 2>>"$LOG_FILE" || log "WARN: 无新变更可提交或提交失败"
  if git push "$REMOTE" "$BRANCH" 2>>"$LOG_FILE"; then
    log "Push 完成"
  else
    log "WARN: Push 失败，下次同步会重试"
  fi
else
  log "无本地修改，跳过 Push"
fi

log "--- 同步结束 ---"

实现 **Synology NAS** 上目录 `存客宝AI` 与 **GitHub 仓库** [fnvtk/skills](https://github.com/fnvtk/skills.git) 的双向自动同步，部署后无需再参与。

## 流程说明

- **GitHub → NAS**：每 5 分钟在 NAS 上执行 `git pull`，拉取远程更新。
- **NAS → GitHub**：若 NAS 上有本地修改，同一轮同步中会自动 `git add`、`git commit`、`git push`。
- **Token**：部署时由本机通过 SSH 将 GitHub Token 写入 NAS 的 `存客宝AI/.github_sync_token`，仅 NAS 本地使用，不提交到 Git。

## 一键部署（本机 Mac 执行）

1. 确保本机可 SSH 到 NAS（如 `ssh fnvtk@192.168.110.101`，需 legacy 算法见项目 `.cursorrules`）。
2. 设置 GitHub Token 并执行部署脚本：

```bash
cd "/Users/karuo/Documents/开发/4、小工具/synology群晖nas"
export GITHUB_TOKEN=ghp_你的Token
./scripts/deploy_github_sync_to_nas.sh
```

卡若AI 可从记忆/配置中读取 GitHub Token，代你执行上述命令完成部署。

## 脚本说明

| 文件 | 运行位置 | 作用 |
|------|----------|------|
| `git_sync_nas.sh` | NAS | 单次同步：pull 再视情况 push，写日志到 `SynologyDrive/.sync_存客宝AI.log` |
| `deploy_github_sync_to_nas.sh` | 本机 | 上传同步脚本、写入 Token、初始化/克隆仓库、添加 crontab 每 5 分钟执行 |

## 可选配置（环境变量）

在 NAS 上执行 `git_sync_nas.sh` 时可通过环境变量覆盖默认值（也可在 crontab 里设置）：

- `GITHUB_SYNC_REPO_DIR`：仓库目录，默认 `/volume1/SynologyDrive/存客宝AI`。
- `GITHUB_SYNC_TOKEN_FILE`：Token 文件路径，默认 `$REPO_DIR/.github_sync_token`。
- `GITHUB_SYNC_BRANCH`：分支名，默认 `main`。

## 查看同步日志

```bash
ssh -o KexAlgorithms=+diffie-hellman-group1-sha1 -o Ciphers=+aes128-cbc,3des-cbc,aes192-cbc,aes256-cbc fnvtk@192.168.110.101 'tail -f /volume1/SynologyDrive/.sync_存客宝AI.log'
```

## 注意事项

- NAS 上需已安装 `git`（套件中心或 SSH 安装）。
- 若 `存客宝AI` 路径在 DSM 上实际为 `homes/fnvtk/SynologyDrive/存客宝AI`，则 NAS 上真实路径可能是 `/volume1/homes/fnvtk/SynologyDrive/存客宝AI`，部署前可在 NAS 上确认后设置 `GITHUB_SYNC_REPO_DIR` 或修改脚本中的默认路径。
- 若仓库默认分支为 `master`，可在 crontab 中设置 `GITHUB_SYNC_BRANCH=master` 或在脚本里改默认值。
