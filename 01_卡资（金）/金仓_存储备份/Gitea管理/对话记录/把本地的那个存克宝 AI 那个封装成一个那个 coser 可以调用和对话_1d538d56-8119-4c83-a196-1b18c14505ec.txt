class ExecuteRequest(BaseModel):

@app.post("/api/ai/execute")
def api_execute(req: ExecuteRequest):
    """简单执行：直接传任务与可选组，返回路由与执行说明。"""
    ai = get_ai()
    group_enum = TeamGroup.AUTO
    if req.group != "auto":
        m = {
            "运维组": TeamGroup.DEVOPS,
            "项目组": TeamGroup.PROJECT,
            "开发组": TeamGroup.DEVELOP,
            "测试组": TeamGroup.TEST,
        }
        group_enum = m.get(req.group, TeamGroup.AUTO)

    return {

        "message": result.message,
        "group": result.group,
        "skill": result.skill,
        "data": result.data,
    }


@app.post("/v1/chat/completions")
def openai_chat_completions(body: OpenAICompatRequest):
    """
    OpenAI 兼容接口：把 messages 中最后一条 user 内容当作「任务」执行，
    返回格式兼容 OpenAI Chat Completions，便于 Cursor/其他客户端统一调用。
    """
    if not body.messages:
        raise HTTPException(status_code=400, detail="messages 不能为空")
    # 取最后一条 user 消息作为任务
    task = ""
    for m in reversed(body.messages):
        if m.role == "user" and m.content:
            task = m.content.strip()
            break
    if not task:
        raise HTTPException(status_code=400, detail="需要至少一条 user 消息作为任务")
    ai = get_ai()
    result = ai.execute(task)
    # 构造 OpenAI 风格回复
    content = (
        f"【存客宝AI】\n"
        f"任务：{task}\n"
        f"路由：{result.group}\n"
        f"技能：{result.skill}\n"
        f"说明：{result.message}\n"
        f"执行指引：\n{result.data.get('instructions', '') if result.data else ''}"
    )
    return {
        "id": "cunkebao-ai-1",
        "object": "chat.completion",
        "model": body.model or "cunkebao-ai",
        "choices": [
            {
                "index": 0,
                "message": {"role": "assistant", "content": content},
                "finish_reason": "stop",
            }
        ],
        "usage": {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0},
    }
