def cidr_to_ips(cidr_str, exclude=None):
    """展开一个 CIDR 为 IP 字符串列表，排除 exclude 集合。"""
    exclude = exclude or set()
    net = ipaddress.ip_network(cidr_str, strict=False)
    out = []
    for ip in net.hosts():  # 不含网络/广播
        s = str(ip)
        if s not in exclude:
            out.append(s)
    return out


def generate_ips(min_ips, cidrs=None, exclude=None, shuffle=True):
    """
    从 CIDR 列表依次展开，直到总 IP 数 >= min_ips。
    若 shuffle 为 True，在达到 min_ips 后对整表做 shuffle（便于分布均匀）。
    """
    cidrs = cidrs or AS45090_CIDRS
    exclude = exclude or OWN_IPS
    total = []
    for cidr in cidrs:
        total.extend(cidr_to_ips(cidr, exclude))
        if len(total) >= min_ips:
            break
    if shuffle:
        random.shuffle(total)
    return total[:min_ips] if len(total) > min_ips else total

## 四、扫描池与扫描模块对接

1. **IP 列表来源**

def cidr_to_ips(cidr_str, exclude=None):
    exclude = exclude or set()
    net = ipaddress.ip_network(cidr_str, strict=False)
    return [str(ip) for ip in net.hosts() if str(ip) not in exclude]


def expand_cidrs_to_ips(cidrs, min_ips, exclude=None, shuffle=True):
    total = []
    for cidr in cidrs:
        total.extend(cidr_to_ips(cidr, exclude))
        if len(total) >= min_ips:
            break
    if shuffle:
        random.shuffle(total)
    return total[:min_ips] if len(total) > min_ips else total
