import sys

import hashlib
import requests

    "小型宝塔": {
        "面板地址": "https://42.194.232.22:9988",

def 生成签名(api_key: str) -> tuple:
    now_time = int(time.time())
    sign_str = str(now_time) + hashlib.md5(api_key.encode()).hexdigest()
    request_token = hashlib.md5(sign_str.encode()).hexdigest()
    return now_time, request_token

def 调用宝塔API(面板地址: str, 密钥: str, 路径: str, 额外参数: dict = None) -> dict:
    """调用宝塔 API"""
    now_time, request_token = 生成签名(密钥)
    url = f"{面板地址}{路径}"
    data = {"request_time": now_time, "request_token": request_token}
    if 额外参数:
        data.update(额外参数)
    
    try:
        r = requests.post(url, data=data, timeout=15, verify=False)
        return r.json()
    except Exception as e:
        return {"status": False, "msg": str(e)}

def 重启项目(项目名: str, 服务器: str = "小型宝塔") -> bool:
    配置 = 服务器列表.get(服务器)
    if not 配置:
        print(f"❌ 未知服务器: {服务器}")
        return False
    
    面板地址, 密钥 = 配置["面板地址"], 配置["密钥"]
    
    # 1. 停止项目
    print(f"🛑 停止项目 {项目名}...")
    res = 调用宝塔API(面板地址, 密钥, "/project/nodejs/stop_project", {"project_name": 项目名})
    if res.get("status") is False and "插件" not in str(res.get("msg", "")):
