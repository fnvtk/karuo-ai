def 生成签名(api_key: str) -> tuple:
    now_time = int(time.time())
    sign_str = str(now_time) + hashlib.md5(api_key.encode("utf-8")).hexdigest()
    request_token = hashlib.md5(sign_str.encode("utf-8")).hexdigest()
    return now_time, request_token


def 宝塔POST(面板地址: str, 密钥: str, action_path: str, extra_data: dict, 通过小型宝塔转发: bool) -> dict:
    now_time, request_token = 生成签名(密钥)
    url = f"{面板地址.rstrip('/')}/{action_path.lstrip('/')}"
    data = {"request_time": now_time, "request_token": request_token, **extra_data}
    if not 通过小型宝塔转发:
        try:
            r = requests.post(url, data=data, timeout=20, verify=False)
            return r.json()
        except Exception as e:
            return {"status": False, "msg": str(e)}
    data_str = "&".join(f"{k}={urllib.parse.quote(str(v))}" for k, v in data.items())
    script = f'''curl -sk -X POST -d "{data_str}" "{url}"'''
    cmd = [

        return {"status": False, "msg": str(e)}

def 是否已启动(项目: dict) -> bool:
    """从 API 返回的项里判断是否运行中（pid 存在且非空、或 status 表示运行）"""
    pid = 项目.get("pid") or 项目.get("project_pid")
    if pid is not None and str(pid).strip() and str(pid).strip() != "--":
        return True
    status = (项目.get("status") or 项目.get("run_status") or "").strip()
    if "运行" in status or "run" in status.lower() or status == "1":
        return True
    return False


def 启动项目(配置: dict, 项目名: str) -> bool:
    panel = 配置["面板地址"]
    key = 配置["密钥"]
    forward = 配置.get("通过小型宝塔转发", False)
    for path in ["project/nodejs/start_project", "plugin?name=nodejs&action=start_project"]:
        res = 宝塔POST(panel, key, path, {"project_name": 项目名}, forward)
        if res is None:

        if res.get("status") is True:
            return True
        if res.get("status") is False and "msg" in res:
            # 可能已在运行

    return False
