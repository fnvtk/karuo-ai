"""
神射手 - 按手机号查询用户（人肉式详细输出）
用法: python3 query_user_by_phone.py 15880802661
"""
import sys
import json
from datetime import datetime

def default_serializer(obj):
    if isinstance(obj, datetime):
        return obj.isoformat()
    raise TypeError(type(obj))

def main():
    phone = (sys.argv[1] or "").strip()
    if not phone or len(phone) != 11:
        print(json.dumps({"error": "请提供11位手机号", "usage": "python3 query_user_by_phone.py 15880802661"}, ensure_ascii=False, indent=2))
        sys.exit(1)

    try:
        from pymongo import MongoClient
    except ImportError:
        print(json.dumps({"error": "未安装 pymongo", "fix": "pip install pymongo"}, ensure_ascii=False, indent=2))
        sys.exit(2)

    # 神射手 Skill 中的连接配置
    uri = "mongodb://admin:admin123@localhost:27017/?authSource=admin"
    try:
        client = MongoClient(uri, serverSelectionTimeoutMS=5000)
        client.admin.command("ping")
    except Exception as e:
        print(json.dumps({
            "error": "MongoDB 连接失败",
            "detail": str(e),
            "hint": "请确认 MongoDB 已启动（本地或 docker datacenter_mongodb）"
        }, ensure_ascii=False, indent=2))
        sys.exit(3)

    db = client["KR"]
    coll = db["用户估值"]

    # 查询方式：user_key 或 phone_masked（脱敏格式 158****2661）
    user_key = f"UK_phone_{phone}"
    phone_masked = f"{phone[:3]}****{phone[-4:]}" if len(phone) == 11 else None

    doc = None
    doc = coll.find_one({"user_key": user_key})
    if not doc and phone_masked:
        doc = coll.find_one({"core_profile.phone_masked": phone_masked})
    if not doc:
        doc = coll.find_one({"$or": [
            {"user_key": user_key},
            {"core_profile.phone_masked": phone_masked},
            {"core_profile.phone": phone},
        ]})

    if not doc:
        print(json.dumps({
            "found": False,
            "phone": phone,
            "user_key_tried": user_key,
            "phone_masked_tried": phone_masked,
            "message": "神射手库 KR.用户估值 中未找到该手机号对应记录",
        }, ensure_ascii=False, indent=2, default=default_serializer))
        return

    # 移除 MongoDB _id 便于 JSON 输出
    if "_id" in doc:
        doc["_id"] = str(doc["_id"])
    print(json.dumps({"found": True, "phone": phone, "data": doc}, ensure_ascii=False, indent=2, default=default_serializer))

if __name__ == "__main__":
    main()

**卡若AI 收到消息后的执行步骤：**

1. **难度评估**（5 分制：步骤数、创造性、代码、推理链、输出要求）  
2. **模型选择**：≤3 分 → 本地 qwen2.5；>3 分 → 高级模型（Claude/Opus）  
3. **任务路由**：关键词 → 卡资/卡人/卡木/卡火/卡土  
4. **按需读**：只读和本次任务相关的记忆或 SKILL.md  
5. **执行**：调工具、或 ollama、或直接回复  
6. **汇报**：末尾写本次调用的模型与成本

## 2. 判断步骤怎样省 Token？

- **规则**：规则文件只写必要条款，长说明放到 `.cursor/docs/` 等，按需读。  
- **记忆 / Skill**：不每轮全读，只读和当前任务相关的；Skill 可只读相关段落（offset/limit）。  
- **引用**：少 @ 整本大文件，多引用具体段落或文件路径。  
- **对话**：长对话可新开或先总结再继续，减少历史轮次。  
- **回复**：让你「简短」「只要结论」时，少展开、少重复。  

判断步骤本身尽量用「关键词 + 任务类型」快速定难度和路由，不在 prompt 里写长说明，也能少占 Token。

## 3. 怎样省模型消耗（少用加号 / 高级模型）？

- **严格按难度分流**：能判成 ≤3 分的（摘要、分类、简单问答、出行清单、任务拆解、查表）→ 一律走**本地 qwen2.5**，不占云端/加号。  
- **你说「用小模型」**：直接按本地模型执行，不再用高级模型做这一项。  
- **先本地后高级**：本地做摘要/拆解/提纲，只把结论或要点交给高级模型精修，减少高级模型输入和调用次数。  
- **能工具就不模型**：能查库、跑脚本、查文档解决的（如神射手查用户），不交给模型生成。  
- **批量用本地**：大批量分类、打标、简单改写用本地；高级模型只做少量关键条或最终审核。  

判断步骤里：**“简单/查表/清单/拆解 → 本地；复杂/创意/架构/多步推理 → 高级”**，这样既省 Token 又省加号/模型消耗。
