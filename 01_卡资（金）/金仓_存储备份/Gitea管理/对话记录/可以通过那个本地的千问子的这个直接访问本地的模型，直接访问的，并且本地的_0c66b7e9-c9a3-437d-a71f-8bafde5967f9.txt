def load_trae_token():
    """从 Trae 配置文件读取认证 token"""
    try:
        if not os.path.exists(TRAE_STORAGE_PATH):
            logger.error(f"找不到 Trae 配置文件: {TRAE_STORAGE_PATH}")
            logger.error("请确保已安装并登录 Trae CN")
            return None
            
        with open(TRAE_STORAGE_PATH, 'r', encoding='utf-8') as f:
            storage = json.load(f)
        
        auth_str = storage.get("iCubeAuthInfo://icube.cloudide", "{}")
        auth = json.loads(auth_str)
        
        token = auth.get("token")
        expires = auth.get("expiredAt", "")
        username = auth.get("account", {}).get("username", "未知")
        
        if not token:
            logger.error("Trae 认证信息为空，请先登录 Trae")
            return None
            
        logger.info(f"✅ 已加载 Trae 认证信息")
        logger.info(f"   用户: {username}")
        logger.info(f"   Token 有效期: {expires}")
        
        return token
        
    except Exception as e:
        logger.error(f"读取 Trae 配置失败: {e}")
        return None

class TraeProxyHandler(BaseHTTPRequestHandler):
    """处理 OpenAI 格式的请求，转发到 Trae API"""
    
    trae_token = None
    
    def log_message(self, format, *args):
        """自定义日志格式"""
        logger.info(f"{self.address_string()} - {format % args}")
    
    def send_cors_headers(self):
        """发送 CORS 头"""
        self.send_header("Access-Control-Allow-Origin", "*")
        self.send_header("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type, Authorization")
    
    def do_OPTIONS(self):
        """处理 CORS 预检请求"""
        self.send_response(200)
        self.send_cors_headers()
        self.end_headers()
    
    def do_GET(self):
        """处理 GET 请求"""
        if self.path == "/v1/models":
            self.handle_models()
        elif self.path == "/" or self.path == "/health":
            self.handle_health()
        else:
            self.send_error(404, "Not Found")
    
    def do_POST(self):
        """处理 POST 请求"""
        if self.path == "/v1/chat/completions":
            self.handle_chat_completions()
        else:
            self.send_error(404, "Not Found")
    
    def handle_health(self):
        """健康检查"""
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()
        self.end_headers()
        
        response = {
            "status": "ok",
            "service": "Trae Proxy",
            "version": "1.0.0",
            "trae_connected": self.trae_token is not None
        }
        self.wfile.write(json.dumps(response).encode())
    
    def handle_models(self):
        """返回可用模型列表"""
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()
        self.end_headers()
        
        models = [
            {"id": "claude-3.7-sonnet", "object": "model", "owned_by": "trae"},
            {"id": "claude-3.5-sonnet", "object": "model", "owned_by": "trae"},
            {"id": "claude-opus-4.5", "object": "model", "owned_by": "trae"},
            {"id": "gpt-4o", "object": "model", "owned_by": "trae"},
            {"id": "gpt-4", "object": "model", "owned_by": "trae"},
            {"id": "doubao", "object": "model", "owned_by": "trae"},
        ]
        
        response = {"object": "list", "data": models}
        self.wfile.write(json.dumps(response).encode())
    
    def handle_chat_completions(self):
        """处理聊天补全请求"""
        if not self.trae_token:
            self.send_error(401, "Trae token not loaded")
            return
        
        try:
            # 读取请求体
            content_length = int(self.headers.get("Content-Length", 0))
            body = self.rfile.read(content_length)
            request_data = json.loads(body)
            
            # 转换模型名称
            model = request_data.get("model", "default")
            trae_model = MODEL_MAPPING.get(model, MODEL_MAPPING["default"])
            
            logger.info(f"📤 请求模型: {model} -> {trae_model}")
            
            # 构建 Trae API 请求
            trae_request = {
                "model": trae_model,
                "messages": request_data.get("messages", []),
                "stream": request_data.get("stream", False),
                "temperature": request_data.get("temperature", 0.7),
                "max_tokens": request_data.get("max_tokens", 4096),
            }
            
            # 检查是否流式响应
            is_stream = request_data.get("stream", False)
            
            # 发送请求到 Trae
            url = f"{TRAE_API_HOST}{TRAE_CHAT_ENDPOINT}"
            headers = {
                "Content-Type": "application/json",
                "Authorization": f"Bearer {self.trae_token}",
                "User-Agent": "Trae-Proxy/1.0",
            }
            
            req = Request(
                url,
                data=json.dumps(trae_request).encode(),
                headers=headers,
                method="POST"
            )
            
            try:
                response = urlopen(req, timeout=120)
                
                if is_stream:
                    # 流式响应
                    self.send_response(200)
                    self.send_header("Content-Type", "text/event-stream")
                    self.send_header("Cache-Control", "no-cache")
                    self.send_cors_headers()
                    self.end_headers()
                    
                    for line in response:
                        self.wfile.write(line)
                        self.wfile.flush()
                else:
                    # 非流式响应
                    result = json.loads(response.read())
                    
                    self.send_response(200)
                    self.send_header("Content-Type", "application/json")
                    self.send_cors_headers()
                    self.end_headers()
                    
                    self.wfile.write(json.dumps(result).encode())
                    
                logger.info(f"✅ 请求完成")
                
            except HTTPError as e:
                error_body = e.read().decode() if e.fp else str(e)
                logger.error(f"Trae API 错误: {e.code} - {error_body}")
                self.send_error(e.code, error_body)
                
            except URLError as e:
                logger.error(f"网络错误: {e}")
                self.send_error(503, f"Network error: {e}")
                
        except json.JSONDecodeError as e:
            logger.error(f"JSON 解析错误: {e}")
            self.send_error(400, f"Invalid JSON: {e}")
            
        except Exception as e:
            logger.error(f"处理请求失败: {e}")
            self.send_error(500, str(e))

# ============ Token 管理 ============
class TokenManager:
    """管理 Trae Token，支持自动刷新"""
    
    def __init__(self):
        self.token = None
        self.expires_at = None
        self.username = None
        self.load_token()
    
    def load_token(self):
        """加载 Token"""
        try:
            if not os.path.exists(TRAE_STORAGE_PATH):
                logger.error(f"找不到 Trae 配置文件: {TRAE_STORAGE_PATH}")
                return False
            
            with open(TRAE_STORAGE_PATH, 'r', encoding='utf-8') as f:
                storage = json.load(f)
            
            auth_str = storage.get("iCubeAuthInfo://icube.cloudide", "{}")
            auth = json.loads(auth_str)
            
            self.token = auth.get("token")
            self.expires_at = auth.get("expiredAt", "")
            self.username = auth.get("account", {}).get("username", "未知")
            
            if self.token:
                logger.info(f"✅ Token 加载成功 - 用户: {self.username}")
                logger.info(f"   Token 有效期: {self.expires_at}")
                return True
            else:
                logger.error("Token 为空，请先登录 Trae")
                return False
                
        except Exception as e:
            logger.error(f"加载 Token 失败: {e}")
            return False
    
    def get_token(self):
        """获取有效 Token"""
        # 简单检查是否过期
        if self.expires_at:
            try:
                expires = datetime.fromisoformat(self.expires_at.replace('Z', '+00:00'))
                if datetime.now(expires.tzinfo) > expires:
                    logger.warning("Token 已过期，尝试重新加载...")
                    self.load_token()

class TraeProxyHandler(BaseHTTPRequestHandler):
    """处理 OpenAI 格式的请求"""
    
    def log_message(self, format, *args):
        logger.info(f"{self.address_string()} - {format % args}")
    
    def send_cors_headers(self):
        self.send_header("Access-Control-Allow-Origin", "*")
        self.send_header("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type, Authorization")
    
    def do_OPTIONS(self):
        self.send_response(200)
        self.send_cors_headers()
        self.end_headers()
    
    def do_GET(self):
        if self.path == "/v1/models":
            self.handle_models()
        elif self.path in ["/", "/health", "/status"]:
            self.handle_health()
        else:
            self.send_error(404, "Not Found")
    
    def do_POST(self):
        if self.path == "/v1/chat/completions":
            self.handle_chat_completions()
        else:
            self.send_error(404, "Not Found")
    
    def handle_health(self):

        self.send_cors_headers()
        self.end_headers()

            "user": token_manager.username,
            "token_expires": token_manager.expires_at,
            "pid": os.getpid(),
            "uptime": time.time() - start_time
        }
        self.wfile.write(json.dumps(response, ensure_ascii=False).encode())
    
    def handle_models(self):
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()

    def handle_chat_completions(self):
        token = token_manager.get_token()
        if not token:
            self.send_error(401, "Trae token not available")
            return
        
        try:
            content_length = int(self.headers.get("Content-Length", 0))
            body = self.rfile.read(content_length)
            request_data = json.loads(body)
            
            model = request_data.get("model", "default")
            trae_model = MODEL_MAPPING.get(model, MODEL_MAPPING["default"])
            
            logger.info(f"📤 请求: {model} -> {trae_model}")
            
            trae_request = {
                "model": trae_model,
                "messages": request_data.get("messages", []),
                "stream": request_data.get("stream", False),
                "temperature": request_data.get("temperature", 0.7),
                "max_tokens": request_data.get("max_tokens", 4096),
            }
            
            is_stream = request_data.get("stream", False)
            
            url = f"{TRAE_API_HOST}/api/ai/chat/completions"
            headers = {
                "Content-Type": "application/json",
                "Authorization": f"Bearer {token}",
                "User-Agent": "Trae-Proxy-Daemon/2.0",
            }
            
            req = Request(
                url,
                data=json.dumps(trae_request).encode(),
                headers=headers,
                method="POST"
            )
            
            try:
                response = urlopen(req, timeout=180)
                
                if is_stream:
                    self.send_response(200)
                    self.send_header("Content-Type", "text/event-stream")
                    self.send_header("Cache-Control", "no-cache")
                    self.send_cors_headers()
                    self.end_headers()
                    
                    for line in response:
                        self.wfile.write(line)
                        self.wfile.flush()
                else:
                    result = json.loads(response.read())
                    
                    self.send_response(200)
                    self.send_header("Content-Type", "application/json")
                    self.send_cors_headers()
                    self.end_headers()
                    
                    self.wfile.write(json.dumps(result).encode())
                
                logger.info(f"✅ 请求完成")
                
            except HTTPError as e:
                error_body = e.read().decode() if e.fp else str(e)
                logger.error(f"Trae API 错误: {e.code} - {error_body[:200]}")
                self.send_error(e.code, error_body[:200])
                
            except URLError as e:
                logger.error(f"网络错误: {e}")
                self.send_error(503, f"Network error: {e}")
                
        except json.JSONDecodeError as e:
            logger.error(f"JSON 解析错误: {e}")
            self.send_error(400, f"Invalid JSON: {e}")
            
        except Exception as e:
            logger.error(f"处理请求失败: {e}")
            self.send_error(500, str(e))

def check_ollama():
    """检查 Ollama 是否运行"""
    try:
        req = Request(f"{OLLAMA_HOST}/api/tags")
        response = urlopen(req, timeout=5)
        return response.status == 200
    except:
        return False

def start_ollama():
    """启动 Ollama 服务"""
    logger.info("启动 Ollama 服务...")
    
    env = os.environ.copy()
    env["OLLAMA_HOST"] = "0.0.0.0:11434"
    env["OLLAMA_ORIGINS"] = "*"
    
    subprocess.Popen(
        ["ollama", "serve"],
        env=env,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL
    )
    
    # 等待启动
    for _ in range(10):
        time.sleep(1)
        if check_ollama():
            logger.info("✅ Ollama 服务已启动")
            return True
    
    logger.error("❌ Ollama 服务启动失败")
    return False

def ensure_ollama():
    """确保 Ollama 运行"""
    if check_ollama():
        logger.info("✅ Ollama 服务已运行")
        return True
    return start_ollama()

# ============ HTTP 处理器 ============
class LocalAIProxyHandler(BaseHTTPRequestHandler):
    
    def log_message(self, format, *args):
        logger.info(f"{self.address_string()} - {format % args}")
    
    def send_cors_headers(self):
        self.send_header("Access-Control-Allow-Origin", "*")
        self.send_header("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type, Authorization")
    
    def do_OPTIONS(self):
        self.send_response(200)
        self.send_cors_headers()
        self.end_headers()
    
    def do_GET(self):
        if self.path == "/v1/models":
            self.handle_models()
        elif self.path in ["/", "/health", "/status"]:
            self.handle_health()
        else:
            self.send_error(404)
    
    def do_POST(self):
        if self.path == "/v1/chat/completions":
            self.handle_chat()
        else:
            self.send_error(404)
    
    def handle_health(self):
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()
        self.end_headers()
        
        response = {
            "status": "running",
            "service": "Local AI Proxy",
            "version": "1.0.0",
            "backend": "Ollama (qwen2.5:1.5b)",
            "ollama_status": "connected" if check_ollama() else "disconnected",
            "pid": os.getpid()
        }
        self.wfile.write(json.dumps(response).encode())
    
    def handle_models(self):
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()
        self.end_headers()
        
        models = [
            {"id": "gpt-4", "object": "model", "owned_by": "local"},
            {"id": "gpt-4o", "object": "model", "owned_by": "local"},
            {"id": "claude-3.7-sonnet", "object": "model", "owned_by": "local"},
            {"id": "deepseek-chat", "object": "model", "owned_by": "local"},
        ]
        
        self.wfile.write(json.dumps({"object": "list", "data": models}).encode())
    
    def handle_chat(self):
        if not check_ollama():
            if not start_ollama():
                self.send_error(503, "Ollama service unavailable")
                return
        
        try:
            content_length = int(self.headers.get("Content-Length", 0))
            body = self.rfile.read(content_length)
            request_data = json.loads(body)
            
            model = request_data.get("model", "default")
            ollama_model = MODEL_MAPPING.get(model, MODEL_MAPPING["default"])
            
            logger.info(f"📤 请求: {model} -> {ollama_model}")
            
            # 转换为 Ollama 格式
            ollama_request = {
                "model": ollama_model,
                "messages": request_data.get("messages", []),
                "stream": request_data.get("stream", False),
                "options": {
                    "temperature": request_data.get("temperature", 0.7),
                    "num_predict": request_data.get("max_tokens", 2048),
                }
            }
            
            is_stream = request_data.get("stream", False)
            
            req = Request(
                f"{OLLAMA_HOST}/api/chat",
                data=json.dumps(ollama_request).encode(),
                headers={"Content-Type": "application/json"},
                method="POST"
            )
            
            response = urlopen(req, timeout=180)
            
            if is_stream:
                self.send_response(200)
                self.send_header("Content-Type", "text/event-stream")
                self.send_header("Cache-Control", "no-cache")
                self.send_cors_headers()
                self.end_headers()
                
                for line in response:
                    try:
                        chunk = json.loads(line)
                        # 转换为 OpenAI 格式
                        openai_chunk = {
                            "id": "chatcmpl-local",
                            "object": "chat.completion.chunk",
                            "model": model,
                            "choices": [{
                                "index": 0,
                                "delta": {"content": chunk.get("message", {}).get("content", "")},
                                "finish_reason": "stop" if chunk.get("done") else None
                            }]
                        }
                        self.wfile.write(f"data: {json.dumps(openai_chunk)}\n\n".encode())
                        self.wfile.flush()
                    except:
                        pass
                
                self.wfile.write(b"data: [DONE]\n\n")
            else:
                result = json.loads(response.read())
                
                # 转换为 OpenAI 格式
                openai_response = {
                    "id": "chatcmpl-local",
                    "object": "chat.completion",
                    "model": model,
                    "choices": [{
                        "index": 0,
                        "message": {
                            "role": "assistant",
                            "content": result.get("message", {}).get("content", "")
                        },
                        "finish_reason": "stop"
                    }],
                    "usage": {
                        "prompt_tokens": result.get("prompt_eval_count", 0),
                        "completion_tokens": result.get("eval_count", 0),
                        "total_tokens": result.get("prompt_eval_count", 0) + result.get("eval_count", 0)
                    }
                }
                
                self.send_response(200)
                self.send_header("Content-Type", "application/json")
                self.send_cors_headers()
                self.end_headers()
                
                self.wfile.write(json.dumps(openai_response).encode())
            
            logger.info("✅ 请求完成")
            
        except HTTPError as e:
            logger.error(f"Ollama API 错误: {e}")
            self.send_error(e.code, str(e))
        except Exception as e:
            logger.error(f"处理请求失败: {e}")
            self.send_error(500, str(e))

class TraeTokenManager:
    def __init__(self):
        self.token = None
        self.device_id = None
        self.machine_id = None
        self.user_id = None
        self.username = None
        self.load()
    
    def load(self):
        try:
            if not os.path.exists(TRAE_STORAGE_PATH):
                logger.warning("Trae 配置文件不存在")
                return False
            
            with open(TRAE_STORAGE_PATH, 'r', encoding='utf-8') as f:
                storage = json.load(f)
            
            # 获取认证信息
            auth_str = storage.get("iCubeAuthInfo://icube.cloudide", "{}")
            auth = json.loads(auth_str)
            
            self.token = auth.get("token")
            self.user_id = auth.get("userId")
            self.username = auth.get("account", {}).get("username", "未知")
            
            # 获取设备信息
            self.machine_id = storage.get("telemetry.machineId", "")
            self.device_id = storage.get("telemetry.devDeviceId", "")
            
            if self.token:
                logger.info(f"✅ Trae 认证加载成功 - 用户: {self.username}")
                return True
            
            return False
        except Exception as e:
            logger.error(f"加载 Trae 认证失败: {e}")
            return False
    
    def get_headers(self):
        """获取 Trae API 请求头"""
        return {
            "Content-Type": "application/json",
            "Authorization": f"Bearer {self.token}",
            "X-Device-Id": self.device_id or str(uuid.uuid4().int)[:18],
            "X-Machine-Id": self.machine_id or hashlib.sha256(os.uname().nodename.encode()).hexdigest(),
            "X-Device-Type": "mac",
            "X-Os-Version": "macOS 26.2",
            "X-Device-Cpu": "Apple",
            "X-Device-Brand": "Mac",
            "x-ide-version-code": "20251222",
            "User-Agent": "TraeClient/3.3.25",
        }

def call_trae(messages, model="claude-3-7-sonnet", stream=False, **kwargs):
    """调用 Trae API"""
    if not trae_manager.token:
        trae_manager.load()
        if not trae_manager.token:
            raise Exception("Trae 未登录，请先打开 Trae CN 并登录")
    
    trae_request = {
        "model": model,
        "messages": messages,
        "stream": stream,
        "temperature": kwargs.get("temperature", 0.7),
        "max_tokens": kwargs.get("max_tokens", 4096),
    }
    
    # 尝试多个可能的端点
    endpoints = [
        "/api/ide/v1/chat/completions",
        "/api/ai/chat/completions",
        "/v1/chat/completions",
    ]
    
    last_error = None
    for endpoint in endpoints:
        try:
            url = f"{TRAE_API_HOST}{endpoint}"
            req = Request(
                url,
                data=json.dumps(trae_request).encode(),
                headers=trae_manager.get_headers(),
                method="POST"
            )
            
            response = urlopen(req, timeout=120)
            result = json.loads(response.read())
            
            # 转换为 OpenAI 格式
            if "choices" not in result:
                result = {
                    "id": f"chatcmpl-trae-{int(time.time())}",
                    "object": "chat.completion",
                    "model": model,
                    "choices": [{
                        "index": 0,
                        "message": {
                            "role": "assistant",
                            "content": result.get("content", result.get("message", {}).get("content", str(result)))
                        },
                        "finish_reason": "stop"
                    }]
                }
            
            logger.info(f"✅ Trae API 调用成功 (endpoint: {endpoint})")
            return result
            
        except HTTPError as e:
            last_error = f"HTTP {e.code}: {e.read().decode()[:200]}"
            logger.warning(f"Trae 端点 {endpoint} 失败: {last_error}")
        except Exception as e:
            last_error = str(e)
            logger.warning(f"Trae 端点 {endpoint} 异常: {e}")
    
    raise Exception(f"Trae API 调用失败: {last_error}")

# ============ HTTP 处理器 ============
class UnifiedProxyHandler(BaseHTTPRequestHandler):
    
    def log_message(self, format, *args):
        logger.info(f"{self.address_string()} - {format % args}")
    
    def send_cors_headers(self):
        self.send_header("Access-Control-Allow-Origin", "*")
        self.send_header("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type, Authorization")
    
    def do_OPTIONS(self):
        self.send_response(200)
        self.send_cors_headers()
        self.end_headers()
    
    def do_GET(self):
        if self.path == "/v1/models":
            self.handle_models()
        elif self.path in ["/", "/health", "/status"]:
            self.handle_health()
        else:
            self.send_error(404)
    
    def do_POST(self):
        if self.path == "/v1/chat/completions":
            self.handle_chat()
        else:
            self.send_error(404)
    
    def handle_health(self):
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()
        self.end_headers()
        
        response = {
            "status": "running",
            "service": "Unified AI Proxy",
            "version": "1.0.0",
            "backends": {
                "trae": {
                    "status": "connected" if trae_manager.token else "disconnected",
                    "user": trae_manager.username,
                    "models": ["claude-3.7-sonnet", "claude-3.5-sonnet", "gpt-4o", "gemini"]
                },
                "ollama": {
                    "status": "connected" if check_ollama() else "disconnected",
                    "models": ["qwen2.5:1.5b"]
                }
            },
            "pid": os.getpid()
        }
        self.wfile.write(json.dumps(response, ensure_ascii=False).encode())
    
    def handle_models(self):
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_cors_headers()
        self.end_headers()
        
        models = [
            # Trae 模型
            {"id": "trae-claude-3.7", "object": "model", "owned_by": "trae", "description": "Claude 3.7 Sonnet via Trae (免费)"},
            {"id": "trae-claude-3.5", "object": "model", "owned_by": "trae", "description": "Claude 3.5 Sonnet via Trae (免费)"},
            {"id": "trae-gpt-4o", "object": "model", "owned_by": "trae", "description": "GPT-4o via Trae (免费)"},
            {"id": "claude-3.7-sonnet", "object": "model", "owned_by": "trae", "description": "Claude 3.7 (Trae)"},
            {"id": "gpt-4", "object": "model", "owned_by": "trae", "description": "GPT-4 (Trae)"},
            # 本地模型
            {"id": "local-qwen", "object": "model", "owned_by": "local", "description": "Qwen 2.5 1.5B (本地)"},
        ]
        
        self.wfile.write(json.dumps({"object": "list", "data": models}).encode())
    
    def handle_chat(self):
        try:
            content_length = int(self.headers.get("Content-Length", 0))
            body = self.rfile.read(content_length)
            request_data = json.loads(body)
            
            model = request_data.get("model", "default")
            messages = request_data.get("messages", [])
            
            # 路由到对应后端
            backend, actual_model = MODEL_ROUTES.get(model, MODEL_ROUTES["default"])
            
            logger.info(f"📤 请求: {model} -> {backend}:{actual_model}")
            
            try:
                if backend == "trae":
                    result = call_trae(
                        messages, 
                        actual_model,
                        stream=request_data.get("stream", False),
                        temperature=request_data.get("temperature", 0.7),
                        max_tokens=request_data.get("max_tokens", 4096)
                    )
                else:  # ollama
                    if not check_ollama():
                        # 如果本地模型不可用，回退到 Trae
                        logger.warning("本地 Ollama 不可用，回退到 Trae API")
                        result = call_trae(messages, "claude-3-7-sonnet")
                    else:
                        result = call_ollama(
                            messages,
                            actual_model,
                            stream=request_data.get("stream", False),
                            temperature=request_data.get("temperature", 0.7),
                            max_tokens=request_data.get("max_tokens", 2048)
                        )
                
                self.send_response(200)
                self.send_header("Content-Type", "application/json")
                self.send_cors_headers()
                self.end_headers()
                self.wfile.write(json.dumps(result).encode())
                
                logger.info(f"✅ 请求完成 ({backend})")
                
            except Exception as e:
                logger.error(f"后端调用失败: {e}")
                
                # 尝试回退
                if backend == "trae" and check_ollama():
                    logger.info("Trae 失败，回退到本地模型")
                    result = call_ollama(messages, "qwen2.5:1.5b")
                    self.send_response(200)
                    self.send_header("Content-Type", "application/json")
                    self.send_cors_headers()
                    self.end_headers()
                    self.wfile.write(json.dumps(result).encode())
                else:
                    self.send_error(503, str(e))
                
        except Exception as e:
            logger.error(f"处理请求失败: {e}")
            self.send_error(500, str(e))
