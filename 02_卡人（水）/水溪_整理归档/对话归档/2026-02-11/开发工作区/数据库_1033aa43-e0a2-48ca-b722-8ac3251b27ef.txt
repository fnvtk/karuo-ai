 * 知己 · 统一日期为 2026-01-13 以后
 * POST: 将用户注册时间、测试日期等全部改为 2026年1月13日之后的随机日期

import { NextResponse } from "next/server"
import { connectToDatabase, COLLECTIONS } from "@/lib/mongodb"

const START = new Date("2026-01-14T00:00:00.000Z").getTime()
const END = new Date("2026-02-11T23:59:59.999Z").getTime()

function randomDateAfterJan13(): string {
  return new Date(START + Math.random() * (END - START)).toISOString()
}

export async function POST() {
  try {
    const { db } = await connectToDatabase()
    const usersCol = db.collection(COLLECTIONS.USERS)
    const testCol = db.collection(COLLECTIONS.TEST_RECORDS)

    const users = await usersCol.find({}).toArray()
    const testRecords = await testCol.find({}).toArray()

    let usersUpdated = 0
    for (const u of users) {
      const createdAt = randomDateAfterJan13()
      const updatedAt = randomDateAfterJan13()
      const lastTestAt = u.lastTestAt ? randomDateAfterJan13() : undefined
      await usersCol.updateOne(
        { _id: u._id },
        { $set: { createdAt, updatedAt, ...(lastTestAt && { lastTestAt }) } }
      )
      usersUpdated++
    }

    let testsUpdated = 0
    for (const t of testRecords) {
      const createdAt = randomDateAfterJan13()
      await testCol.updateOne(
        { _id: t._id },
        { $set: { createdAt } }
      )
      testsUpdated++
    }

    return NextResponse.json({
      success: true,
      data: { usersUpdated, testsUpdated, message: "全部日期已统一为 2026-01-13 之后" },
    })
  } catch (error: unknown) {
    console.error("[知己Admin] fix-dates 失败:", error)
    return NextResponse.json({ success: false, error: (error as Error).message }, { status: 500 })
  }
}
