def _now_hms() -> str:
    return time.strftime("%H:%M:%S", time.localtime())


def _write_json(obj: dict) -> None:
    tmp = f"{PROGRESS_FILE}.tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(obj, f, ensure_ascii=False, indent=2)
    os.replace(tmp, PROGRESS_FILE)


def _append_log(line: str) -> None:
    with open(LOG_FILE, "a", encoding="utf-8") as f:
        f.write(f"[{_now_hms()}] {line}\n")


def _human_bytes(n: float) -> str:
    units = ["B", "KB", "MB", "GB", "TB"]
    u = 0
    while n >= 1024 and u < len(units) - 1:
        n /= 1024.0
        u += 1
    return f"{n:.2f}{units[u]}" if u > 0 else f"{int(n)}{units[u]}"


def _human_speed(bps: float) -> str:
    return f"{_human_bytes(bps)}/s" if bps > 0 else "--"


def _human_eta(seconds: float) -> str:
    if seconds <= 0 or seconds == float("inf"):
        return "--"
    seconds = int(seconds)
    h, m, s = seconds // 3600, (seconds % 3600) // 60, seconds % 60
    return f"{h}h{m:02d}m{s:02d}s" if h > 0 else f"{m}m{s:02d}s"


def _get_local_size(path: str) -> int:
    """获取本地目录总大小（使用 du）"""
    try:
        result = subprocess.run(
            ["du", "-sk", path],
            capture_output=True,
            text=True,
            timeout=30
        )
        if result.returncode == 0:
            size_kb = int(result.stdout.split()[0])
            return size_kb * 1024
    except Exception:
        pass
    return 0


def _get_remote_size(ssh_user: str, ssh_host: str, remote_path: str) -> int:
    """获取远端目录大小"""
    try:
        cmd = f"ssh -o ConnectTimeout=5 {ssh_user}@{ssh_host} 'du -sk \"{remote_path}\" 2>/dev/null | cut -f1'"
        result = subprocess.run(
            cmd,
            shell=True,
            capture_output=True,
            text=True,
            timeout=15
        )
        if result.returncode == 0 and result.stdout.strip():
            size_kb = int(result.stdout.strip())
            return size_kb * 1024
    except Exception:
        pass
    return 0


def _is_rsync_running() -> bool:
    """检查 rsync 是否在运行"""
    try:
        result = subprocess.run(
            ["ps", "aux"],
            capture_output=True,
            text=True,
            timeout=5
        )
        return "rsync" in result.stdout and "照片图库.photoslibrary" in result.stdout
    except Exception:
        return False


def _monitor_loop(ssh_user: str, ssh_host: str, local_path: str, remote_path: str):
    """监控循环"""
    total = _get_local_size(local_path)
    _append_log(f"本地相册: {local_path}")
    _append_log(f"远端目录: {ssh_user}@{ssh_host}:{remote_path}")
    _append_log(f"总大小: {_human_bytes(total)}")
    
    last_bytes = 0
    last_time = time.time()
    ema_bps = None

    while True:
        try:
            remote_bytes = _get_remote_size(ssh_user, ssh_host, remote_path)
            now = time.time()
            dt = max(now - last_time, 1e-6)
            db = max(remote_bytes - last_bytes, 0)
            inst_bps = db / dt

            if ema_bps is None:
                ema_bps = inst_bps
            else:
                ema_bps = 0.25 * inst_bps + 0.75 * ema_bps

            percent = int(min(100, (remote_bytes / total) * 100)) if total > 0 else 0
            remaining = max(total - remote_bytes, 0)
            eta = (remaining / ema_bps) if ema_bps > 0 else float("inf")

            running = _is_rsync_running()
            
            if percent >= 100 or remote_bytes >= total:
                stage = "完成"
                message = "相册上传完成"
            elif running:
                stage = "上传中"
                message = "正在传输相册文件..."
            else:
                stage = "等待"
                message = "未检测到传输进程"

            progress = {
                "stage": stage,
                "percent": percent,
                "message": message,
                "timestamp": _now_hms(),
                "connection": "局域网",
                "host": f"{ssh_host}",
                "details": {
                    "transferred": _human_bytes(remote_bytes),
                    "total": _human_bytes(total),
                    "speed": _human_speed(ema_bps or 0),
                    "eta": _human_eta(eta),
                    "current_file": "正在扫描文件..." if running else "--"
                }
            }
            _write_json(progress)

            # 每30秒记录一次日志
            if int(last_time) // 30 != int(now) // 30:
                _append_log(
                    f"{stage} {percent}% | "
                    f"{_human_bytes(remote_bytes)}/{_human_bytes(total)} | "
                    f"{_human_speed(ema_bps or 0)} | ETA {_human_eta(eta)}"
                )

            last_bytes = remote_bytes
            last_time = now

        except Exception as e:
            _append_log(f"❌ 监控异常: {e}")
            _write_json({
                "stage": "错误",
                "percent": 0,
                "message": f"监控异常: {e}",
                "timestamp": _now_hms(),
                "connection": "局域网",
                "host": ssh_host,
                "details": {
                    "transferred": "--",
                    "total": "--",
                    "speed": "--",
                    "eta": "--",
                    "current_file": "--"
                }
            })

        time.sleep(3)

# 群晖 NAS 矩阵管理 Skill

专业的群晖 NAS 管理助手，提供自动化部署、实时监控、数据迁移、远程访问等全方位解决方案。

## 核心能力

### 1. NAS 连接与管理
- **快速连接**: 自动连接到 NAS-2 (192.168.2.201)
- **SSH 管理**: 用户 `fnvtk`，支持免密登录
- **多 NAS 管理**: 支持矩阵化管理多台 NAS 设备
- **状态监控**: 实时监控 NAS 运行状态、资源使用情况

### 2. 数据传输与同步
- **大文件传输**: 支持 >10GB 文件传输，自动启动监控服务（端口 8890）
- **增量同步**: 使用 rsync 实现高效增量备份
- **断点续传**: 传输中断后可自动恢复
- **实时监控**: 提供 Web 界面显示传输进度、速度、ETA

#### 传输监控规范
所有大于 10GB 的数据传输必须：
1. 启动监控服务（端口 8890）
2. 使用 rsync 解析进度写入 `/tmp/xxx_progress.json`
3. 提供实时监控页面：进度百分比、传输速度、预计剩余时间、当前文件名
4. 深色渐变背景，自动刷新间隔 2 秒
5. 脚本位置：
   - 监控面板: `群晖NAS矩阵管理/07_监控面板/xxx_monitor.py`
   - 传输脚本: `群晖NAS矩阵管理/01_部署自动化/xxx_copy.sh`

### 3. MongoDB 数据库管理
- **数据迁移**: 从本地到 NAS 的 MongoDB 迁移
- **实时同步**: 增量同步、副本集配置
- **性能优化**: WiredTiger 配置优化
- **外网访问**: 端口映射配置（默认 27017）

### 4. 网络与远程访问
- **端口映射配置**: 
  - H3C 路由器（光猫）: Telnet 自动配置
  - 小米路由器: Web 界面自动配置
  - DMZ 模式配置
- **双重 NAT 解决**: 最优 DMZ 配置方案
- **QuickConnect**: 外网访问备份方案
- **SSH 端口**: 22 (内网), 22022 (外网)

### 5. 虚拟机与远程桌面
- **宝塔中控面板**: 虚拟机内安装宝塔，统一管理多台服务器
- **Windows 虚拟机**: 远程桌面部署与管理
- **Android 虚拟机**: Docker 部署移动应用测试环境
- **资源分配**: CPU、内存、存储自动化配置

### 6. 服务部署
- **Docker Compose**: 一键部署常用服务
  - MongoDB
  - Git Server
  - 向量数据库
  - 监控服务
- **自动化脚本**: 完整的部署流程自动化
- **配置备份**: 自动备份系统配置

    return time.strftime("%H:%M:%S", time.localtime())


def _write_json(obj):
    tmp = f"{PROGRESS_FILE}.tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(obj, f, ensure_ascii=False, indent=2)
    os.replace(tmp, PROGRESS_FILE)


def _append_log(line):
    with open(LOG_FILE, "a", encoding="utf-8") as f:
        f.write(f"[{_now_hms()}] {line}\n")


def _human_bytes(n):
    units = ["B", "KB", "MB", "GB", "TB"]
    u = 0
    while n >= 1024 and u < len(units) - 1:
        n /= 1024.0
        u += 1
    return f"{n:.2f}{units[u]}" if u > 0 else f"{int(n)}{units[u]}"


def _human_speed(bps):
    return f"{_human_bytes(bps)}/s" if bps > 0 else "--"


def _human_eta(seconds):
    if seconds <= 0 or seconds == float("inf"):
        return "--"
    seconds = int(seconds)
    h, m, s = seconds // 3600, (seconds % 3600) // 60, seconds % 60
    return f"{h}h{m:02d}m{s:02d}s" if h > 0 else f"{m}m{s:02d}s"


def _get_remote_size(ssh_user, ssh_host, remote_path):
    """获取远端目录/文件大小"""
    try:
        cmd = f"ssh -o ConnectTimeout=5 {ssh_user}@{ssh_host} 'du -sk \"{remote_path}\" 2>/dev/null | cut -f1'"
        result = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=15)
        if result.returncode == 0 and result.stdout.strip():
            return int(result.stdout.strip()) * 1024
    except Exception:
        pass
    return 0


def _is_rsync_running(match_keyword):
    """检查 rsync 是否在运行"""
    try:
        result = subprocess.run(["ps", "aux"], capture_output=True, text=True, timeout=5)
        return "rsync" in result.stdout and match_keyword in result.stdout
    except Exception:
        return False


def _monitor_loop(ssh_user, ssh_host, local_path, remote_path, total_size, match_keyword):
    """监控循环"""
    _append_log(f"本地路径: {local_path}")
    _append_log(f"远端路径: {ssh_user}@{ssh_host}:{remote_path}")
    _append_log(f"总大小: {_human_bytes(total_size)}")
    
    last_bytes = 0
    last_time = time.time()
    ema_bps = None

    while True:
        try:
            remote_bytes = _get_remote_size(ssh_user, ssh_host, remote_path)
            now = time.time()
            dt = max(now - last_time, 1e-6)
            db = max(remote_bytes - last_bytes, 0)
            inst_bps = db / dt

            if ema_bps is None:
                ema_bps = inst_bps
            else:
                ema_bps = 0.25 * inst_bps + 0.75 * ema_bps

            percent = int(min(100, (remote_bytes / total_size) * 100)) if total_size > 0 else 0
            remaining = max(total_size - remote_bytes, 0)
            eta = (remaining / ema_bps) if ema_bps > 0 else float("inf")

            running = _is_rsync_running(match_keyword)
            
            if percent >= 100 or remote_bytes >= total_size:
                stage = "完成"
                message = "传输完成"
            elif running:
                stage = "上传中"
                message = "正在传输文件..."
            else:
                stage = "等待"
                message = "未检测到传输进程"

            progress = {
                "stage": stage,
                "percent": percent,
                "message": message,
                "timestamp": _now_hms(),
                "connection": "局域网",
                "host": ssh_host,
                "details": {
                    "transferred": _human_bytes(remote_bytes),
                    "total": _human_bytes(total_size),
                    "speed": _human_speed(ema_bps or 0),
                    "eta": _human_eta(eta),
                    "current_file": "传输中..." if running else "--"
                }
            }
            _write_json(progress)

            if int(last_time) // 30 != int(now) // 30:
                _append_log(
                    f"{stage} {percent}% | {_human_bytes(remote_bytes)}/{_human_bytes(total_size)} | "
                    f"{_human_speed(ema_bps or 0)} | ETA {_human_eta(eta)}"
                )

            last_bytes = remote_bytes
            last_time = now

        except Exception as e:
            _append_log(f"❌ 监控异常: {e}")

        time.sleep(3)

def main():
    parser = argparse.ArgumentParser(description="NAS 传输监控服务")
    parser.add_argument("--port", type=int, default=8890, help="监控服务端口")
    parser.add_argument("--ssh-user", required=True, help="SSH 用户名")
    parser.add_argument("--ssh-host", required=True, help="SSH 主机地址")
    parser.add_argument("--local-path", required=True, help="本地文件/目录路径")
    parser.add_argument("--remote-path", required=True, help="远端文件/目录路径")
    parser.add_argument("--total-size", type=int, required=True, help="总大小（字节）")
    parser.add_argument("--match-keyword", default="rsync", help="进程匹配关键词")
    args = parser.parse_args()

    # 初始化
    _write_json({
        "stage": "等待",
        "percent": 0,
        "message": "监控服务已启动...",
        "timestamp": _now_hms(),
        "connection": "局域网",
        "host": args.ssh_host,
        "details": {"transferred": "--", "total": "--", "speed": "--", "eta": "--", "current_file": "--"}
    })

    # 启动监控线程
    t = threading.Thread(
        target=_monitor_loop,
        args=(args.ssh_user, args.ssh_host, args.local_path, args.remote_path, args.total_size, args.match_keyword),
        daemon=True
    )
    t.start()

    # 启动 HTTP 服务
    with TCPServer(("", args.port), Handler) as httpd:
        print(f"✅ 监控服务启动: http://localhost:{args.port}")
        _append_log(f"监控地址: http://localhost:{args.port}")
        httpd.serve_forever()


if __name__ == "__main__":
    main()

## 提示与技巧

### 💡 技巧 1: 后台传输
大文件传输会自动在后台运行，你可以：
- 关闭监控页面（传输继续）
- 随时重新打开: http://localhost:8890
- 查看日志: `tail -f /tmp/transfer_upload.log`

### 💡 技巧 2: 断点续传
如果传输中断：
- 重新运行相同命令
- rsync 会自动跳过已传输文件
- 从断点处继续传输

### 💡 技巧 3: 多任务并行
可以同时进行多个操作：
- 一个终端传输文件
- 另一个终端查看状态
- 监控页面实时显示进度

### 💡 技巧 4: 自然语言
使用自然语言描述需求：
```
你: 我想把电脑上的照片备份到 NAS，大概有 200GB，需要多久？
```

AI 会自动理解并提供准确答案。

## 常见问题

### Q: 监控页面打不开？
A: 检查端口 8890 是否被占用：
```bash
lsof -i :8890
```

### Q: 传输速度很慢？
A: 检查是否在用 WiFi，建议使用有线连接。

### Q: SSH 连接失败？
A: 配置 SSH 密钥：
```bash
ssh-copy-id fnvtk@192.168.2.201
```

### Q: 如何停止传输？
A: 找到进程并终止：
```bash
ps aux | grep rsync
kill [PID]
```
