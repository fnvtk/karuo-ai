from http.server import HTTPServer, SimpleHTTPRequestHandler
from urllib.parse import parse_qs, urlparse

_next_index = 0


def load_accounts():
    """加载账号数据"""
    try:
        with open(ACCOUNTS_FILE, "r", encoding="utf-8") as f:
            data = json.load(f)
        return data.get("accounts", [])
    except Exception as e:
        print(f"⚠️ 加载账号失败: {e}")
        return []


def get_available_accounts():
    """获取可用账号（排除已取消/失效的）"""
    accounts = load_accounts()
    return [a for a in accounts if a.get("status") != "cancelled"]


def get_next_account():
    """轮询获取下一个可用账号"""
    global _next_index
    available = get_available_accounts()
    if not available:
        return None
    idx = _next_index % len(available)
    _next_index = (idx + 1) % len(available)
    return available[idx]


def format_account_output(acc, format_type="json"):
    """
    格式化账号输出
    format_type: json | line | xianyu
    - json: 完整 JSON
    - line: 账号----邮箱密码----Cursor密码（闲鱼格式）
    - xianyu: 同上，兼容闲鱼卖家格式
    """
    if format_type in ("line", "xianyu"):
        return f"{acc['account']}----{acc['email_password']}----{acc['cursor_password']}"
    return acc

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)

    def do_GET(self):

        path = parsed.path
        params = parse_qs(parsed.query)

        if path == "/":

import json
import urllib.request
import urllib.error

# 隧道默认端口（与 local_llm 5888 对应，Cursor 隧道用 5889）
TUNNEL_URL = "http://localhost:5889"


def _request(path, params=None):
    """发起 GET 请求"""
    url = f"{TUNNEL_URL}{path}"
    if params:
        qs = "&".join(f"{k}={v}" for k, v in params.items())
        url = f"{url}?{qs}"
    try:
        req = urllib.request.Request(url)
        with urllib.request.urlopen(req, timeout=5) as resp:
            return json.loads(resp.read().decode())
    except urllib.error.URLError as e:
        raise ConnectionError(f"Cursor 账号隧道未启动 (localhost:5889): {e}")


def get_next_account(format_type="json"):
    """
    获取下一个可用账号（轮询）
    format_type: json | line
    - json: 返回 dict
    - line: 返回闲鱼格式字符串 账号----邮箱密码----Cursor密码
    """
    data = _request("/api/account/next", {"format": format_type})
    if "error" in data:
        return None
    return data.get("account")


def list_accounts(format_type="json"):
    """列出所有可用账号"""
    data = _request("/api/accounts", {"format": format_type})
    return data.get("accounts", [])


def get_account_by_id(acc_id, format_type="json"):
    """按 ID 获取指定账号"""
    data = _request(f"/api/account/{acc_id}", {"format": format_type})
    if "error" in data:
        return None
    return data.get("account")


def check_tunnel():
    """检查隧道是否在线"""
    try:
        data = _request("/api/status")
        return data.get("status") == "online"
    except Exception:
        return False


__all__ = [
    "get_next_account",
    "list_accounts",
    "get_account_by_id",
    "check_tunnel",
    "TUNNEL_URL",
