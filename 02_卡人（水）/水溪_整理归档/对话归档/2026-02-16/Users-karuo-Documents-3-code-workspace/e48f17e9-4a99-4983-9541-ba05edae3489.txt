"""

from typing import Dict, Any, List
from skills.base import BaseSkill
import logging

logger = logging.getLogger(__name__)


class XianyuSkill(BaseSkill):
    """闲鱼控制技能（服务端定义，执行在设备端或 ADB）"""

    PACKAGE = "com.taobao.idlefish"
    NAME = "闲鱼"

    async def send_message(self, to_id: str, content: str, msg_type: str = "text") -> Dict[str, Any]:
        """发送闲鱼私信（设备端执行或 ADB UI 自动化）"""
        try:
            self.launch()
            self.sleep(2)
            self.click_text("消息")
            self.sleep(1)
            self.click_text("搜索")
            self.sleep(0.5)
            self.input_text(to_id)
            self.sleep(1)
            if not self.click_text(to_id):
                return {"success": False, "error": f"未找到用户: {to_id}"}
            self.sleep(1)
            self.input_text(content, clear=False)
            self.sleep(0.3)
            self.click_text("发送")
            logger.info(f"闲鱼私信发送成功: {to_id}")
            return {"success": True, "message_id": f"xy_{int(self.d.info.get('currentTime', 0))}"}
        except Exception as e:
            logger.error(f"闲鱼发送失败: {e}")
            return {"success": False, "error": str(e)}

    async def get_messages(self, limit: int = 20) -> Dict[str, Any]:
        """获取私信列表（设备端或 UI 树解析）"""
        try:
            self.launch()
            self.sleep(2)
            self.click_text("消息")
            self.sleep(1)
            ui_tree = self.get_ui_tree()
            messages = self._parse_messages_from_ui_tree(ui_tree, limit)
            return {"success": True, "messages": messages}
        except Exception as e:
            return {"success": False, "error": str(e), "messages": []}

    @staticmethod
    def _parse_messages_from_ui_tree(xml: str, limit: int) -> List[Dict[str, Any]]:
        """从 UI 树解析消息占位"""
        messages = []
        if not xml:
            return messages
        import re
        for i, m in enumerate(re.finditer(r'\btext="([^"]{1,200})"', xml)):
            if i >= limit:
                break
            text = m.group(1).strip()
            if text and text not in ("消息", "搜索", "发送", "私信"):
                messages.append({
                    "message_id": f"xy_ui_{i}",
                    "content": text,
                    "from_id": "",
                    "to_id": "",
                    "timestamp": 0,
                    "is_self": False,
                })
        return messages

Skill 聊天总线 - 各 Skill 一起执行时的共享聊天与交互
"""

import time
import logging
from typing import Dict, Any, List, Optional

logger = logging.getLogger(__name__)


class SkillChatBus:
    """Skill 间共享消息总线，同一轮执行内可读写"""

    def __init__(self):
        self._messages: List[Dict[str, Any]] = []
        self._session_id: Optional[str] = None

    def clear(self, session_id: str = None):
        """清空当前会话，开始新一轮（如新一条自然语言命令）"""
        self._messages.clear()
        self._session_id = session_id or str(int(time.time() * 1000))

    def append(
        self,
        from_skill: str,
        message: str,
        to_skill: Optional[str] = None,
        data: Optional[Dict[str, Any]] = None,
    ):
        """Skill 或执行器发一条消息，其它 Skill 可读"""
        self._messages.append({
            "from_skill": from_skill,
            "to_skill": to_skill,
            "message": message,
            "data": data or {},
            "ts": time.time(),
            "index": len(self._messages),
        })
        logger.debug(f"[SkillBus] {from_skill} -> {to_skill or 'all'}: {message[:50]}")

    def get_messages(
        self,
        since_index: int = 0,
        from_skill: Optional[str] = None,
        to_skill: Optional[str] = None,
    ) -> List[Dict[str, Any]]:
        """取消息列表，可按发送方/接收方过滤"""
        out = []
        for m in self._messages:
            if m["index"] < since_index:
                continue
            if from_skill and m.get("from_skill") != from_skill:
                continue
            if to_skill and m.get("to_skill") and m.get("to_skill") != to_skill:
                continue
            out.append(m)
        return out

    def get_last(self, from_skill: Optional[str] = None) -> Optional[Dict[str, Any]]:
        """取最后一条（可选指定来自哪个 Skill）"""
        for m in reversed(self._messages):
            if from_skill is None or m.get("from_skill") == from_skill:
                return m
        return None

    def session_id(self) -> Optional[str]:
        return self._session_id
