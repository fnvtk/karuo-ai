# 1) 生成纪要图片（Pillow + 中文字体）
from PIL import Image, ImageDraw, ImageFont
import os

text = open("/tmp/soul88_summary.md", "r", encoding="utf-8").read()
# 去掉 markdown 标题符，保留换行
lines = text.replace("**", "").replace("#", "").strip().split("\n")
# 每行约 36 字宽，约 22 行/屏，总高度按行数
font_path = "/System/Library/Fonts/PingFang.ttc"
if not os.path.exists(font_path):
    font_path = "/System/Library/Fonts/Supplemental/Songti.ttc"
try:
    font = ImageFont.truetype(font_path, 22)
except Exception:
    font = ImageFont.load_default()

# 画布：宽 720，高按行数估算（约 28 像素/行）
line_h = 28
w = 720
h = max(1200, len(lines) * line_h + 80)
img = Image.new("RGB", (w, h), color=(255, 255, 255))
draw = ImageDraw.Draw(img)

y = 40
for line in lines:
    if not line.strip():
        y += line_h
        continue
    # 简单折行：约 32 个字符一行
    chunk = 32
    for i in range(0, len(line), chunk):
        draw.text((40, y), line[i:i+chunk], fill=(0, 0, 0), font=font)
        y += line_h
    y += 4
# 裁剪到实际内容高度
img = img.crop((0, 0, w, min(y + 40, h)))
img.save("/tmp/soul88_summary.png")
print("OK image saved", img.size)
