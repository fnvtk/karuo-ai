# 验证码提取正则（按优先级）
CODE_PATTERNS = [
    r'验证码[：:]\s*(\d{4,8})',
    r'动态码[：:]\s*(\d{4,8})',
    r'校验码[：:]\s*(\d{4,8})',

    r'\b(\d{4,8})\b',  # 4-8位数字兜底

def _decode_str(s):
    """解码邮件头/正文"""
    if s is None:
        return ""
    if isinstance(s, str):
        return s
    try:
        parts = decode_header(s)
        result = []
        for part, enc in parts:
            if isinstance(part, bytes):
                result.append(part.decode(enc or "utf-8", errors="replace"))
            else:
                result.append(str(part))
        return "".join(result)
    except Exception:
        return str(s)


def _get_credential_from_index():
    """从金盾 账号与API索引.md 读取邮箱凭证（本机）"""

def get_credentials():
    """获取邮箱凭证：优先环境变量，其次索引"""

    auth_code = os.environ.get("EMAIL_QQ_AUTH_CODE")

def extract_codes(text):
    """从文本中提取验证码"""
    codes = []
    seen = set()
    for pat in CODE_PATTERNS:
        for m in re.finditer(pat, text, re.IGNORECASE):
            code = m.group(1)
            if code not in seen and 4 <= len(code) <= 8:
                seen.add(code)
                codes.append(code)
    return codes


def fetch_emails(account, auth_code, folder="INBOX", limit=20, since_days=None):
    """通过 IMAP 拉取邮件"""
    mail = imaplib.IMAP4_SSL(IMAP_HOST, IMAP_PORT)
    try:
        mail.login(account, auth_code)
        mail.select(folder, readonly=True)
        _, data = mail.search(None, "ALL")
        msg_ids = data[0].split()
        if not msg_ids:
            return []
        msg_ids = msg_ids[-limit:]  # 最近 N 封
        emails = []
        for msg_id in reversed(msg_ids):
            _, msg_data = mail.fetch(msg_id, "(RFC822)")
            raw = msg_data[0][1]
            msg = email.message_from_bytes(raw)
            subject = _decode_str(msg.get("Subject", ""))
            from_addr = _decode_str(msg.get("From", ""))
            date_str = msg.get("Date", "")
            body = ""
            if msg.is_multipart():
                for part in msg.walk():
                    ctype = part.get_content_type()
                    if ctype == "text/plain":
                        try:
                            body = part.get_payload(decode=True).decode("utf-8", errors="replace")
                        except Exception:
                            body = str(part.get_payload())
                        break
            else:
                try:
                    body = msg.get_payload(decode=True).decode("utf-8", errors="replace")
                except Exception:
                    body = str(msg.get_payload())
            emails.append({
                "subject": subject,
                "from": from_addr,
                "date": date_str,
                "body": body,
            })
        return emails
    finally:
        try:
            mail.logout()
        except Exception:
            pass
