// 卡若智能白板 - 简化版分享模态框
// 只保留分享和导出两个核心功能

import { useState, useEffect } from "react"
import { X, Share2, Download, Copy, Check, QrCode, ImageIcon, FileJson, Loader2 } from "lucide-react"
import { useWhiteboardStore } from "@/lib/whiteboard-store"
import { exportCanvasToImage, downloadFile, generateShareData } from "@/lib/export-utils"
import { cn } from "@/lib/utils"

export function ShareModal() {
  const { showShareModal, setShowShareModal, session, exportSession } = useWhiteboardStore()
  const [copied, setCopied] = useState(false)
  const [shareUrl, setShareUrl] = useState("")
  const [qrCodeUrl, setQrCodeUrl] = useState("")
  const [exporting, setExporting] = useState(false)

  // 生成分享链接和二维码
  useEffect(() => {
    if (showShareModal) {
      const shareData = generateShareData(session)
      const url = `${window.location.origin}/share/${shareData}`
      setShareUrl(url)
      // 使用第三方API生成二维码
      setQrCodeUrl(`https://api.qrserver.com/v1/create-qr-code/?size=200x200&data=${encodeURIComponent(url)}`)
    }
  }, [showShareModal, session])

  if (!showShareModal) return null

  // 复制分享链接
  const handleCopyLink = async () => {
    await navigator.clipboard.writeText(shareUrl)
    setCopied(true)
    setTimeout(() => setCopied(false), 2000)
  }

  // 导出当前页截图
  const handleExportCurrentPage = async () => {
    setExporting(true)
    const currentPage = session.pages[session.currentPageIndex]
    const blob = await exportCanvasToImage(currentPage)
    if (blob) {
      downloadFile(blob, `${session.name}-第${session.currentPageIndex + 1}页.png`)
    }
    setExporting(false)
  }

  // 导出所有页截图
  const handleExportAllPages = async () => {
    setExporting(true)
    for (let i = 0; i < session.pages.length; i++) {
      const page = session.pages[i]
      const blob = await exportCanvasToImage(page)
      if (blob) {
        setTimeout(() => {
          downloadFile(blob, `${session.name}-第${i + 1}页.png`)
        }, i * 500)
      }
    }
    setExporting(false)
  }

  // 导出JSON数据
  const handleExportJSON = () => {
    const data = exportSession()
    const blob = new Blob([data], { type: "application/json" })
    downloadFile(blob, `${session.name}.json`)
  }

  return (
    <div className="fixed inset-0 z-[100] flex items-center justify-center bg-background/80 backdrop-blur-sm">
      <div className="glass-panel w-full max-w-2xl mx-4 rounded-2xl overflow-hidden animate-in fade-in zoom-in-95 duration-200">
        {/* 头部 */}
        <div className="flex items-center justify-between p-4 border-b border-border/50">
          <h2 className="text-lg font-semibold">分享与导出</h2>
          <button
            onClick={() => setShowShareModal(false)}
            className="p-2 rounded-lg hover:bg-secondary/50 transition-colors"
          >
            <X className="w-5 h-5" />
          </button>
        </div>

        <div className="p-6 space-y-6">
          {/* 分享区域 */}
          <div className="space-y-4">
            <div className="flex items-center gap-2">
              <Share2 className="w-5 h-5 text-primary" />
              <h3 className="text-base font-medium">分享链接</h3>
            </div>

            {/* 链接卡片 */}
            <div className="bg-secondary/30 rounded-xl p-4 space-y-3">
              <div className="flex gap-2">
                <div className="flex-1 bg-background rounded-lg px-3 py-2 text-sm truncate font-mono border">
                  {shareUrl}
                </div>
                <button
                  onClick={handleCopyLink}
                  className={cn(
                    "px-4 py-2 rounded-lg flex items-center gap-2 transition-colors",
                    copied ? "bg-green-500 text-white" : "bg-primary text-primary-foreground hover:bg-primary/90",
                  )}
                >
                  {copied ? (
                    <>
                      <Check className="w-4 h-4" />
                      已复制
                    </>
                  ) : (
                    <>
                      <Copy className="w-4 h-4" />
                      复制
                    </>
                  )}
                </button>
              </div>

              {/* 二维码 */}
              <div className="flex items-center gap-4">
                <div className="bg-white p-2 rounded-lg">
                  {qrCodeUrl ? (
                    <img src={qrCodeUrl} alt="分享二维码" className="w-24 h-24" />
                  ) : (
                    <div className="w-24 h-24 flex items-center justify-center">
                      <Loader2 className="w-6 h-6 animate-spin text-muted-foreground" />
                    </div>
                  )}
                </div>
                <div className="flex-1 text-sm text-muted-foreground">
                  <p className="mb-1">扫描二维码快速分享</p>
                  <p className="text-xs">
                    会议：{session.name} · {session.pages.length} 页
                  </p>
                </div>
              </div>
            </div>
          </div>

          {/* 导出区域 */}
          <div className="space-y-4">
            <div className="flex items-center gap-2">
              <Download className="w-5 h-5 text-primary" />
              <h3 className="text-base font-medium">导出文件</h3>
            </div>

            <div className="grid grid-cols-3 gap-3">
              {/* 导出当前页 */}
              <button
                onClick={handleExportCurrentPage}
                disabled={exporting}
                className="glass-button p-4 rounded-xl flex flex-col items-center gap-2 hover:bg-secondary/50 transition-colors disabled:opacity-50"
              >
                <ImageIcon className="w-6 h-6 text-blue-500" />
                <span className="text-sm font-medium">当前页</span>
                <span className="text-xs text-muted-foreground">PNG图片</span>
              </button>

              {/* 导出所有页 */}
              <button
                onClick={handleExportAllPages}
                disabled={exporting}
                className="glass-button p-4 rounded-xl flex flex-col items-center gap-2 hover:bg-secondary/50 transition-colors disabled:opacity-50"
              >
                {exporting ? (
                  <Loader2 className="w-6 h-6 text-green-500 animate-spin" />
                ) : (
                  <ImageIcon className="w-6 h-6 text-green-500" />
                )}
                <span className="text-sm font-medium">全部页面</span>
                <span className="text-xs text-muted-foreground">{session.pages.length} 张图片</span>
              </button>

              {/* 导出JSON */}
              <button
                onClick={handleExportJSON}
                disabled={exporting}
                className="glass-button p-4 rounded-xl flex flex-col items-center gap-2 hover:bg-secondary/50 transition-colors disabled:opacity-50"
              >
                <FileJson className="w-6 h-6 text-purple-500" />
                <span className="text-sm font-medium">数据备份</span>
                <span className="text-xs text-muted-foreground">JSON格式</span>
              </button>
            </div>

            {/* 导出说明 */}
            <div className="bg-amber-50 dark:bg-amber-900/20 rounded-lg p-3">
              <p className="text-xs text-amber-700 dark:text-amber-300">
                <strong>导出说明：</strong>PNG图片适合分享和打印；JSON数据可用于备份和恢复整个会议内容。
              </p>
            </div>
          </div>
        </div>
      </div>
    </div>
  )
}

  Share2,
  Hand,
  MessageSquare,
  Download,
  Upload,
  Sparkles,
  ExternalLink,
  ImageIcon,
  Loader2,
  Mail,
  Webhook,
  Camera,
  Shield,
} from "lucide-react"
import { useWhiteboardStore } from "@/lib/whiteboard-store"
import { exportCanvasToImage, downloadFile, exportAllPagesToImages } from "@/lib/export-utils"
import { cn } from "@/lib/utils"

export function SettingsModal() {
  const {
    showSettingsModal,
    setShowSettingsModal,
    isDarkMode,
    toggleDarkMode,
    isAutoSaveEnabled,
    setShowAISettings,
    exportSession,
    importSession,
    session,
    aiConfig,
    autoShareConfig,
    setAutoShareConfig,
    emailConfig,
    setEmailConfig,
  } = useWhiteboardStore()

  // 当前激活的标签页 - 重新排序：通用>AI>导出>分享推送>手势>语音
  const [activeTab, setActiveTab] = useState<"general" | "ai" | "export" | "share" | "gesture" | "voice">("general")
  const [exporting, setExporting] = useState(false)

  // 百度语音API配置
  const [baiduAppId, setBaiduAppId] = useState("")
  const [baiduApiKey, setBaiduApiKey] = useState("")
  const [baiduSecretKey, setBaiduSecretKey] = useState("")

  // 手势识别配置
  const [gestureEnabled, setGestureEnabled] = useState(false)
  const [gestureAutoStart, setGestureAutoStart] = useState(false)

  // 加载保存的配置
  useEffect(() => {
    if (typeof window !== "undefined") {
      // 加载百度语音配置
      const baiduConfig = localStorage.getItem("baidu_speech_config")
      if (baiduConfig) {
        try {
          const config = JSON.parse(baiduConfig)
          setBaiduAppId(config.appId || "")
          setBaiduApiKey(config.apiKey || "")
          setBaiduSecretKey(config.secretKey || "")
        } catch (e) {}
      }
      // 加载手势识别配置
      setGestureEnabled(localStorage.getItem("gesture_enabled") === "true")
      setGestureAutoStart(localStorage.getItem("gesture_auto_start") === "true")
    }
  }, [showSettingsModal])

  if (!showSettingsModal) return null

  // 导出JSON
  const handleExport = () => {
    const data = exportSession()
    const blob = new Blob([data], { type: "application/json" })
    const url = URL.createObjectURL(blob)
    const a = document.createElement("a")
    a.href = url
    a.download = `${session.name}-${new Date().toISOString().split("T")[0]}.json`
    a.click()
    URL.revokeObjectURL(url)
  }

  // 导入JSON
  const handleImport = () => {
    const input = document.createElement("input")
    input.type = "file"
    input.accept = ".json"
    input.onchange = async (e) => {
      const file = (e.target as HTMLInputElement).files?.[0]
      if (file) {
        const text = await file.text()
        const success = importSession(text)
        if (!success) {
          alert("导入失败：文件格式不正确")
        } else {
          setShowSettingsModal(false)
        }
      }
    }
    input.click()
  }

  // 导出当前页图片
  const handleExportCurrentPageImage = async () => {
    setExporting(true)
    try {
      const currentPage = session.pages[session.currentPageIndex]
      const blob = await exportCanvasToImage(currentPage, "png")
      if (blob) {
        downloadFile(blob, `${session.name}-${currentPage.name}.png`)
      }
    } catch (e) {
      console.error("[卡若] 导出失败", e)
    }
    setExporting(false)
  }

  // 导出全部页图片
  const handleExportAllPagesImages = async () => {
    setExporting(true)
    try {
      const blobs = await exportAllPagesToImages(session)
      blobs.forEach((blob, index) => {
        setTimeout(() => {
          downloadFile(blob, `${session.name}-第${index + 1}页.png`)
        }, index * 300)
      })
    } catch (e) {
      console.error("[卡若] 导出失败", e)
    }
    setExporting(false)
  }

  // 保存手势识别配置
  const handleSaveGestureConfig = () => {
    localStorage.setItem("gesture_enabled", gestureEnabled.toString())
    localStorage.setItem("gesture_auto_start", gestureAutoStart.toString())
    // 触发手势识别开关事件
    window.dispatchEvent(new Event("toggle-gesture"))
    alert("手势识别配置已保存！")
  }

  // 保存百度语音配置
  const handleSaveBaiduConfig = () => {
    localStorage.setItem(
      "baidu_speech_config",
      JSON.stringify({ appId: baiduAppId, apiKey: baiduApiKey, secretKey: baiduSecretKey }),
    )
    alert("百度语音配置已保存！")
  }

  // 标签页配置 - 重新排序
  const tabs = [
    { id: "general" as const, label: "通用", icon: Monitor },
    { id: "ai" as const, label: "AI功能", icon: Bot },
    { id: "export" as const, label: "导出", icon: Download },
    { id: "share" as const, label: "推送", icon: Share2 },
    { id: "gesture" as const, label: "手势", icon: Hand },
    { id: "voice" as const, label: "语音", icon: MessageSquare },
  ]

  return (
    <div className="fixed inset-0 z-[100] flex items-center justify-center bg-background/80 backdrop-blur-sm">
      <div className="glass-panel w-full max-w-2xl mx-4 rounded-2xl overflow-hidden max-h-[85vh] flex flex-col">
        {/* 头部 */}
        <div className="flex items-center justify-between p-4 border-b border-border/50 shrink-0">
          <h2 className="text-lg font-semibold">设置</h2>
          <button
            onClick={() => setShowSettingsModal(false)}
            className="p-2 rounded-lg hover:bg-secondary/50 transition-colors"
          >
            <X className="w-5 h-5" />
          </button>
        </div>

        {/* 标签页 */}
        <div className="flex border-b border-border/50 shrink-0 overflow-x-auto">
          {tabs.map((tab) => (
            <button
              key={tab.id}
              onClick={() => setActiveTab(tab.id)}
              className={cn(
                "flex-1 flex items-center justify-center gap-2 py-3 text-sm font-medium transition-colors whitespace-nowrap px-4",
                activeTab === tab.id
                  ? "text-primary border-b-2 border-primary"
                  : "text-muted-foreground hover:text-foreground",
              )}
            >
              <tab.icon className="w-4 h-4" />
              <span>{tab.label}</span>
            </button>
          ))}
        </div>

        {/* 设置内容 */}
        <div className="p-4 space-y-4 overflow-y-auto flex-1">
          {/* 通用设置 */}
          {activeTab === "general" && (
            <>
              <div className="space-y-3">
                <h3 className="text-sm font-medium text-muted-foreground uppercase tracking-wide">外观</h3>
                <div className="flex items-center justify-between p-3 bg-secondary/30 rounded-xl">
                  <span className="text-sm">深色主题</span>
                  <button
                    onClick={toggleDarkMode}
                    className={cn(
                      "w-12 h-6 rounded-full transition-colors relative",
                      isDarkMode ? "bg-primary" : "bg-muted",
                    )}
                  >
                    <div
                      className={cn(
                        "absolute top-0.5 w-5 h-5 rounded-full bg-white transition-transform",
                        isDarkMode ? "translate-x-6" : "translate-x-0.5",
                      )}
                    />
                  </button>
                </div>
              </div>

              <div className="space-y-3">
                <h3 className="text-sm font-medium text-muted-foreground uppercase tracking-wide flex items-center gap-1.5">
                  <Shield className="w-4 h-4" />
                  数据管理
                </h3>
                <div className="flex items-center justify-between p-3 bg-secondary/30 rounded-xl">
                  <div>
                    <span className="text-sm">自动保存</span>
                    <p className="text-xs text-muted-foreground">每分钟自动保存</p>
                  </div>
                  <div className={cn("w-12 h-6 rounded-full relative", isAutoSaveEnabled ? "bg-primary" : "bg-muted")}>
                    <div
                      className={cn(
                        "absolute top-0.5 w-5 h-5 rounded-full bg-white transition-transform",
                        isAutoSaveEnabled ? "translate-x-6" : "translate-x-0.5",
                      )}
                    />
                  </div>
                </div>
                <div className="grid grid-cols-2 gap-3">
                  <button
                    onClick={handleExport}
                    className="flex items-center justify-center gap-2 p-3 bg-secondary/30 rounded-xl hover:bg-secondary/50 transition-colors"
                  >
                    <Download className="w-4 h-4" />
                    <span className="text-sm">导出JSON</span>
                  </button>
                  <button
                    onClick={handleImport}
                    className="flex items-center justify-center gap-2 p-3 bg-secondary/30 rounded-xl hover:bg-secondary/50 transition-colors"
                  >
                    <Upload className="w-4 h-4" />
                    <span className="text-sm">导入数据</span>
                  </button>
                </div>
              </div>
            </>
          )}

          {/* AI功能设置 */}
          {activeTab === "ai" && (
            <>
              <div className="space-y-3">
                <h3 className="text-sm font-medium text-muted-foreground uppercase tracking-wide flex items-center gap-1.5">
                  <Bot className="w-4 h-4" />
                  AI服务配置
                </h3>
                <button
                  onClick={() => {
                    setShowSettingsModal(false)
                    setShowAISettings(true)
                  }}
                  className="w-full flex items-center justify-between p-4 bg-secondary/30 rounded-xl hover:bg-secondary/50 transition-colors"
                >
                  <div className="flex items-center gap-3">
                    <div className="w-10 h-10 rounded-xl bg-gradient-to-br from-purple-500 to-pink-500 flex items-center justify-center">
                      <Sparkles className="w-5 h-5 text-white" />
                    </div>
                    <div className="text-left">
                      <span className="text-sm font-medium">AI服务配置</span>
                      <p className="text-xs text-muted-foreground">
                        {aiConfig?.apiKey ? `已配置：${aiConfig.provider}` : "点击配置免费AI服务"}
                      </p>
                    </div>
                  </div>
                  <ExternalLink className="w-4 h-4 text-muted-foreground" />
                </button>
              </div>

              <div className="bg-blue-50 dark:bg-blue-900/20 rounded-xl p-4 border border-blue-200 dark:border-blue-800">
                <h4 className="text-sm font-medium text-blue-700 dark:text-blue-400 mb-2">AI功能说明</h4>
                <ul className="text-xs text-blue-600 dark:text-blue-500 space-y-1 list-disc list-inside">
                  <li>智能会议纪要生成</li>
                  <li>自动提取行动项</li>
                  <li>思维导图一键生成</li>
                  <li>语音识别转文字</li>
                  <li>自动总结会议名称</li>
                </ul>
              </div>
            </>
          )}

          {/* 导出设置 */}
          {activeTab === "export" && (
            <>
              <div className="space-y-3">
                <h3 className="text-sm font-medium text-muted-foreground uppercase tracking-wide flex items-center gap-1.5">
                  <ImageIcon className="w-4 h-4" />
                  导出截图
                </h3>
                <div className="grid grid-cols-2 gap-3">
                  <button
                    onClick={handleExportCurrentPageImage}
                    disabled={exporting}
                    className="flex flex-col items-center gap-2 p-4 bg-secondary/30 rounded-xl hover:bg-secondary/50 transition-colors disabled:opacity-50"
                  >
                    {exporting ? (
                      <Loader2 className="w-6 h-6 animate-spin" />
                    ) : (
                      <ImageIcon className="w-6 h-6 text-blue-500" />
                    )}
                    <span className="text-sm font-medium">当前页截图</span>
                  </button>
                  <button
                    onClick={handleExportAllPagesImages}
                    disabled={exporting}
                    className="flex flex-col items-center gap-2 p-4 bg-secondary/30 rounded-xl hover:bg-secondary/50 transition-colors disabled:opacity-50"
                  >
                    {exporting ? (
                      <Loader2 className="w-6 h-6 animate-spin" />
                    ) : (
                      <ImageIcon className="w-6 h-6 text-green-500" />
                    )}
                    <span className="text-sm font-medium">全部页截图</span>
                    <span className="text-xs text-muted-foreground">{session.pages.length} 页</span>
                  </button>
                </div>
              </div>

              <div className="bg-amber-50 dark:bg-amber-900/20 rounded-xl p-4 border border-amber-200 dark:border-amber-800">
                <p className="text-xs text-amber-700 dark:text-amber-300">
                  <strong>导出说明：</strong>PNG图片适合直接分享和打印；JSON数据可用于备份和恢复完整的会议内容。
                </p>
              </div>
            </>
          )}

          {/* 分享推送设置 */}
          {activeTab === "share" && (
            <>
              <div className="p-4 bg-blue-50 dark:bg-blue-900/20 rounded-xl border border-blue-200 dark:border-blue-800 mb-4">
                <p className="text-xs text-blue-600 dark:text-blue-400">
                  配置后，点击右上角"退出"按钮时将自动推送会议内容到配置的群和邮箱
                </p>
              </div>

              <div className="space-y-3">
                <h3 className="text-sm font-medium text-muted-foreground uppercase tracking-wide flex items-center gap-1.5">
                  <Webhook className="w-4 h-4" />
                  即时通讯Webhook
                </h3>

                <div className="space-y-3">
                  <div className="p-3 bg-secondary/30 rounded-xl space-y-2">
                    <label className="text-xs text-muted-foreground">飞书群Webhook</label>
                    <input
                      type="text"
                      value={autoShareConfig?.feishuWebhook || ""}
                      onChange={(e) => setAutoShareConfig?.({ ...autoShareConfig, feishuWebhook: e.target.value })}
                      placeholder="https://open.feishu.cn/open-apis/bot/v2/hook/..."
                      className="w-full px-3 py-2 text-sm bg-background rounded-lg border outline-none focus:ring-2 ring-primary"
                    />
                  </div>
                  <div className="p-3 bg-secondary/30 rounded-xl space-y-2">
                    <label className="text-xs text-muted-foreground">钉钉群Webhook</label>
                    <input
                      type="text"
                      value={autoShareConfig?.dingtalkWebhook || ""}
                      onChange={(e) => setAutoShareConfig?.({ ...autoShareConfig, dingtalkWebhook: e.target.value })}
                      placeholder="https://oapi.dingtalk.com/robot/send?access_token=..."
                      className="w-full px-3 py-2 text-sm bg-background rounded-lg border outline-none focus:ring-2 ring-primary"
                    />
                  </div>
                  <div className="p-3 bg-secondary/30 rounded-xl space-y-2">
                    <label className="text-xs text-muted-foreground">企业微信Webhook</label>
                    <input
                      type="text"
                      value={autoShareConfig?.wechatWebhook || ""}
                      onChange={(e) => setAutoShareConfig?.({ ...autoShareConfig, wechatWebhook: e.target.value })}
                      placeholder="https://qyapi.weixin.qq.com/cgi-bin/webhook/send?key=..."
                      className="w-full px-3 py-2 text-sm bg-background rounded-lg border outline-none focus:ring-2 ring-primary"
                    />
                  </div>
                </div>
              </div>

              <div className="space-y-3">
                <h3 className="text-sm font-medium text-muted-foreground uppercase tracking-wide flex items-center gap-1.5">
                  <Mail className="w-4 h-4" />
                  邮件配置
                </h3>
                <div className="p-3 bg-secondary/30 rounded-xl space-y-3">
                  <div className="grid grid-cols-2 gap-3">
                    <div className="space-y-1.5">
                      <label className="text-xs text-muted-foreground">SMTP服务器</label>
                      <input
                        type="text"
                        value={emailConfig?.smtpHost || ""}
                        onChange={(e) => setEmailConfig?.({ ...emailConfig, smtpHost: e.target.value })}
                        placeholder="smtp.qq.com"
                        className="w-full px-2 py-1.5 text-xs bg-background rounded border outline-none"
                      />
                    </div>
                    <div className="space-y-1.5">
                      <label className="text-xs text-muted-foreground">端口</label>
                      <input
                        type="number"
                        value={emailConfig?.smtpPort || 465}
                        onChange={(e) => setEmailConfig?.({ ...emailConfig, smtpPort: Number(e.target.value) })}
                        className="w-full px-2 py-1.5 text-xs bg-background rounded border outline-none"
                      />
                    </div>
                  </div>
                  <div className="space-y-1.5">
                    <label className="text-xs text-muted-foreground">发件邮箱</label>
                    <input
                      type="email"
                      value={emailConfig?.smtpUser || ""}
                      onChange={(e) => setEmailConfig?.({ ...emailConfig, smtpUser: e.target.value })}
                      placeholder="your@email.com"
                      className="w-full px-2 py-1.5 text-xs bg-background rounded border outline-none"
                    />
                  </div>
                  <div className="space-y-1.5">
                    <label className="text-xs text-muted-foreground">授权码/密码</label>
                    <input
                      type="password"
                      value={emailConfig?.smtpPassword || ""}
                      onChange={(e) => setEmailConfig?.({ ...emailConfig, smtpPassword: e.target.value })}
                      placeholder="SMTP授权码"
                      className="w-full px-2 py-1.5 text-xs bg-background rounded border outline-none"
                    />
                  </div>
                  <div className="space-y-1.5">
                    <label className="text-xs text-muted-foreground">收件人（多个用逗号分隔）</label>
                    <input
                      type="text"
                      value={autoShareConfig?.emailRecipients?.join(",") || ""}
                      onChange={(e) =>
                        setAutoShareConfig?.({
                          ...autoShareConfig,
                          emailRecipients: e.target.value
                            .split(",")
                            .map((s) => s.trim())
                            .filter(Boolean),
                          enabled: true,
                        })
                      }
                      placeholder="user1@email.com, user2@email.com"
                      className="w-full px-2 py-1.5 text-xs bg-background rounded border outline-none"
                    />
                  </div>
                </div>
              </div>
            </>
          )}

          {/* 手势识别设置 */}
          {activeTab === "gesture" && (
            <>
              <div className="p-4 bg-blue-50 dark:bg-blue-900/20 rounded-xl border border-blue-200 dark:border-blue-800">
                <h4 className="text-sm font-medium text-blue-700 dark:text-blue-400 mb-2 flex items-center gap-2">
                  <Camera className="w-4 h-4" />
                  手势识别（实验性功能）
                </h4>
                <p className="text-xs text-blue-600 dark:text-blue-500 mb-3">
                  使用摄像头追踪彩色物体（如荧光笔、彩色指套）在空中的移动，实现隔空书写和拍照。
                </p>

                <div className="space-y-3">
                  <div className="flex items-center justify-between p-2.5 bg-background/50 rounded-lg">
                    <span className="text-sm">启用手势识别</span>
                    <button
                      onClick={() => setGestureEnabled(!gestureEnabled)}
                      className={cn(
                        "w-12 h-6 rounded-full transition-colors relative",
                        gestureEnabled ? "bg-primary" : "bg-muted",
                      )}
                    >
                      <div
                        className={cn(
                          "absolute top-0.5 w-5 h-5 rounded-full bg-white transition-transform",
                          gestureEnabled ? "translate-x-6" : "translate-x-0.5",
                        )}
                      />
                    </button>
                  </div>

                  <div className="flex items-center justify-between p-2.5 bg-background/50 rounded-lg">
                    <div>
                      <span className="text-sm">开机自动启动</span>
                      <p className="text-xs text-muted-foreground">打开白板时自动启用手势识别</p>
                    </div>
                    <button
                      onClick={() => setGestureAutoStart(!gestureAutoStart)}
                      className={cn(
                        "w-12 h-6 rounded-full transition-colors relative",
                        gestureAutoStart ? "bg-primary" : "bg-muted",
                      )}
                    >
                      <div
                        className={cn(
                          "absolute top-0.5 w-5 h-5 rounded-full bg-white transition-transform",
                          gestureAutoStart ? "translate-x-6" : "translate-x-0.5",
                        )}
                      />
                    </button>
                  </div>

                  <button
                    onClick={handleSaveGestureConfig}
                    className="w-full py-2.5 text-sm bg-gradient-to-r from-blue-500 to-cyan-500 text-white rounded-lg font-medium"
                  >
                    保存设置
                  </button>
                </div>
              </div>

              <div className="p-4 bg-secondary/30 rounded-xl">
                <h4 className="text-sm font-medium mb-2">使用说明</h4>
                <ol className="text-xs text-muted-foreground space-y-1.5 list-decimal list-inside">
                  <li>准备一个彩色物体（红色、绿色或蓝色的荧光笔、指套等）</li>
                  <li>启用手势识别后，摄像头会追踪彩色物体</li>
                  <li>在空中移动物体即可在白板上书写</li>
                  <li>双手比心手势保持1秒可触发拍照</li>
                  <li>摄像头预览窗口会显示在右下角</li>
                </ol>
              </div>
            </>
          )}

          {/* 语音识别设置 */}
          {activeTab === "voice" && (
            <>
              <div className="p-4 bg-amber-50 dark:bg-amber-900/20 rounded-xl border border-amber-200 dark:border-amber-800">
                <h4 className="text-sm font-medium text-amber-700 dark:text-amber-400 mb-2">百度语音识别API配置</h4>
                <p className="text-xs text-amber-600 dark:text-amber-500 mb-3">
                  配置百度语音识别API后，可以实现更稳定的实时语音转文字功能。
                  <a
                    href="https://ai.baidu.com/tech/speech"
                    target="_blank"
                    rel="noopener noreferrer"
                    className="underline ml-1"
                  >
                    前往百度AI开放平台申请
                  </a>
                </p>

                <div className="space-y-3">
                  <div className="space-y-1.5">
                    <label className="text-xs text-muted-foreground">App ID</label>
                    <input
                      type="text"
                      value={baiduAppId}
                      onChange={(e) => setBaiduAppId(e.target.value)}
                      placeholder="输入百度App ID"
                      className="w-full px-3 py-2 text-sm bg-background rounded-lg border outline-none focus:ring-2 ring-amber-500"
                    />
                  </div>
                  <div className="space-y-1.5">
                    <label className="text-xs text-muted-foreground">API Key</label>
                    <input
                      type="text"
                      value={baiduApiKey}
                      onChange={(e) => setBaiduApiKey(e.target.value)}
                      placeholder="输入百度API Key"
                      className="w-full px-3 py-2 text-sm bg-background rounded-lg border outline-none focus:ring-2 ring-amber-500"
                    />
                  </div>
                  <div className="space-y-1.5">
                    <label className="text-xs text-muted-foreground">Secret Key</label>
                    <input
                      type="password"
                      value={baiduSecretKey}
                      onChange={(e) => setBaiduSecretKey(e.target.value)}
                      placeholder="输入百度Secret Key"
                      className="w-full px-3 py-2 text-sm bg-background rounded-lg border outline-none focus:ring-2 ring-amber-500"
                    />
                  </div>
                  <button
                    onClick={handleSaveBaiduConfig}
                    className="w-full py-2.5 text-sm bg-gradient-to-r from-amber-400 to-orange-500 text-white rounded-lg font-medium"
                  >
                    保存配置
                  </button>
                </div>
              </div>

              <div className="p-4 bg-secondary/30 rounded-xl">
                <h4 className="text-sm font-medium mb-2">使用说明</h4>
                <ol className="text-xs text-muted-foreground space-y-1.5 list-decimal list-inside">

## ✨ 新功能详解

### 1. 双手拍照手势 📸

**功能说明**  
通过双手比心手势自动触发拍照，解放双手，让会议记录更自然。

**使用步骤**
1. 进入设置 > 手势识别
2. 启用"手势识别"开关
3. 准备两个彩色物体（如红色指套、荧光笔）
4. 双手在摄像头前展示颜色物体保持1秒
5. 系统自动识别并倒计时3秒
6. 拍照完成，照片自动添加到白板

**使用技巧**
- 建议使用鲜艳的纯色物体（红/绿/蓝）
- 确保光线充足，背景干净
- 双手距离摄像头30-50cm效果最佳
- 也可以直接点击"拍照"按钮手动拍照

**常见问题**
- Q: 检测不到手势？
- A: 检查颜色灵敏度设置，调整为30-40

- Q: 拍照时机不准确？
- A: 确保双手同时出现在画面中，保持静止1秒
