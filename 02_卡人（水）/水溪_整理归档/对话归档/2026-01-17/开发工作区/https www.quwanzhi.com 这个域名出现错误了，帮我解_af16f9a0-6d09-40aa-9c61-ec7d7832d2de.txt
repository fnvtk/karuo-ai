    def 从其他服务器同步证书(self, 源服务器: str, 域名列表: List[str]) -> Dict:
        """
        从其他服务器同步证书（用于通配符证书共享）
        
        :param 源服务器: 源服务器名称
        :param 域名列表: 需要同步的域名列表
        :return: 同步结果
        """

        try:
            import paramiko
        except ImportError:
            return {"status": False, "msg": "需要安装paramiko: pip install paramiko"}
        
        源配置 = self.服务器配置[源服务器]
        
        日志.info(f"正在从 {源服务器} 同步证书...")
        
        # 连接源服务器

        ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
        
        try:
            ssh.connect(
                源配置["ip"], 
                port=22, 
                username=源配置["ssh_user"], 
                password=源配置["ssh_pass"],
                timeout=15
            )
        except Exception as e:
            return {"status": False, "msg": f"连接源服务器失败: {e}"}
        
        # 读取证书（优先使用通配符证书）
        通配符证书路径 = "/www/server/panel/vhost/cert/www.quwanzhi.com"

        fullchain = stdout.read().decode()
        
        stdin, stdout, stderr = ssh.exec_command(f"cat {通配符证书路径}/privkey.pem")
        privkey = stdout.read().decode()
        
        ssh.close()
        
        if not fullchain or not privkey:
            return {"status": False, "msg": "读取源证书失败"}
        
        日志.info(f"✓ 已读取源证书 (fullchain: {len(fullchain)}字节)")
        
        # 同步到目标域名
        成功数 = 0
        失败数 = 0
        
        for 域名 in 域名列表:
            cert_path = f"/www/server/panel/vhost/cert/{域名}"
            
            # 确保目录存在
            self.api.创建目录(cert_path)
            
            # 保存证书
            r1 = self.api.保存文件内容(f"{cert_path}/fullchain.pem", fullchain)
            r2 = self.api.保存文件内容(f"{cert_path}/privkey.pem", privkey)
            
            if r1.get('status') and r2.get('status'):
                日志.info(f"  ✓ {域名} 证书已同步")
                成功数 += 1
            else:
                日志.error(f"  ✗ {域名} 同步失败")
                失败数 += 1
        
        # 重载nginx
        self.api.发送请求("system?action=ServiceAdmin", {"name": "nginx", "type": "reload"})
        
        return {
            "status": True,
            "msg": f"同步完成: 成功 {成功数} 个, 失败 {失败数} 个",
            "成功": 成功数,
            "失败": 失败数
        }
    
    def 批量修复过期证书(self, 源服务器: str = "小型宝塔") -> Dict:
        """
        批量修复所有过期证书
        
        :param 源服务器: 通配符证书来源服务器
        :return: 修复结果
        """
        # 先检查状态
        状态 = self.检查证书状态(显示详情=False)
        
        if not 状态.get('status'):
            return 状态
        
        过期列表 = 状态['data']['已过期']
        即将过期 = 状态['data']['即将过期']
        
        需要修复 = [cert['域名'] for cert in 过期列表 + 即将过期]
        
        if not 需要修复:
            日志.info("没有需要修复的证书")
            return {"status": True, "msg": "所有证书状态正常"}
        
        日志.info(f"需要修复 {len(需要修复)} 个证书")
        
        # 先尝试执行自动续期
        续期结果 = self.执行证书续期()
        
        if 续期结果.get('status'):
            日志.info("已触发自动续期任务，等待执行...")
            import time
            time.sleep(10)
            
            # 重新检查
            新状态 = self.检查证书状态(显示详情=False)
            仍过期 = [cert['域名'] for cert in 新状态['data']['已过期']]
            
            if not 仍过期:
                return {"status": True, "msg": "自动续期成功，所有证书已更新"}
            
            需要修复 = 仍过期
        
        # 如果自动续期失败或仍有过期，从其他服务器同步
        日志.info(f"从 {源服务器} 同步证书...")
        return self.从其他服务器同步证书(源服务器, 需要修复)
