import fs from 'fs';

// 脑图根目录 - 我的脑图
const MINDMAP_ROOT = '/Users/karuo/Documents/我的脑图';

export interface MindMapNode {
  name: string;
  path: string;
  type: 'file' | 'directory';
  ext?: string;
  children?: MindMapNode[];
}

/**
 * 递归读取目录结构（限制深度，避免过大）
 */
function readDirRecursive(dirPath: string, maxDepth: number = 3, currentDepth: number = 0): MindMapNode[] {
  if (currentDepth >= maxDepth) return [];

  try {
    const entries = fs.readdirSync(dirPath, { withFileTypes: true });
    const nodes: MindMapNode[] = [];

    for (const entry of entries) {
      const fullPath = path.join(dirPath, entry.name);
      const relativePath = path.relative(MINDMAP_ROOT, fullPath);

      if (entry.isDirectory()) {
        const children = readDirRecursive(fullPath, maxDepth, currentDepth + 1);
        nodes.push({
          name: entry.name,
          path: relativePath,
          type: 'directory',
          children: children.length > 0 ? children : undefined,
        });
      } else {
        const ext = path.extname(entry.name).toLowerCase();
        nodes.push({
          name: entry.name,
          path: relativePath,
          type: 'file',
          ext: ext || undefined,
        });
      }
    }

    return nodes.sort((a, b) => {
      // 目录优先
      if (a.type !== b.type) return a.type === 'directory' ? -1 : 1;
      return a.name.localeCompare(b.name);
    });
  } catch (err) {
    console.error('读取目录失败:', dirPath, err);
    return [];
  }
}

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const subPath = searchParams.get('path') || '';

  const fullPath = path.join(MINDMAP_ROOT, subPath);

  // 安全校验：确保路径在根目录内
  const resolvedPath = path.resolve(fullPath);
  const resolvedRoot = path.resolve(MINDMAP_ROOT);
  if (!resolvedPath.startsWith(resolvedRoot)) {
    return NextResponse.json({ error: '无效路径' }, { status: 400 });
  }

  try {
    const stat = fs.statSync(fullPath);
    if (!stat.isDirectory()) {
      return NextResponse.json({ error: '不是目录' }, { status: 400 });
    }

    const nodes = readDirRecursive(fullPath);
    return NextResponse.json({
      path: subPath || '/',
      nodes,
      root: MINDMAP_ROOT,
    });
  } catch (err) {
    console.error('API 错误:', err);
    return NextResponse.json(
      { error: '读取失败', detail: String(err) },
      { status: 500 }
    );
  }
}

/** 从 xmind content.xml 提取主题文本 */
function parseXmindContent(xml: string): string[] {
  const texts: string[] = [];
  const titleMatches = xml.matchAll(/<title>([^<]*)<\/title>/g);
  for (const m of titleMatches) {
    const t = m[1].trim();
    if (t && !['画布 1', 'Central Topic', 'Main Topic', 'Subtopic', 'Callout', 'Floating Topic', 'Org-Chart Topic', 'Topic'].includes(t)) {
      texts.push(t.replace(/&#x([0-9A-Fa-f]+);/g, (_, hex) => String.fromCharCode(parseInt(hex, 16))).replace(/&#(\d+);/g, (_, n) => String.fromCharCode(parseInt(n, 10))));
    }
  }
  const noteMatches = xml.matchAll(/<plain>([^<]*)<\/plain>/g);
  for (const m of noteMatches) {
    const t = m[1].trim().replace(/&#13;\s*/g, '\n');
    if (t && t.length > 2) texts.push(t);
  }
  return [...new Set(texts)];
}

/** 从 mmap Document.xml 提取 PlainText */
function parseMmapContent(xml: string): string[] {
  const texts: string[] = [];
  const matches = xml.matchAll(/PlainText="([^"]*)"/g);
  const skip = new Set(['Central Topic', 'Main Topic', 'Subtopic', 'Callout', 'Floating Topic', 'Org-Chart Topic', 'Topic']);
  for (const m of matches) {
    let t = m[1]
      .replace(/&lt;/g, '<')
      .replace(/&gt;/g, '>')
      .replace(/&amp;/g, '&')
      .replace(/&#xA;/g, '\n')
      .replace(/<br\s*\/?>/gi, '\n')
      .trim();
    if (t && !skip.has(t) && t.length > 1) texts.push(t);
  }
  return [...new Set(texts)];
}

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const filePath = searchParams.get('path') || '';

  const fullPath = path.join(MINDMAP_ROOT, filePath);
  const resolved = path.resolve(fullPath);
  const root = path.resolve(MINDMAP_ROOT);
  if (!resolved.startsWith(root)) {
    return NextResponse.json({ error: '无效路径' }, { status: 400 });
  }

  try {
    const stat = fs.statSync(fullPath);
    if (!stat.isFile()) {
      return NextResponse.json({ error: '不是文件' }, { status: 400 });
    }
    const ext = path.extname(fullPath).toLowerCase();
    if (!['.xmind', '.mmap'].includes(ext)) {
      return NextResponse.json({ error: '不支持的文件格式' }, { status: 400 });
    }

    const buf = fs.readFileSync(fullPath);
    const zip = await JSZip.loadAsync(buf);
    let texts: string[] = [];

    if (ext === '.xmind') {
      const content = zip.file('content.xml');
      if (content) {
        const xml = await content.async('string');
        texts = parseXmindContent(xml);
      }
    } else if (ext === '.mmap') {
      const doc = zip.file('Document.xml');
      if (doc) {
        const xml = await doc.async('string');
        texts = parseMmapContent(xml);
      }
    }

    const { summary, category, keywords } = generateSummary(path.basename(fullPath), texts);

    return NextResponse.json({
      name: path.basename(fullPath),
      path: filePath,
      summary,
      category,
      keywords,
      topics: texts.slice(0, 50),
      topicCount: texts.length,
    });
  } catch (err) {
    console.error('解析失败:', err);
    return NextResponse.json({ error: '解析失败', detail: String(err) }, { status: 500 });
  }
}

/** 可展开子类目卡片 - 简洁分析 + 深度分析 + 下一步落地 */
function ExpandableCard({
  title,
  subtitle,
  brief,
  deepAnalysis,
  nextSteps,
  children,
}: {
  title: string;
  subtitle?: string;
  brief?: string;
  deepAnalysis?: string;
  nextSteps?: string[];
  children?: React.ReactNode;
}) {
  const [open, setOpen] = useState(false);
  const hasExpand = !!(deepAnalysis || (nextSteps && nextSteps.length > 0));

  return (
    <Collapsible.Root open={open} onOpenChange={setOpen}>
      <div className="glass-card overflow-hidden">
        <Collapsible.Trigger asChild>
          <button className="w-full p-4 flex items-start gap-3 text-left">
            <div className="flex-1 min-w-0">
              <h3 className="font-medium text-slate-800">{title}</h3>
              {subtitle && <p className="text-xs text-slate-500 mt-0.5">{subtitle}</p>}
              {brief && <p className="text-sm text-slate-600 mt-2">{brief}</p>}
            </div>
            {hasExpand && (
              <span className="flex-shrink-0 text-slate-400 transition-transform duration-200" style={{ transform: open ? 'rotate(90deg)' : 'none' }}>
                <ChevronRight className="w-5 h-5" />
              </span>
            )}
          </button>
        </Collapsible.Trigger>
        <Collapsible.Content>
          <div className="px-4 pb-4 pt-0 border-t border-white/20 mt-0">
            {deepAnalysis && (
              <div className="mt-3">
                <p className="text-xs text-slate-500 mb-1">深度分析</p>
                <p className="text-sm text-slate-600 leading-relaxed">{deepAnalysis}</p>
              </div>
            )}
            {nextSteps && nextSteps.length > 0 && (
              <div className="mt-3">
                <p className="text-xs text-slate-500 mb-2 flex items-center gap-1">
                  <ListTodo className="w-3.5 h-3.5" /> 下一步落地
                </p>
                <ul className="space-y-1">
                  {nextSteps.map((s, i) => (
                    <li key={i} className="text-sm text-slate-600 flex items-start gap-2">
                      <span className="text-slate-400">{i + 1}.</span>
                      <span>{s}</span>
                    </li>
                  ))}
                </ul>
              </div>
            )}
            {children}
          </div>
        </Collapsible.Content>
      </div>
    </Collapsible.Root>
  );
}

/** 递归扫描目录下的 .md 文件 */
function scanMdFiles(dir: string, base = ''): { name: string; path: string }[] {
  const result: { name: string; path: string }[] = [];
  const fullPath = path.join(PROJECT_ROOT, dir);
  if (!fs.existsSync(fullPath)) return result;
  const entries = fs.readdirSync(fullPath, { withFileTypes: true });
  for (const e of entries) {
    const rel = base ? `${base}/${e.name}` : e.name;
    if (e.isDirectory()) {
      result.push(...scanMdFiles(`${dir}/${e.name}`, rel));
    } else if (e.name.endsWith('.md') && e.name.includes('经验融合')) {
      result.push({ name: e.name.replace('经验融合_', '').replace('.md', ''), path: `${dir}/${e.name}` });
    }
  }
  return result;
}

/** GET /api/project - 获取项目结构 */
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const action = searchParams.get('action');
  const filePath = searchParams.get('path');

  if (action === 'content' && filePath) {
    const fullPath = path.join(PROJECT_ROOT, filePath);
    const resolved = path.resolve(fullPath);
    const root = path.resolve(PROJECT_ROOT);
    if (!resolved.startsWith(root)) {
      return NextResponse.json({ error: '无效路径' }, { status: 400 });
    }
    try {
      const content = fs.readFileSync(fullPath, 'utf-8');
      return NextResponse.json({ content, path: filePath });
    } catch (err) {
      return NextResponse.json({ error: '读取失败', detail: String(err) }, { status: 500 });
    }
  }

  const structure: Record<string, { dir: string; files: { name: string; path: string }[] }> = {};
  for (const [w, cfg] of Object.entries(WUXING_MAP)) {
    const scanned = scanMdFiles(cfg.dir);
    structure[w] = { dir: cfg.dir, files: scanned.length > 0 ? scanned : cfg.files };
  }

  let readme = '';
  try {
    readme = fs.readFileSync(path.join(PROJECT_ROOT, 'README_项目总览.md'), 'utf-8');
  } catch {
    readme = '';
  }

  return NextResponse.json({
    name: '经验融合',
    tagline: '私域流量获客 + 小团队标准化运营 + 数据驱动分润 + 技术自动化实现',
    projectRoot: PROJECT_ROOT,
    structure,
    readme,
  });
}
