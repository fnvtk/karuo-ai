+def should_skip(path: Path) -> bool:

+def split_triggers(raw: str) -> List[str]:
+    if not raw:
+        return []
+    normalized = re.sub(r"[，、|/]+", ",", raw)

+    parts = [p.strip() for p in normalized.split(",")]
+    # 去重但保序
+    seen = set()
+    out: List[str] = []
+    for p in parts:

+            out.append(p)
+    return out

+def parse_frontmatter(content: str) -> Dict[str, str]:
+    match = re.match(r"^---\s*\n(.*?)\n---\s*\n?", content, re.DOTALL)

+                "description": description,
+                "triggers": triggers,

+def validate_index_paths() -> Tuple[int, int, List[str]]:

+        "skills": rows,
+    }

+        json.dumps(out, ensure_ascii=False, indent=2),
+        encoding="utf-8",
+    )

# -*- coding: utf-8 -*-
"""
自动拆分超大 SKILL.md 到 references/

规则：
1. 扫描全部 SKILL.md，找出 >300 行的文件
2. 将正文中的"大段代码块""大段表格""大段API文档"等自动识别为可拆段
3. 拆出到同级 references/ 目录下
4. 在 SKILL.md 原位保留简短引用行

用法：
  python3 split_large_skills.py          # 预览
  python3 split_large_skills.py --apply  # 执行
"""

import re
import sys
from pathlib import Path
from datetime import datetime

ROOT = Path(__file__).resolve().parent.parent.parent
THRESHOLD = 300  # 超过此行数才拆
TARGET_MAX = 150  # 拆后目标行数

SKIP = ['.git', 'node_modules', '__pycache__', '.venv', '_归档', '_合并前备份', '存客宝备份']


def should_skip(p: Path) -> bool:
    return any(k in str(p) for k in SKIP)


def find_splittable_sections(lines: list[str]) -> list[dict]:
    """
    识别可拆出的段落：
    - 连续代码块（```...```）>20行
    - 连续表格（|...|）>15行
    - 以 ## 开头的大段（>40行的二级标题段）
    """
    sections = []
    i = 0
    n = len(lines)

    while i < n:
        line = lines[i]

        # 1) 代码块
        if line.strip().startswith('```') and not line.strip().startswith('```\n'):
            start = i
            i += 1
            while i < n and not (lines[i].strip().startswith('```') and len(lines[i].strip()) <= 10):
                i += 1
            end = min(i + 1, n)
            block_len = end - start
            if block_len > 20:
                # 往上找最近的标题作为段名
                title = _find_nearest_heading(lines, start)
                sections.append({
                    'type': 'code_block',
                    'start': start,
                    'end': end,
                    'length': block_len,
                    'title': title,
                })
            i = end
            continue

        # 2) 表格段
        if '|' in line and line.strip().startswith('|'):
            start = i
            while i < n and '|' in lines[i] and lines[i].strip().startswith('|'):
                i += 1
            block_len = i - start
            if block_len > 15:
                title = _find_nearest_heading(lines, start)
                sections.append({
                    'type': 'table',
                    'start': start,
                    'end': i,
                    'length': block_len,
                    'title': title,
                })
            continue

        # 3) 大段二级标题
        if line.startswith('## ') and not line.startswith('## ⚠'):
            start = i
            heading = line.strip().lstrip('#').strip()
            i += 1
            while i < n and not (lines[i].startswith('## ') or lines[i].startswith('# ')):
                i += 1
            block_len = i - start
            if block_len > 40:
                sections.append({
                    'type': 'section',
                    'start': start,
                    'end': i,
                    'length': block_len,
                    'title': heading,
                })
            continue

        i += 1

    return sections


def _find_nearest_heading(lines, idx):
    for j in range(idx - 1, max(idx - 10, -1), -1):
        if j >= 0 and lines[j].strip().startswith('#'):
            return lines[j].strip().lstrip('#').strip()
    return f"段落_{idx}"


def sanitize_filename(name: str) -> str:
    name = re.sub(r'[/\\:*?"<>|]', '_', name)
    name = re.sub(r'\s+', '_', name)
    return name[:60]


def process_skill(skill_path: Path, apply: bool = False) -> dict:
    content = skill_path.read_text(encoding='utf-8', errors='ignore')
    lines = content.split('\n')
    total = len(lines)

    if total <= THRESHOLD:
        return {'path': str(skill_path.relative_to(ROOT)), 'total': total, 'action': 'skip', 'splits': []}

    sections = find_splittable_sections(lines)
    if not sections:
        return {'path': str(skill_path.relative_to(ROOT)), 'total': total, 'action': 'no_splittable', 'splits': []}

    # 按从大到小排序，优先拆大的
    sections.sort(key=lambda s: -s['length'])

    refs_dir = skill_path.parent / 'references'
    splits_done = []
    removed_lines = 0

    # 计算需要拆多少才能接近 TARGET_MAX
    lines_to_remove = total - TARGET_MAX

    # 从最大的段开始拆，直到总行数降到合理范围
    selected = []
    cumulative = 0
    for sec in sections:
        if cumulative >= lines_to_remove:
            break
        selected.append(sec)
        cumulative += sec['length']

    if not selected:
        return {'path': str(skill_path.relative_to(ROOT)), 'total': total, 'action': 'no_splittable', 'splits': []}

    # 按位置从后往前拆（避免偏移）
    selected.sort(key=lambda s: -s['start'])

    for sec in selected:
        safe_title = sanitize_filename(sec['title'])
        ref_filename = f"拆分_{safe_title}.md"
        ref_content = '\n'.join(lines[sec['start']:sec['end']])

        # 替换原文为引用行
        ref_line = f"> **详细内容**：见 `references/{ref_filename}`\n"
        lines[sec['start']:sec['end']] = [ref_line]
        removed_lines += sec['length'] - 1

        splits_done.append({
            'title': sec['title'],
            'type': sec['type'],
            'length': sec['length'],
            'filename': ref_filename,
        })

        if apply:
            refs_dir.mkdir(exist_ok=True)
            (refs_dir / ref_filename).write_text(
                f"# {sec['title']}\n\n"
                f"> 从 SKILL.md 自动拆分，{datetime.now().strftime('%Y-%m-%d %H:%M')}\n\n"
                f"{ref_content}\n",
                encoding='utf-8'
            )

    if apply:
        skill_path.write_text('\n'.join(lines), encoding='utf-8')

    return {
        'path': str(skill_path.relative_to(ROOT)),
        'total': total,
        'new_total': total - removed_lines,
        'action': 'split',
        'splits': splits_done,
    }


def main():
    apply = '--apply' in sys.argv
    mode = "执行" if apply else "预览"
    print(f"=== 超大 Skill 拆分（{mode}模式） ===\n")

    all_skills = sorted(ROOT.rglob("SKILL.md"))
    results = []

    for sp in all_skills:
        if should_skip(sp):
            continue
        result = process_skill(sp, apply=apply)
        if result['action'] != 'skip':
            results.append(result)

    if not results:
        print("所有 SKILL.md 均 ≤300 行，无需拆分。")
        return

    for r in results:
        if r['action'] == 'split':
            print(f"📦 {r['path']}：{r['total']}→{r['new_total']}行")
            for s in r['splits']:
                print(f"   ↳ [{s['type']}] {s['title']} ({s['length']}行) → references/{s['filename']}")
        elif r['action'] == 'no_splittable':
            print(f"⚠️  {r['path']}：{r['total']}行，未找到可拆段")

    split_count = sum(1 for r in results if r['action'] == 'split')
    total_splits = sum(len(r['splits']) for r in results if r['action'] == 'split')
    print(f"\n📊 统计：{split_count}个Skill被拆分，共拆出{total_splits}个段落到references")

    if not apply:
        print("\n加 --apply 执行实际拆分。")


if __name__ == "__main__":
    main()

"""

import re
import sys
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent.parent

# 从 SKILL_INDEX.md 建立 path → keywords 映射
def build_index_keywords() -> dict:
    """解析 SKILL_INDEX.md，提取每个 SKILL 的关键词"""
    if not INDEX_MD.exists():
        return {}
    content = INDEX_MD.read_text(encoding='utf-8', errors='ignore')
    mapping = {}
    # 匹配表格行：| ID | 关键词 | 执行人 | Skill | SKILL.md 路径 |
    for line in content.split('\n'):
        if '|' not in line or 'SKILL.md' not in line:
            continue
        cols = [c.strip() for c in line.split('|')]
        if len(cols) < 6:
            continue
        # cols: ['', ID, 关键词, 执行人, Skill, 路径, '']
        keywords_str = cols[2] if len(cols) > 2 else ''
        skill_name = cols[4] if len(cols) > 4 else ''
        path_str = cols[5] if len(cols) > 5 else ''
        # 提取 path
        path_match = re.search(r'`([^`]*SKILL\.md)`', path_str)
        if path_match:
            path = path_match.group(1)
            mapping[path] = {
                'keywords': keywords_str,
                'skill_name': skill_name,
            }
    return mapping


def extract_first_heading(content: str) -> str:
    """提取第一个 # 标题"""
    for line in content.split('\n'):
        if line.startswith('# ') and not line.startswith('## '):
            return line.lstrip('#').strip()
    return ''


def extract_ability_section(content: str) -> str:
    """提取 ## 能力说明 段落的第一段文字"""
    match = re.search(r'## 能力说明\s*\n(.*?)(?=\n## |\Z)', content, re.DOTALL)
    if match:
        text = match.group(1).strip()
        # 取第一段非空行
        for line in text.split('\n'):
            line = line.strip().lstrip('- ').strip()
            if line and not line.startswith('|') and not line.startswith('>'):
                return line[:100]
    return ''


def parse_frontmatter(content: str):
    """解析已有 frontmatter"""
    match = re.match(r'^---\s*\n(.*?)\n---\s*\n?', content, re.DOTALL)
    if not match:
        return {}, content
    fm_text = match.group(1)
    fm = {}
    for line in fm_text.split('\n'):
        if ':' in line:
            key, val = line.split(':', 1)
            fm[key.strip()] = val.strip()
    rest = content[match.end():]
    return fm, rest


def rebuild_frontmatter(fm: dict, body: str) -> str:
    """重建文件内容"""
    lines = ['---']
    order = ['name', 'description', 'triggers', 'owner', 'group']
    for k in order:
        if k in fm:
            lines.append(f'{k}: {fm[k]}')
    for k, v in fm.items():
        if k not in order:
            lines.append(f'{k}: {v}')
    lines.append('---')
    return '\n'.join(lines) + '\n\n' + body.lstrip('\n')

def infer_owner_group(skill_path: Path):
    rel = skill_path.relative_to(ROOT)
    parts = list(rel.parts)
    group = '未知'
    owner = '未知'
    for part in parts:
        for prefix, g in GROUP_MAP.items():
            if part == prefix or part.startswith(prefix):
                group = g
                break
    for i, part in enumerate(parts):
        if part == '_团队成员' and i + 1 < len(parts):
            owner = parts[i + 1]
            break
    if owner == '未知':
        owner = LEADER_BY_GROUP.get(group, '未知')
    return owner, group


def main():
    apply = '--apply' in sys.argv

        # 补 owner/group

        if not fm.get('owner'):
            fm['owner'] = owner
            changed = True
        if not fm.get('group'):
            fm['group'] = group
            changed = True

        # 补 description
        if not fm.get('description'):
            heading = extract_first_heading(body)
            ability = extract_ability_section(body)
            if ability:
                desc = ability
            elif heading:
                desc = heading
            else:
                desc = fm.get('name', skill_path.parent.name)
            # 查 INDEX 补充关键词
            idx_info = index_keywords.get(rel, {})
            keywords = idx_info.get('keywords', '')

            fm['description'] = desc
            changed = True

        # 补 triggers
        if not fm.get('triggers'):
            idx_info = index_keywords.get(rel, {})
            keywords = idx_info.get('keywords', '')
            if keywords:
                fm['triggers'] = keywords
                changed = True
            else:
                # 从 name 生成基础触发词
                name = fm.get('name', '')
                if name:
                    fm['triggers'] = name
                    changed = True

        if changed:
            patched += 1
            print(f"  ✏️  {rel}")
            print(f"      desc: {fm.get('description', '')[:60]}...")
            print(f"      triggers: {fm.get('triggers', '')[:60]}")
