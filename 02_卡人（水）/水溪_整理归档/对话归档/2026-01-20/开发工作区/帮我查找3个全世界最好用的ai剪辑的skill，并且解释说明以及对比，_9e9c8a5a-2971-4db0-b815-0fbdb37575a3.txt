## 🏆 全球最佳AI剪辑工具 TOP 3 对比

### 1. Descript —— 「文字即剪辑」的革命者

**核心理念：** 把视频当成Word文档编辑

| 项目 | 内容 |
|------|------|
| **官网** | descript.com |
| **定价** | 免费版 / $12~$24/月 |
| **最适合** | 播客、访谈、教程、Talking Head类内容 |

**核心AI功能：**
- **Underlord AI助手**：一键找出视频"高光片段"、自动删除"嗯啊"口水词
- **文字剪辑**：直接编辑文字稿，对应视频自动剪切（删除一句话=删除一段视频）
- **Overdub 语音克隆**：录入自己的声音后，输入文字就能生成"你的声音"纠正口误
- **眼神校正**：AI让你看起来像在直视镜头（即使你在看提词器）
- **Studio Sound**：一键去除背景噪音，让手机录音变成录音棚效果

**优势：** 彻底改变剪辑逻辑，特别适合内容创作者快速产出高质量长视频

**劣势：** 视觉特效能力弱，不适合短视频爆款套路

---

### 2. Runway ML —— AI视觉特效的「好莱坞级」先锋

**核心理念：** 用AI实现传统特效无法做到的事

| 项目 | 内容 |
|------|------|
| **官网** | runwayml.com |
| **定价** | $15~$95+/月 |
| **最适合** | 专业创作者、电影级特效、艺术创作、广告大片 |

**核心AI功能：**
- **Gen-3/Gen-4 文生视频**：输入文字描述，直接生成高质量视频片段
- **Motion Brush 运动画笔**：在静态图上「刷」出运动区域，让图片动起来
- **AI抠像**：无需绿幕，一键去除复杂背景
- **Inpainting 物体擦除**：自动移除视频中的路人、杂物、Logo
- **风格迁移**：把视频转换成动画、油画等艺术风格

**优势：** 生成式AI领域绝对领先，可以创造"不可能拍到"的画面

**劣势：** 学习曲线陡峭，价格较贵，不适合日常简单剪辑

### 3. CapCut/剪映 —— 短视频「效率之王」

**核心理念：** 最快速度产出能爆的内容

|------|------|

| **最适合** | 抖音/快手/TikTok/Reels短视频创作者、电商卖家 |

**核心AI功能：**
- **AI一键成片**：丢入素材→选择风格→自动出片（含卡点、转场、字幕）
- **智能字幕**：支持200+语言，准确率98%+，自动识别语气词
- **AI脚本生成**：输入产品/主题，自动生成带货文案
- **海量爆款模板**：热门视频同款，一键套用
- **跨平台同步**：手机→电脑无缝切换

**优势：** 上手零门槛，模板紧跟热点，适合批量产出短视频

**劣势：** 模板化痕迹重，难以做出差异化内容；字节系工具数据安全存疑

## 常见问题

### Q: 转录很慢怎么办？
A: 使用更小的模型：`--model small`（默认medium）

### Q: 切片没有声音？
A: 添加音频流参数：`ffmpeg -i input.mp4 -c:v copy -c:a aac output.mp4`

### Q: 如何只提取某个人说的内容？
A: 开启说话人分离：`--diarize true`（需要额外安装pyannote）

def transcribe_video(video_path: str, output_dir: Path, model: str = "medium", language: str = None) -> dict:
    """转录视频"""
    print(f"🎤 转录视频: {video_path}")
    print(f"   使用模型: {model}")
    
    import whisper
    
    # 加载模型
    print("   加载Whisper模型...")
    model_obj = whisper.load_model(model)
    
    # 转录

    print(f"     - JSON: {transcript_path}")
    print(f"     - SRT:  {srt_path}")
    print(f"     - TXT:  {txt_path}")
    
    return result


def generate_srt(segments: list, output_path: Path):
    """生成SRT字幕文件"""
    with open(output_path, "w", encoding="utf-8") as f:
        for i, seg in enumerate(segments, 1):
            start = format_timestamp_srt(seg["start"])
            end = format_timestamp_srt(seg["end"])
            text = seg["text"].strip()
            f.write(f"{i}\n{start} --> {end}\n{text}\n\n")


def generate_timestamped_text(segments: list, output_path: Path):
    """生成带时间戳的文本（用于AI分析）"""
    with open(output_path, "w", encoding="utf-8") as f:
        for seg in segments:
            timestamp = format_timestamp(seg["start"])
            text = seg["text"].strip()
            f.write(f"[{timestamp}] {text}\n")


def format_timestamp(seconds: float) -> str:
    """格式化时间戳 HH:MM:SS"""
    hours = int(seconds // 3600)
    minutes = int((seconds % 3600) // 60)

def transcribe(input_path: str, output_dir: str = None, model: str = "medium", language: str = None):

        input_path: 输入文件路径

    print(f"🎤 开始转录: {input_path}")

    }

        options["language"] = language

def sanitize_filename(name: str, max_length: int = 50) -> str:
    """清理文件名，移除非法字符"""
    # 保留字母、数字、中文、空格、下划线、连字符
    safe_chars = []
    for c in name:
        if c.isalnum() or c in " _-" or '\u4e00' <= c <= '\u9fff':
            safe_chars.append(c)
    
    result = "".join(safe_chars).strip()
    if len(result) > max_length:
        result = result[:max_length]
    
    return result or "clip"


def clip_video(input_path: str, start_time: str, end_time: str, output_path: str, 
               fast_mode: bool = False):
    """
    切片单个视频
    
    Args:
        input_path: 输入视频路径
        start_time: 开始时间
        end_time: 结束时间
        output_path: 输出路径
        fast_mode: 快速模式（使用copy编码，可能不精确）
    """
    if fast_mode:
        # 快速模式：使用copy编码，速度快但可能不精确
        cmd = [
            "ffmpeg",
            "-ss", start_time,
            "-i", input_path,
            "-to", end_time,
            "-c", "copy",
            "-avoid_negative_ts", "1",
            "-y",
            output_path
        ]
    else:
        # 精确模式：重新编码，速度慢但精确
        cmd = [
            "ffmpeg",
            "-i", input_path,
            "-ss", start_time,
            "-to", end_time,
            "-c:v", "libx264",
            "-preset", "fast",
            "-crf", "23",
            "-c:a", "aac",
            "-b:a", "128k",
            "-y",
            output_path
        ]
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    
    if result.returncode != 0:
        raise RuntimeError(f"FFmpeg错误: {result.stderr}")

    """格式化时间戳 HH:MM:SS"""
    hours = int(seconds // 3600)

def parse_srt_timestamp(time_str: str) -> float:
    """解析SRT时间戳为秒数"""
    # 格式: HH:MM:SS,mmm
    match = re.match(r'(\d+):(\d+):(\d+),(\d+)', time_str)
    if match:
        h, m, s, ms = map(int, match.groups())
        return h * 3600 + m * 60 + s + ms / 1000
    return 0


def parse_srt_file(srt_path: str) -> list:
    """解析SRT文件为segments列表"""
    segments = []
    
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 分割为块
    blocks = content.strip().split('\n\n')
    
    for block in blocks:
        lines = block.strip().split('\n')
        if len(lines) >= 3:
            # 第一行是序号
            try:
                seg_id = int(lines[0])
            except ValueError:
                continue
            
            # 第二行是时间戳
            time_line = lines[1]
            if ' --> ' in time_line:
                start_str, end_str = time_line.split(' --> ')
                start = parse_srt_timestamp(start_str.strip())
                end = parse_srt_timestamp(end_str.strip())
            else:
                continue
            
            # 剩余行是文本
            text = ' '.join(lines[2:])
            
            segments.append({
                "id": seg_id - 1,
                "start": start,
                "end": end,
                "text": text
            })
    
    return segments


def transcribe(input_path: str, output_dir: str = None, model: str = "medium", language: str = None):
    """
    转录视频/音频文件（使用命令行whisper）

        language: 语言代码（如zh/en，None为自动检测）
    """
    input_path = Path(input_path)
    if not input_path.exists():
        print(f"❌ 文件不存在: {input_path}")

def parse_srt_file(srt_path: str) -> list:
    """解析SRT文件为segments列表"""
    import re
    segments = []
    
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    blocks = content.strip().split('\n\n')
    
    for block in blocks:
        lines = block.strip().split('\n')
        if len(lines) >= 3:
            try:
                seg_id = int(lines[0])
            except ValueError:
                continue
            
            time_line = lines[1]
            if ' --> ' in time_line:
                start_str, end_str = time_line.split(' --> ')
                start = parse_srt_timestamp(start_str.strip())
                end = parse_srt_timestamp(end_str.strip())
            else:
                continue
            
            text = ' '.join(lines[2:])
            
            segments.append({
                "id": seg_id - 1,
                "start": start,
                "end": end,
                "text": text
            })
    
    return segments


def parse_srt_timestamp(time_str: str) -> float:
    """解析SRT时间戳为秒数"""
    import re
    match = re.match(r'(\d+):(\d+):(\d+),(\d+)', time_str)
    if match:
        h, m, s, ms = map(int, match.groups())
        return h * 3600 + m * 60 + s + ms / 1000
    return 0

def format_timestamp(seconds: float) -> str:
    """格式化时间戳 HH:MM:SS"""
    hours = int(seconds // 3600)
    minutes = int((seconds % 3600) // 60)
    secs = int(seconds % 60)
    return f"{hours:02d}:{minutes:02d}:{secs:02d}"


def parse_whisper_cpp_output(output_file: str) -> dict:
    """解析 whisper.cpp 的输出"""
    segments = []
    full_text = []
    
    with open(output_file, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 解析格式: [00:00:00.000 --> 00:00:05.000]  Text here
    pattern = r'\[(\d{2}:\d{2}:\d{2}\.\d{3}) --> (\d{2}:\d{2}:\d{2}\.\d{3})\]\s*(.*)'
    
    for i, match in enumerate(re.finditer(pattern, content)):
        start_str, end_str, text = match.groups()
        
        # 解析时间
        def parse_time(t):
            parts = t.split(':')
            h, m = int(parts[0]), int(parts[1])
            s = float(parts[2])
            return h * 3600 + m * 60 + s
        
        start = parse_time(start_str)
        end = parse_time(end_str)
        text = text.strip()
        
        if text:
            segments.append({
                "id": i,
                "start": start,
                "end": end,
                "text": text
            })
            full_text.append(text)
    
    return {
        "text": " ".join(full_text),
        "segments": segments
    }

def extract_audio(video_path: str, audio_path: str):
    """从视频提取音频（16kHz WAV，whisper.cpp要求）"""
    print(f"🎵 提取音频...")
    
    cmd = [
        "ffmpeg", "-y",
        "-i", video_path,
        "-ar", "16000",  # 16kHz采样率

        "-c:a", "pcm_s16le",  # 16位PCM
        audio_path
    ]
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    if result.returncode != 0:
        print(f"❌ 音频提取失败: {result.stderr}")
        sys.exit(1)
    
    print(f"✅ 音频提取完成")

                # SRT时间格式
                def to_srt_time(s):
                    h = int(s // 3600)
                    m = int((s % 3600) // 60)
                    sec = int(s % 60)
                    ms = int((s % 1) * 1000)
                    return f"{h:02d}:{m:02d}:{sec:02d},{ms:03d}"
                
                f.write(f"{i}\n")

        return result_data
    else:
