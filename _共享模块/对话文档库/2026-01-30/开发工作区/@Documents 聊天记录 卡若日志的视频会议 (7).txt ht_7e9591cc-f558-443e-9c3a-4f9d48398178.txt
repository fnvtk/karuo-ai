def build_file_content_blocks(filename, content):
    """将文件内容构建为飞书文档块"""
    blocks = []
    
    # 文件标题（二级标题）
    blocks.append({
        "block_type": 4,  # heading2
        "heading2": {
            "elements": [{
                "text_run": {
                    "content": f"📄 {filename}"
                }
            }]
        }
    })
    
    # 上传时间
    blocks.append({
        "block_type": 2,  # text
        "text": {
            "elements": [{
                "text_run": {
                    "content": f"⏰ 上传时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}",
                    "text_element_style": {
                        "text_color": 7  # 灰色
                    }
                }
            }]
        }
    })
    
    # 分割线
    blocks.append({
        "block_type": 22,  # divider
        "divider": {}
    })
    
    # 文件内容（按行拆分，避免单块过长）
    lines = content.split('\n')
    current_chunk = []
    chunk_size = 0
    max_chunk_size = 2000  # 每块最大字符数
    
    for line in lines:
        if chunk_size + len(line) > max_chunk_size and current_chunk:
            # 写入当前块
            blocks.append({
                "block_type": 2,  # text
                "text": {
                    "elements": [{
                        "text_run": {
                            "content": '\n'.join(current_chunk)
                        }
                    }]
                }
            })
            current_chunk = []
            chunk_size = 0
        
        current_chunk.append(line)
        chunk_size += len(line) + 1
    
    # 写入剩余内容
    if current_chunk:
        blocks.append({
            "block_type": 2,  # text
            "text": {
                "elements": [{
                    "text_run": {
                        "content": '\n'.join(current_chunk)
                    }
                }]
            }
        })
    
    # 结束分割线
    blocks.append({
        "block_type": 22,  # divider
        "divider": {}
    })
    
    return blocks
