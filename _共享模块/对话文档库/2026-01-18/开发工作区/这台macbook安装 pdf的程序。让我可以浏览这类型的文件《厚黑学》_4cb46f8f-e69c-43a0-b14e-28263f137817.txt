from pypdf import PdfReader, PdfWriter
import os

input_path = "/Users/karuo/Documents/个人/4.读书笔记/《厚黑学》全集PDF版_作者：李宗吾.pdf"
output_path = "/Users/karuo/Documents/个人/4.读书笔记/厚黑学_修复版.pdf"

try:
    reader = PdfReader(input_path)
    print(f"原PDF页数: {len(reader.pages)}")
    print(f"是否加密: {reader.is_encrypted}")
    
    writer = PdfWriter()
    for page in reader.pages:
        writer.add_page(page)
    
    with open(output_path, "wb") as f:
        writer.write(f)
    
    print(f"修复版已保存到: {output_path}")
    print(f"文件大小: {os.path.getsize(output_path) / 1024 / 1024:.2f} MB")
