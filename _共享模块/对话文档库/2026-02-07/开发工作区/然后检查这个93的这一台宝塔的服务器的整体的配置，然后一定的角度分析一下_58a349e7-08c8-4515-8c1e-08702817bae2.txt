def _load_creds():

def _size_fmt(size_mb):
    if size_mb is None:
        return "-"
    try:
        n = int(size_mb)
        if n >= 1024:
            return f"{n/1024:.1f}GB"
        return f"{n}MB"
    except Exception:
        return str(size_mb)


def main():

    if not sid or not skey:
