"""
本地图像生成 - 使用 Ollama 的 Z-Image Turbo / FLUX.2 Klein
解决 Cursor generate_image 连接超时问题，完全本地运行，无需网络

用法:
  python local_image_gen.py "动漫风格少女，有刘海，戴手套，旁边有小猫"
  python local_image_gen.py "your prompt" --output ./output.png
  python local_image_gen.py --check  # 检查模型是否已安装
"""

import argparse
import os
import subprocess
import sys
from pathlib import Path

# 默认输出目录
DEFAULT_OUTPUT_DIR = Path(__file__).parent.parent / "outputs"
OLLAMA_IMAGE_MODELS = ["x/z-image-turbo", "x/flux2-klein:4b", "x/flux2-klein:9b"]


def check_ollama() -> bool:
    """检查 ollama 是否运行"""
    try:
        result = subprocess.run(
            ["ollama", "list"],
            capture_output=True,
            text=True,
            timeout=10,
        )
        return result.returncode == 0
    except (subprocess.TimeoutExpired, FileNotFoundError):
        return False


def check_image_model() -> str | None:
    """检查是否有图像生成模型，返回第一个可用的"""
    try:
        result = subprocess.run(
            ["ollama", "list"],
            capture_output=True,
            text=True,
            timeout=10,
        )
        if result.returncode != 0:
            return None
        output = result.stdout.lower()
        for model in OLLAMA_IMAGE_MODELS:
            # 检查模型名（如 z-image-turbo, flux2-klein）
            short_name = model.split("/")[-1].split(":")[0]
            if short_name in output:
                return model
        return None
    except Exception:
        return None


def generate_image(
    prompt: str,
    model: str = "x/z-image-turbo",
    output_dir: Path | None = None,
    width: int | None = None,
    height: int | None = None,
) -> Path | None:
    """
    使用 ollama 生成图像
    返回保存的图片路径，失败返回 None
    """
    output_dir = output_dir or DEFAULT_OUTPUT_DIR
    output_dir.mkdir(parents=True, exist_ok=True)

    # ollama run 会保存到当前目录，所以先 cd 到输出目录
    env = os.environ.copy()
    env["OLLAMA_MODELS"] = str(Path.home() / ".ollama" / "models")

    cmd = ["ollama", "run", model, prompt]
    try:
        result = subprocess.run(
            cmd,
            cwd=str(output_dir),
            capture_output=True,
            text=True,
            timeout=120,  # 图像生成可能需要 30-90 秒
            env=env,
        )
        if result.returncode != 0:
            print(f"❌ 生成失败: {result.stderr}", file=sys.stderr)
            return None

        # ollama 会保存为 00001.png 等，找最新的
        png_files = sorted(output_dir.glob("*.png"), key=os.path.getmtime, reverse=True)
        if png_files:
            return png_files[0]
        return None
    except subprocess.TimeoutExpired:
        print("❌ 生成超时（120秒），可尝试简化提示词", file=sys.stderr)
        return None
    except Exception as e:
        print(f"❌ 异常: {e}", file=sys.stderr)
        return None


def main():

    parser.add_argument("-o", "--output", type=Path, help="输出目录或文件名")

        print("✅ Ollama 已运行")

        model = check_image_model()
        if model:
            print(f"✅ 图像模型已安装: {model}")
            print("\n可直接使用:")
            print(f'  python local_image_gen.py "你的提示词"')
        else:
            print("⚠️ 未检测到图像生成模型，请先安装:")
            print("   ollama pull x/z-image-turbo   # 约12GB，支持中英文")
            print("   或 ollama pull x/flux2-klein:4b  # 约5.7GB，更快")
        sys.exit(0)

    if not args.prompt:
        parser.print_help()
        print("\n示例:")
        print('  python local_image_gen.py "动漫风格少女，有刘海，戴手套，旁边有小猫"')
        sys.exit(1)

    if not check_ollama():
        print("❌ Ollama 未运行，请先启动", file=sys.stderr)
        sys.exit(1)

    model = check_image_model() or args.model
    output_dir = args.output if args.output and args.output.is_dir() else DEFAULT_OUTPUT_DIR

    print(f"🎨 正在生成: {args.prompt[:50]}...")
    print(f"   模型: {model} | 输出: {output_dir}")

    path = generate_image(args.prompt, model=model, output_dir=output_dir)

        sys.exit(1)

    main()

## 一、问题分析

### 1.1 错误现象

- **错误提示**：Connection Error - "The connection stalled. Please try again."
- **出现场景**：使用 Cursor 内置的 `generate_image` 工具生成图片时
- **请求 ID**：每次失败会显示唯一 ID（如 `b24a6e09-b140-4800-9e6f-d960b58caebc`）

### 1.2 根本原因

| 原因 | 说明 |
|:---|:---|
| **云端超时** | Cursor 的 generate_image 连接云端 API，图像生成需 30-90 秒，易触发超时 |
| **网络不稳定** | 国内访问 Cursor 云端可能受网络波动、VPN、防火墙影响 |
| **服务端负载** | Cursor 图像服务高负载时响应变慢，导致连接中断 |
| **平台限制** | 非 Cursor 可控，属于 Cursor 与模型提供方之间的连接问题 |

### 1.3 官方反馈

- Cursor 论坛有类似反馈：[Image Generation Connection Error in Cursor 2.4](https://forum.cursor.com/t/image-generation-connection-error-in-cursor-2-4/149660)
- 建议联系支持：hi@cursor.com 或 docs.cursor.com

## 二、临时缓解（Cursor 内）

1. **点击 Resume**：出错弹窗中的「Resume」有时可恢复
2. **缩短提示词**：减少描述长度，降低处理时间
3. **更换网络**：尝试切换 WiFi/热点，或关闭 VPN
4. **错峰使用**：避开高峰时段再试

def generate_image(prompt: str, output_path: Path, width: int = 1024, height: int = 1024) -> bool:
    """
    使用 Pollinations API 生成图像（无需任何依赖）
    返回 True 表示成功
    """
    output_path.parent.mkdir(parents=True, exist_ok=True)

            timeout=120,

        if result.returncode != 0:
            return False
        return output_path.exists() and output_path.stat().st_size > 0

        return False
