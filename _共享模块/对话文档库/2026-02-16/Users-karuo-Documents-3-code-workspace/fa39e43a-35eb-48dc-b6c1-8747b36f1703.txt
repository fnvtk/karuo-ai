async function createPortraitTemplate(data: {
  name: string
  description: string
  criteria: any
}) {
  try {
    const client = await getMongoClient()
    const collection = client.db('KR').collection('用户估值')
    
    // 构建查询条件统计用户数
    const query: any = {}
    if (data.criteria.minScore !== undefined) {
      query.user_evaluation_score = { $gte: data.criteria.minScore }
    }
    if (data.criteria.maxScore !== undefined) {
      query.user_evaluation_score = {
        ...(query.user_evaluation_score || {}),
        $lt: data.criteria.maxScore
      }
    }
    if (data.criteria.userLevel) {
      query.user_level = data.criteria.userLevel
    }
    if (data.criteria.province) {
      query.province = data.criteria.province
    }
    if (data.criteria.tags?.length) {
      query.source_channels = { $in: data.criteria.tags }
    }
    
    // 统计符合条件的用户数
    const [userCount, avgScoreResult] = await Promise.all([
      collection.countDocuments(query),
      collection.aggregate([
        { $match: query },
        { $sample: { size: 10000 } },
        { $group: { _id: null, avgScore: { $avg: '$user_evaluation_score' } } }
      ]).toArray()
    ])
    
    const avgRfm = avgScoreResult[0]?.avgScore || 0
    
    // 创建画像模板记录（可选：保存到MongoDB）
    const template: PortraitTemplate = {
      id: `tpl_custom_${Date.now()}`,
      name: data.name,
      description: data.description,
      criteria: data.criteria,
      userCount,
      avgRfm: Math.round(avgRfm),
      status: 'active',
      createdAt: new Date().toISOString(),
      createdBy: 'admin'
    }
    
    // TODO: 保存到 shensheshou.portrait_templates 集合
    // await client.db('shensheshou').collection('portrait_templates').insertOne(template)
    
    return template
  } catch (error) {
    console.error('创建画像失败:', error)
    throw error
  }
}

// 辅助函数：解析规则字符串为MongoDB查询
function parseRuleToQuery(rules: string): any {
  const query: any = {}
  
  // 简单规则解析
  // 支持格式: field >= value, field = value, field IN (...)
  const patterns = [
    { regex: /(\w+)\s*>=\s*(\d+)/g, handler: (field: string, value: string) => ({ [field]: { $gte: parseInt(value) } }) },
    { regex: /(\w+)\s*<=\s*(\d+)/g, handler: (field: string, value: string) => ({ [field]: { $lte: parseInt(value) } }) },
    { regex: /(\w+)\s*>\s*(\d+)/g, handler: (field: string, value: string) => ({ [field]: { $gt: parseInt(value) } }) },
    { regex: /(\w+)\s*<\s*(\d+)/g, handler: (field: string, value: string) => ({ [field]: { $lt: parseInt(value) } }) },
    { regex: /(\w+)\s*=\s*'([^']+)'/g, handler: (field: string, value: string) => ({ [field]: value }) },
    { regex: /(\w+)\s*=\s*"([^"]+)"/g, handler: (field: string, value: string) => ({ [field]: value }) },
  ]
  
  // 字段映射
  const fieldMap: { [key: string]: string } = {
    'rfm_score': 'user_evaluation_score',
    'province': 'province',
    'city': 'city',
    'user_level': 'user_level',
  }
  
  for (const pattern of patterns) {
    let match
    while ((match = pattern.regex.exec(rules)) !== null) {
      const field = fieldMap[match[1]] || match[1]
      const value = match[2]
      const condition = pattern.handler(field, value)
      Object.assign(query, condition)
    }
  }
  
  return query
}

// 标签接口

  id: string
  name: string
  category: string
  type: 'system' | 'custom' | 'ai'
  description: string
  userCount: number
  rules?: string
  createdAt: string
  enabled: boolean
}

// 标签分类

  { id: 'value', name: '价值标签', icon: Star, color: 'bg-yellow-100 text-yellow-700', bgColor: 'bg-yellow-50' },
  { id: 'behavior', name: '行为标签', icon: Activity, color: 'bg-blue-100 text-blue-700', bgColor: 'bg-blue-50' },
  { id: 'region', name: '地域标签', icon: MapPin, color: 'bg-green-100 text-green-700', bgColor: 'bg-green-50' },
  { id: 'lifecycle', name: '生命周期', icon: Clock, color: 'bg-purple-100 text-purple-700', bgColor: 'bg-purple-50' },
  { id: 'preference', name: '偏好标签', icon: Heart, color: 'bg-red-100 text-red-700', bgColor: 'bg-red-50' },
]

const CLEANING_RULES = [

export default function TagsPage() {
  const [tags, setTags] = useState<Tag[]>([])
  const [loading, setLoading] = useState(true)
  const [searchQuery, setSearchQuery] = useState("")
  const [activeCategory, setActiveCategory] = useState("all")
  const [activeTab, setActiveTab] = useState("tags")
  const [showAddDialog, setShowAddDialog] = useState(false)
  const [showEditDialog, setShowEditDialog] = useState(false)
  const [showDeleteDialog, setShowDeleteDialog] = useState(false)
  const [selectedTag, setSelectedTag] = useState<Tag | null>(null)
  const [cleaningRules, setCleaningRules] = useState(CLEANING_RULES)
  
  // 新标签/编辑表单
  const [tagForm, setTagForm] = useState({
    name: '',
    category: 'value',
    description: '',
    rules: '',
    enabled: true
  })

  // 从API加载真实标签
  useEffect(() => {
    loadTags()
  }, [])
  
  const loadTags = async () => {
    setLoading(true)
    try {
      const res = await fetch('/api/tags')
      const data = await res.json()
      
      if (data.success && data.tags) {
        setTags(data.tags.map((t: any) => ({
          ...t,
          enabled: t.enabled !== false,
          rules: t.rules || '',
        })))
      }
    } catch (error) {
      console.error('加载标签失败:', error)
    } finally {
      setLoading(false)
    }
  }

  const formatNumber = (num: number): string => {
    if (num >= 1000000000) return `${(num / 1000000000).toFixed(2)}B`
    if (num >= 1000000) return `${(num / 1000000).toFixed(1)}M`
    if (num >= 1000) return `${(num / 1000).toFixed(0)}K`
    return num.toLocaleString()
  }

  const getCategoryConfig = (categoryId: string) => {
    return TAG_CATEGORIES.find(c => c.id === categoryId) || TAG_CATEGORIES[0]
  }

  const getTypeColor = (type: string) => {
    switch (type) {
      case 'system': return 'bg-blue-100 text-blue-700'
      case 'ai': return 'bg-purple-100 text-purple-700'
      case 'custom': return 'bg-orange-100 text-orange-700'
      default: return 'bg-gray-100 text-gray-700'
    }
  }

  const getTypeName = (type: string) => {
    switch (type) {
      case 'system': return '系统'
      case 'ai': return 'AI'
      case 'custom': return '自定义'
      default: return type
    }
  }

  // 筛选标签
  const filteredTags = tags.filter(tag => {
    if (activeCategory !== 'all' && tag.category !== activeCategory) return false
    if (searchQuery) {
      const query = searchQuery.toLowerCase()
      return tag.name.toLowerCase().includes(query) || 
             tag.description?.toLowerCase().includes(query)
    }
    return true
  })

  // 按分类统计
  const categoryStats = TAG_CATEGORIES.map(cat => ({
    ...cat,
    count: tags.filter(t => t.category === cat.id).length,
    userCount: tags.filter(t => t.category === cat.id).reduce((sum, t) => sum + t.userCount, 0)
  }))

  // 添加标签
  const handleAddTag = async () => {
    if (!tagForm.name.trim()) {
      alert('请输入标签名称')
      return
    }
    
    setLoading(true)
    try {
      const res = await fetch('/api/tags', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(tagForm)
      })
      
      const data = await res.json()
      if (data.success) {
        alert(data.message || '标签创建成功')
        setShowAddDialog(false)
        setTagForm({ name: '', category: 'value', description: '', rules: '', enabled: true })
        loadTags()
      } else {
        alert(data.error || '创建失败')
      }
    } catch (error) {
      console.error('创建标签失败:', error)
      alert('创建失败，请重试')
    } finally {
      setLoading(false)
    }
  }
  
  // 编辑标签
  const handleEditTag = async () => {
    if (!selectedTag) return
    
    setLoading(true)
    try {
      const res = await fetch('/api/tags', {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          id: selectedTag.id,
          ...tagForm
        })
      })
      
      const data = await res.json()
      if (data.success) {
        alert(data.message || '标签更新成功')
        setShowEditDialog(false)
        setSelectedTag(null)
        loadTags()
      } else {
        alert(data.error || '更新失败')
      }
    } catch (error) {
      console.error('更新标签失败:', error)
      alert('更新失败，请重试')
    } finally {
      setLoading(false)
    }
  }
  
  // 删除标签

    if (!selectedTag) return
    
    setLoading(true)
    try {

        method: 'DELETE'
      })
      
      const data = await res.json()
      if (data.success) {
        alert(data.message || '标签删除成功')
        setShowDeleteDialog(false)
        setSelectedTag(null)
        loadTags()
      } else {
        alert(data.error || '删除失败')
      }
    } catch (error) {
      console.error('删除标签失败:', error)
      alert('删除失败，请重试')

      setLoading(false)
    }

    setTagForm({

      category: tag.category,
      description: tag.description || '',
      rules: tag.rules || '',
      enabled: tag.enabled
    })
    setShowEditDialog(true)
  }
  
  // 打开删除确认

    setSelectedTag(tag)
    setShowDeleteDialog(true)
  }

        <div className="flex items-center justify-between">

        {/* 标签页切换 */}

          {/* 标签列表 */}
          <TabsContent value="tags" className="mt-4 space-y-4">
            {/* 分类统计 */}
            <div className="grid grid-cols-5 gap-4">
              {categoryStats.map(cat => {
                const Icon = cat.icon
                return (
                  <Card 
                    key={cat.id}

                    } ${cat.bgColor} hover:shadow-md`}
                    onClick={() => setActiveCategory(activeCategory === cat.id ? 'all' : cat.id)}
                  >
                    <CardContent className="p-4">
                      <div className="flex items-center gap-3">
                        <div className={`p-2 rounded-lg bg-white/80`}>

                <Input

                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}

              </div>
              {activeCategory !== 'all' && (
                <Button variant="outline" onClick={() => setActiveCategory('all')}>

                </Button>
              )}
            </div>

            {/* 标签列表 */}
            {loading ? (
              <div className="flex items-center justify-center py-12">

                <span className="ml-3 text-gray-500">加载中...</span>

            ) : (
              <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">

                          </div>
                        </div>
                        
                        <p className="text-sm text-gray-600 mb-3 line-clamp-1">{tag.description}</p>
                        
                        {tag.rules && (
                          <div className="text-xs font-mono text-purple-600 bg-purple-50 p-2 rounded mb-3 truncate">
                            {tag.rules}
                          </div>
                        )}
                        
                        <div className="flex items-center justify-between text-xs text-gray-500">
                          <div className="flex items-center gap-1">
                            <Users className="h-3 w-3" />
                            <span className="font-semibold text-gray-900">{formatNumber(tag.userCount)}</span>
                            <span>用户</span>
                          </div>
                          <div className="flex items-center gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
                            <Button variant="ghost" size="icon" className="h-7 w-7" onClick={() => openEditDialog(tag)}>
                              <Edit className="h-3.5 w-3.5" />

                              <Trash2 className="h-3.5 w-3.5" />
                            </Button>
                          </div>
                        </div>
                      </CardContent>
                    </Card>

                })}
              </div>

            {filteredTags.length === 0 && !loading && (

              </div>
            )}

          {/* 清洗规则 */}
          <TabsContent value="cleaning" className="mt-4">
            <Card className="border-0 shadow-lg bg-white/90">

                <CardTitle className="flex items-center gap-2">
                  <Zap className="h-5 w-5 text-yellow-500" />
                  AI清洗规则配置
                </CardTitle>
                <p className="text-sm text-gray-500">配置数据清洗规则，自动标准化和校验数据字段</p>
              </CardHeader>
              <CardContent className="space-y-4">
                {cleaningRules.map((rule, index) => (
                  <Card key={rule.id} className="border shadow-sm">
                    <CardContent className="p-4">
                      <div className="flex items-center justify-between">
                        <div className="flex items-center gap-4">
                          <div className="w-8 h-8 rounded-full bg-purple-100 flex items-center justify-center text-purple-600 font-bold">
                            {index + 1}
                          </div>
                          <div>
                            <div className="font-semibold text-gray-900">{rule.name}</div>
                            <div className="text-sm text-gray-500">{rule.description}</div>
                            <Badge variant="outline" className="mt-1">字段: {rule.field}</Badge>
                          </div>
                        </div>
                        <div className="flex items-center gap-4">
                          <Switch 
                            checked={rule.enabled}
                            onCheckedChange={(checked) => {
                              const newRules = [...cleaningRules]
                              newRules[index].enabled = checked
                              setCleaningRules(newRules)
                            }}
                          />
                          <Button variant="ghost" size="icon">
                            <Settings className="h-4 w-4" />
                          </Button>
                        </div>
                      </div>
                    </CardContent>
                  </Card>
                ))}
                
                <div className="flex justify-end gap-2 pt-4">
                  <Button variant="outline">
                    <Plus className="h-4 w-4 mr-2" />
                    添加规则

                    执行清洗

          <TabsContent value="algorithm" className="mt-4">
            <Card className="border-0 shadow-lg bg-white/90">
              <CardHeader>
                <CardTitle className="flex items-center gap-2">
                  <GitBranch className="h-5 w-5 text-blue-500" />
                  标签清洗算法流程

              <CardContent>
                {/* 流程图 */}
                <div className="relative">
                  {/* 流程步骤 */}
                  <div className="flex items-stretch justify-between gap-4">
                    {/* 步骤1: 数据接入 */}
                    <div className="flex-1">
                      <Card className="border-2 border-blue-200 bg-blue-50 h-full">
                        <CardContent className="p-4 text-center">
                          <div className="w-12 h-12 rounded-full bg-blue-500 text-white flex items-center justify-center mx-auto mb-3">

                          <p className="text-xs text-blue-600 mt-2">
                            多源数据导入<br/>
                            支持CSV/API/数据库
                          </p>
                          <div className="mt-3 space-y-1">
                            <Badge variant="outline" className="text-xs">KR_腾讯</Badge>

                          </div>
