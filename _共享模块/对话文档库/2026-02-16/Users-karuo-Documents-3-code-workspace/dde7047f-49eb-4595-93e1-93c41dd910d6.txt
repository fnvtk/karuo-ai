def timestamp_to_date(ts_str):
    """Unix时间戳转可读日期"""
    try:
        ts = int(ts_str)
        if ts <= 0:
            return ""
        return datetime.fromtimestamp(ts).strftime("%Y-%m-%d %H:%M:%S")
    except (ValueError, TypeError, OSError):
        return ""


def calculate_rfm(docs):
    """
    RFM 用户价值评估
    R (Recency)  - 最近活跃度：基于 lastactivity
    F (Frequency) - 活跃频率：基于 posts 发帖数
    M (Monetary)  - 贡献值：基于 credits 积分
    
    评分规则：1-5分，5分最高
    """
    # 收集数值用于分位数计算
    recencies = []
    frequencies = []
    monetaries = []
    
    for doc in docs:
        try:
            recencies.append(int(doc.get("lastactivity", 0) or 0))
        except (ValueError, TypeError):
            recencies.append(0)
        try:
            frequencies.append(int(doc.get("posts", 0) or 0))
        except (ValueError, TypeError):
            frequencies.append(0)
        try:
            monetaries.append(int(doc.get("credits", 0) or 0))
        except (ValueError, TypeError):
            monetaries.append(0)
    
    # 计算分位数边界（20%, 40%, 60%, 80%）
    def get_quintiles(values):
        sorted_vals = sorted(values)
        n = len(sorted_vals)
        return [
            sorted_vals[int(n * 0.2)],
            sorted_vals[int(n * 0.4)],
            sorted_vals[int(n * 0.6)],
            sorted_vals[int(n * 0.8)],
        ]
    
    r_q = get_quintiles(recencies)
    f_q = get_quintiles(frequencies)
    m_q = get_quintiles(monetaries)
    
    def score(value, quintiles, reverse=False):
        """根据分位数打分1-5"""
        if reverse:
            if value <= quintiles[0]: return 5
            elif value <= quintiles[1]: return 4
            elif value <= quintiles[2]: return 3
            elif value <= quintiles[3]: return 2
            else: return 1
        else:
            if value <= quintiles[0]: return 1
            elif value <= quintiles[1]: return 2
            elif value <= quintiles[2]: return 3
            elif value <= quintiles[3]: return 4
            else: return 5
    
    # 计算每个用户的RFM评分
    rfm_results = []
    for i, doc in enumerate(docs):
        r_val = recencies[i]
        f_val = frequencies[i]
        m_val = monetaries[i]
        
        # R: 越近越好 → 值越大分越高
        r_score = score(r_val, r_q, reverse=False)
        # F: 发帖越多越好
        f_score = score(f_val, f_q, reverse=False)
        # M: 积分越多越好
        m_score = score(m_val, m_q, reverse=False)
        
        # 综合评分
        total = r_score + f_score + m_score
        
        # 用户价值等级
        if total >= 13:
            level = "高价值用户"
        elif total >= 10:
            level = "中高价值用户"
        elif total >= 7:
            level = "中等价值用户"
        elif total >= 4:
            level = "低价值用户"
        else:
            level = "流失用户"
        
        # 用户类型标签
        if r_score >= 4 and f_score >= 4 and m_score >= 4:
            user_type = "重要价值用户"
        elif r_score >= 4 and f_score < 3 and m_score >= 4:
            user_type = "重要发展用户"
        elif r_score < 3 and f_score >= 4 and m_score >= 4:
            user_type = "重要保持用户"
        elif r_score < 3 and f_score < 3 and m_score >= 4:
            user_type = "重要挽留用户"
        elif r_score >= 4 and f_score >= 4 and m_score < 3:
            user_type = "一般价值用户"
        elif r_score >= 4 and f_score < 3 and m_score < 3:
            user_type = "一般发展用户"
        elif r_score < 3 and f_score >= 4 and m_score < 3:
            user_type = "一般保持用户"
        else:
            user_type = "一般挽留用户"
        
        rfm_results.append({
            "R评分": r_score,
            "F评分": f_score,
            "M评分": m_score,
            "RFM总分": total,
            "价值等级": level,
            "用户类型": user_type,
        })
    
    return rfm_results
