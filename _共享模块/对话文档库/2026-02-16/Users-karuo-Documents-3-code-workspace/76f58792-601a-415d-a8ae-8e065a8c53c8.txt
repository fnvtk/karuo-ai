上帝之眼 - 股票热点资讯排行服务

从 forum_posts 按「每只股票在各论坛」汇总两个指标：浏览量(views)、评论数(comments)。
形成股票热点资讯排行，供首页/选股使用。按小时快照写入 stock_hot_ranking。
"""

from typing import List, Dict, Any, Optional
from datetime import datetime
from collections import defaultdict


async def aggregate_from_mongodb(db) -> List[Dict[str, Any]]:
    """
    从 MongoDB forum_posts 聚合：每只股票在各 platform 的 views、replies 汇总。
    返回列表 [{ stock_code, total_views, total_comments, by_platform: { platform: { views, comments } } }, ...]
    """
    pipeline = [
        {"$match": {"content.stocks_mentioned": {"$exists": True, "$ne": []}}},
        {"$unwind": "$content.stocks_mentioned"},
        {"$group": {
            "_id": {"stock_code": "$content.stocks_mentioned", "platform": "$platform"},
            "views": {"$sum": {"$ifNull": ["$meta.views", 0]}},
            "comments": {"$sum": {"$ifNull": ["$meta.replies", 0]}},
        }},
        {"$group": {
            "_id": "$_id.stock_code",
            "by_platform": {
                "$push": {
                    "platform": "$_id.platform",
                    "views": "$views",
                    "comments": "$comments",
                }
            },
            "total_views": {"$sum": "$views"},
            "total_comments": {"$sum": "$comments"},
        }},
        {"$sort": {"total_views": -1, "total_comments": -1}},
        {"$limit": 500},
    ]
    out = []
    async for doc in db.forum_posts.aggregate(pipeline):
        by_platform = {}
        for p in doc.get("by_platform") or []:
            plat = p.get("platform") or "未知"
            by_platform[plat] = {"views": p.get("views", 0), "comments": p.get("comments", 0)}
        out.append({
            "stock_code": doc["_id"],
            "total_views": doc.get("total_views", 0),
            "total_comments": doc.get("total_comments", 0),
            "by_platform": by_platform,
        })
    return out


async def aggregate_from_mysql(mysql_get_posts_for_ranking) -> List[Dict[str, Any]]:
    """
    从 MySQL forum_posts（通过回调拉取帖子列表）在内存中聚合。
    要求每条: platform, views, replies, stocks_mentioned(list)。
    """
    rows = await mysql_get_posts_for_ranking()
    # stock_code -> { total_views, total_comments, by_platform: { platform: { views, comments } } }
    agg = defaultdict(lambda: {"total_views": 0, "total_comments": 0, "by_platform": defaultdict(lambda: {"views": 0, "comments": 0})})
    for r in rows:
        platform = r.get("platform") or "未知"
        views = int(r.get("views") or 0)
        replies = int(r.get("replies") or 0)
        stocks = r.get("stocks_mentioned")
        if not isinstance(stocks, list):
            if isinstance(stocks, str):
                import json
                try:
                    stocks = json.loads(stocks) if stocks else []
                except Exception:
                    stocks = []
            else:
                stocks = []
        for code in stocks:
            if not code or not str(code).strip():
                continue
            code = str(code).strip()
            agg[code]["total_views"] += views
            agg[code]["total_comments"] += replies
            agg[code]["by_platform"][platform]["views"] += views
            agg[code]["by_platform"][platform]["comments"] += replies
    # 转成列表并排序
    result = []
    for stock_code, v in agg.items():
        by_platform = {p: {"views": d["views"], "comments": d["comments"]} for p, d in v["by_platform"].items()}
        result.append({
            "stock_code": stock_code,
            "total_views": v["total_views"],
            "total_comments": v["total_comments"],
            "by_platform": by_platform,
        })
    result.sort(key=lambda x: (-x["total_views"], -x["total_comments"]))
    return result[:500]


def build_ranking_doc(
    items: List[Dict[str, Any]],
    snapshot_time: Optional[datetime] = None,
) -> Dict[str, Any]:
    """构建写入 stock_hot_ranking 的文档（单次快照）。"""
    ts = snapshot_time or datetime.utcnow()
    # 整点
    ts = ts.replace(minute=0, second=0, microsecond=0)
    return {
        "snapshot_time": ts,
        "rank_list": items,
        "created_at": datetime.utcnow(),
    }


async def save_ranking_to_mongodb(db, doc: Dict[str, Any]) -> str:
    """写入 MongoDB stock_hot_ranking 集合，按 snapshot_time 唯一替换当次整点。"""
    ts = doc["snapshot_time"]
    await db.stock_hot_ranking.update_one(
        {"snapshot_time": ts},
        {"$set": {**doc, "updated_at": datetime.utcnow()}},
        upsert=True,
    )
    return str(ts)


async def run_ranking(db, use_mysql: bool, mysql_get_posts_for_ranking=None) -> List[Dict[str, Any]]:
    """
    执行一次排行聚合。若 use_mysql 为 True 且提供了 mysql_get_posts_for_ranking，则从 MySQL 聚合；
    否则从 MongoDB forum_posts 聚合。结果写入 stock_hot_ranking 并返回 rank_list。
    """
    if use_mysql and mysql_get_posts_for_ranking:
        items = await aggregate_from_mysql(mysql_get_posts_for_ranking)
    else:
        items = await aggregate_from_mongodb(db)
    doc = build_ranking_doc(items)
    await save_ranking_to_mongodb(db, doc)
    return items

def _get_crawler_class(source: Dict) -> Optional[Any]:
    """根据 source['crawler'] 动态加载爬虫类；无则返回 None（占位）。"""
    path = source.get("crawler")
    if not path:
        return None
    try:
        mod_name, cls_name = path.split(".")
        from importlib import import_module
        mod = import_module(f"services.collect.{mod_name}")
        return getattr(mod, cls_name)
    except Exception:
        return None


async def crawl_one_source(source: Dict, count: int = 15) -> List[Dict]:
    """
    单源采集。返回与 forum_posts 同构的列表（含 platform, meta.views, meta.replies, content.stocks_mentioned 等）。
    占位源返回 []。
    """
    Klass = _get_crawler_class(source)
    if not Klass:
        return []
    try:
        async with Klass() as crawler:
            posts = await crawler.crawl_hot_posts(count=count)
        # 统一带上 platform 名称
        name = source.get("name", "")
        for p in posts:
            p["platform"] = p.get("platform") or name
        return posts
    except Exception:
        return []

            continue
        for p in posts or []:

            if pid and pid not in seen:

MIN_CONFIDENCE = 0.55       # 至少匹配到一定利好/利空关键字才视为有效
MIN_HOT_SCORE = 10.0       # 热度分下限（views*0.3+likes*0.5+replies*0.2）
MIN_VIEWS_OR_REPLIES = 50  # 浏览量或评论数至少其一达标，视为影响较大

def _text_from_post(post: Dict) -> str:
    c = post.get("content") or {}
    return ((c.get("title") or "") + " " + (c.get("body") or ""))[:2000]


def _hot_score_from_post(post: Dict) -> float:
    meta = post.get("meta") or {}
    v = meta.get("views", 0) or 0
    lk = meta.get("likes", 0) or 0
    rp = meta.get("replies", 0) or 0
    return round(v * 0.3 + lk * 0.5 + rp * 0.2, 2)


def _is_high_impact(post: Dict, hot_score: float) -> bool:
    """是否视为影响较大：热度或互动量达标。"""
    if hot_score >= MIN_HOT_SCORE:
        return True
    meta = post.get("meta") or {}
    v = meta.get("views", 0) or 0
    rp = meta.get("replies", 0) or 0
    return v >= MIN_VIEWS_OR_REPLIES or rp >= MIN_VIEWS_OR_REPLIES


async def analyze_and_filter(
    posts: List[Dict],
    min_confidence: float = MIN_CONFIDENCE,
    allowed_sentiments: tuple = ALLOWED_SENTIMENTS,
) -> List[Dict]:
    """
    对帖子做情感分析（利好/利空关键字），并过滤出：情感为利好或利空、置信度达标、影响较大。
    为每条附加 analysis（含 label、matched 关键词）、impact_score，便于入库与展示。
    """
    from services.nlp.sentiment_analyzer import SentimentAnalyzer
    analyzer = SentimentAnalyzer()
    result = []
    for post in posts:
        text = _text_from_post(post)
        if not text.strip():
            continue
        analysis = analyzer.analyze_with_keywords(text)
        label = analysis.get("label", "中性")
        confidence = analysis.get("confidence", 0)
        if label not in allowed_sentiments or confidence < min_confidence:
            continue
        hot_score = _hot_score_from_post(post)
        if not _is_high_impact(post, hot_score):
            continue
        # 影响力综合分：热度 + 置信度加权，便于排序
        impact_score = round(hot_score * 0.6 + confidence * 100 * 0.4, 2)
        doc = {
            **post,
            "analysis": {
                "label": label,
                "sentiment": analysis.get("sentiment", "neutral"),
                "confidence": confidence,
                "positive_count": analysis.get("positive_count", 0),
                "negative_count": analysis.get("negative_count", 0),
                "matched_positive_keywords": analysis.get("matched_positive_keywords", []),
                "matched_negative_keywords": analysis.get("matched_negative_keywords", []),
            },
            "impact_score": impact_score,
            "hot_score": hot_score,
            "news_analyzed_at": datetime.utcnow(),
        }

+    """
+    单只股票舆情热度指标（用于前台股票详情）：
+    - 总帖子数、总浏览量、总评论数
+    - 平均互动量（每帖评论）
+    - 近 N 日趋势（按天 posts/views/replies）
+    - 热度分（便于前端画分析图）
+    """
+    from datetime import datetime, timedelta
+    since = (datetime.utcnow() - timedelta(days=days)).replace(hour=0, minute=0, second=0, microsecond=0)
+    summary = {
+        "post_count": 0,
+        "views_total": 0,
+        "replies_total": 0,

+        "market_heat_score": 0.0,

+function buildLinePath(values: number[], w = 320, h = 120) {
+  if (!values.length) return ''
+  const max = Math.max(...values, 1)
+  const min = Math.min(...values, 0)
+  const range = Math.max(max - min, 1)
+  return values
+    .map((v, i) => {
+      const x = (i / Math.max(values.length - 1, 1)) * w
+      const y = h - ((v - min) / range) * h
+      return `${i === 0 ? 'M' : 'L'}${x.toFixed(2)} ${y.toFixed(2)}`
+    })
+    .join(' ')
