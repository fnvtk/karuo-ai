# 卡若AI · 本地 → GitHub 同步脚本（不定时/周期性上传）
# 用法：./push-to-github.sh [提交说明]
# 若未传说明则使用默认时间戳说明

  if git remote get-url origin &>/dev/null; then
    GITHUB_REPO_URL=$(git remote get-url origin)
  else
    echo "未配置 GitHub 仓库。请任选一种方式："
    echo "  1) 在脚本内设置 GITHUB_REPO_URL"

  fi

# 确保远程已添加
if ! git remote get-url origin &>/dev/null; then
  git remote add origin "$GITHUB_REPO_URL"
fi
current_url=$(git remote get-url origin)
if [[ "$current_url" != "$GITHUB_REPO_URL" ]]; then
  git remote set-url origin "$GITHUB_REPO_URL"
fi

# 检查是否有变更（未跟踪或已修改）
if git diff --quiet && git diff --cached --quiet && [[ -z $(git status -s) ]]; then

  exit 0
fi

msg="${1:-同步卡若AI：$(date '+%Y-%m-%d %H:%M')}"
git commit -m "$msg" || true
if git rev-parse @{u} &>/dev/null 2>&1; then
  git push -u origin HEAD 2>/dev/null || git push origin HEAD
else
  git push -u origin HEAD
fi
echo "[$(date '+%Y-%m-%d %H:%M')] 已推送到 GitHub。"

# 卡若AI · 本地 → GitHub 同步（路径：个人/卡若AI）
# 用法：./scripts/push-to-github.sh [提交说明]

set -e
REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
cd "$REPO_ROOT"

if ! git remote get-url origin &>/dev/null; then
  echo "未配置 origin。请执行: git remote add origin <仓库URL>"
  exit 1
fi

if git diff --quiet && git diff --cached --quiet && [[ -z $(git status -s) ]]; then
  echo "[$(date '+%Y-%m-%d %H:%M')] 无变更，跳过推送。"
  exit 0
fi

git add -A
msg="${1:-同步卡若AI：$(date '+%Y-%m-%d %H:%M')}"
git commit -m "$msg" || true
git push origin HEAD
echo "[$(date '+%Y-%m-%d %H:%M')] 已推送到 GitHub。"
