def parse_timestamp(time_str: str) -> float:
    """解析时间戳为秒数 (支持 HH:MM:SS 或 MM:SS 格式)"""
    parts = time_str.replace(',', '.').split(":")

def format_timestamp(seconds: float) -> str:
    """格式化秒数为时间戳 HH:MM:SS"""
    h = int(seconds // 3600)
    m = int((seconds % 3600) // 60)
    s = int(seconds % 60)
    return f"{h:02d}:{m:02d}:{s:02d}"


def build_video_filter(config: dict) -> str:
    """构建FFmpeg视频滤镜链"""
    filters = []
    
    # 裁剪(Soul录屏专用)
    if config.get("crop_filter"):
        filters.append(config["crop_filter"])
    
    # 亮度+饱和度
    eq_parts = []
    if config.get("brightness"):
        eq_parts.append(f"brightness={config['brightness']}")
    if config.get("saturation"):
        eq_parts.append(f"saturation={config['saturation']}")
    if eq_parts:
        filters.append(f"eq={':'.join(eq_parts)}")
    
    # 锐化
    if config.get("sharpen"):
        filters.append(f"unsharp={config['sharpen']}")
    
    return ",".join(filters)


def build_audio_filter(config: dict) -> str:
    """构建FFmpeg音频滤镜链"""
    filters = []
    
    # FFT降噪
    if config.get("noise_floor") and config.get("noise_reduction"):
        filters.append(f"afftdn=nf={config['noise_floor']}:nr={config['noise_reduction']}:nt=w")
    
    # 高通滤波(去除低频噪音)
    if config.get("highpass"):
        filters.append(f"highpass=f={config['highpass']}")
    
    # 低通滤波(去除高频噪音)
    if config.get("lowpass"):
        filters.append(f"lowpass=f={config['lowpass']}")
    
    # 音量增益
    if config.get("volume_boost"):
        filters.append(f"volume={config['volume_boost']}")
    
    return ",".join(filters)


def detect_silence(input_path: str, start_sec: float, end_sec: float, 
                   noise_threshold: float = -30, min_duration: float = 0.3) -> list:
    """
    检测片段中的静音/语气词区域(用于自动剪除)
    返回需要保留的时间段列表
    """
    # 使用FFmpeg silencedetect滤镜
    cmd = [
        "ffmpeg", "-y",
        "-ss", str(start_sec),
        "-t", str(end_sec - start_sec),
        "-i", input_path,
        "-af", f"silencedetect=noise={noise_threshold}dB:d={min_duration}",
        "-f", "null", "-"
    ]
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    stderr = result.stderr
    
    # 解析静音区间
    silence_starts = []
    silence_ends = []
    
    for line in stderr.split('\n'):
        if 'silence_start' in line:
            match = re.search(r'silence_start:\s*([\d.]+)', line)
            if match:
                silence_starts.append(float(match.group(1)))
        elif 'silence_end' in line:
            match = re.search(r'silence_end:\s*([\d.]+)', line)
            if match:
                silence_ends.append(float(match.group(1)))
    
    # 构建保留区间
    keep_segments = []
    prev_end = 0
    
    for i, (s_start, s_end) in enumerate(zip(silence_starts, silence_ends)):
        if s_start > prev_end + 0.1:  # 保留非静音部分
            keep_segments.append((prev_end, s_start))
        prev_end = s_end
    
    # 添加最后一段
    total_duration = end_sec - start_sec
    if prev_end < total_duration - 0.1:
        keep_segments.append((prev_end, total_duration))
    
    return keep_segments

"""
Soul派对视频分段转录脚本
- 将长音频分成20分钟段落
- 使用whisper-cpp转录
- 合并字幕并调整时间戳
- 显示进度百分比

import subprocess
import os
import sys
from pathlib import Path

# 配置
AUDIO_PATH = "/Users/karuo/Movies/soul视频/soul79_final/audio.wav"
OUTPUT_DIR = Path("/Users/karuo/Movies/soul视频/soul79_final")
WHISPER_CLI = "/usr/local/opt/whisper-cpp/bin/whisper-cli"
MODEL_PATH = os.path.expanduser("~/.cache/whisper/ggml-small.bin")
SEGMENT_DURATION = 1200  # 20分钟

def get_duration(audio_path):
    """获取音频时长"""
    result = subprocess.run([
        "ffprobe", "-v", "error", "-show_entries", "format=duration",
        "-of", "default=noprint_wrappers=1:nokey=1", audio_path
    ], capture_output=True, text=True)
    return float(result.stdout.strip())

def split_audio(audio_path, output_dir, segment_duration):
    """分割音频"""
    total_duration = get_duration(audio_path)
    segments = []
    seg_dir = output_dir / "segments"
    seg_dir.mkdir(exist_ok=True)
    
    start = 0
    idx = 0
    while start < total_duration:
        seg_path = seg_dir / f"seg_{idx:03d}.wav"
        duration = min(segment_duration, total_duration - start)
        
        subprocess.run([
            "ffmpeg", "-y", "-i", audio_path,
            "-ss", str(start), "-t", str(duration),
            "-ar", "16000", "-ac", "1",
            str(seg_path)
        ], capture_output=True)
        
        segments.append({
            "path": str(seg_path),
            "start": start,
            "index": idx
        })
        
        start += segment_duration
        idx += 1
    
    return segments, total_duration

def transcribe_segment(seg_path, output_prefix):
    """使用whisper-cpp转录单个段落"""
    result = subprocess.run([
        WHISPER_CLI,
        "-m", MODEL_PATH,
        "-l", "zh",
        "--output-srt",
        "-of", output_prefix,
        "-f", seg_path
    ], capture_output=True, text=True)
    return result.returncode == 0

def parse_srt_time(time_str):
    """解析SRT时间格式"""
    h, m, rest = time_str.split(":")
    s, ms = rest.split(",")
    return int(h) * 3600 + int(m) * 60 + int(s) + int(ms) / 1000

def format_srt_time(seconds):
    """格式化为SRT时间"""
    h = int(seconds // 3600)
    m = int((seconds % 3600) // 60)
    s = int(seconds % 60)
    ms = int((seconds % 1) * 1000)
    return f"{h:02d}:{m:02d}:{s:02d},{ms:03d}"

def merge_srt_files(segments, output_dir):
    """合并所有SRT文件并调整时间戳"""
    all_subtitles = []
    
    for seg in segments:
        srt_path = output_dir / "segments" / f"seg_{seg['index']:03d}.srt"
        if not srt_path.exists():
            continue
        
        with open(srt_path, 'r', encoding='utf-8') as f:
            content = f.read().strip()
        
        if not content:
            continue
        
        # 解析SRT
        blocks = content.split('\n\n')
        for block in blocks:
            lines = block.strip().split('\n')
            if len(lines) >= 3:
                time_line = lines[1]
                text = '\n'.join(lines[2:])
                
                # 解析时间
                start_str, end_str = time_line.split(' --> ')
                start_sec = parse_srt_time(start_str) + seg['start']
                end_sec = parse_srt_time(end_str) + seg['start']
                
                all_subtitles.append({
                    'start': start_sec,
                    'end': end_sec,
                    'text': text
                })
    
    # 写入合并后的SRT
    output_path = output_dir / "transcript.srt"
    with open(output_path, 'w', encoding='utf-8') as f:
        for i, sub in enumerate(all_subtitles, 1):
            f.write(f"{i}\n")
            f.write(f"{format_srt_time(sub['start'])} --> {format_srt_time(sub['end'])}\n")
            f.write(f"{sub['text']}\n\n")
    
    # 写入纯文本
    txt_path = output_dir / "transcript.txt"
    with open(txt_path, 'w', encoding='utf-8') as f:
        for sub in all_subtitles:
            f.write(sub['text'] + '\n')
    
    return len(all_subtitles)

def main():
    print("=" * 50)
    print("🎤 Soul派对79场 - 分段转录")
    print("=" * 50)
    
    # 获取总时长
    total_duration = get_duration(AUDIO_PATH)

    print(f"📊 总时长: {total_mins}分钟")
    print(f"📦 分段: 每段{SEGMENT_DURATION//60}分钟")
    
    # 分割音频
    print("\n📦 分割音频...")
    segments, _ = split_audio(AUDIO_PATH, OUTPUT_DIR, SEGMENT_DURATION)
    print(f"   共 {len(segments)} 段")
    
    # 转录
    print("\n🔊 开始转录...")
    for i, seg in enumerate(segments):
        pct = (i + 1) * 100 // len(segments)
        prefix = OUTPUT_DIR / "segments" / f"seg_{seg['index']:03d}"
        
        print(f"\r   ⏳ [{pct:3d}%] 转录段 {i+1}/{len(segments)}...", end="", flush=True)
        
        success = transcribe_segment(seg['path'], str(prefix))
        if not success:
            print(f" ❌")
        else:
            print(f" ✅", end="")
    
    print("\n")
    
    # 合并
    print("📝 合并字幕...")
    count = merge_srt_files(segments, OUTPUT_DIR)
    print(f"   共 {count} 条字幕")
    
    print("\n" + "=" * 50)
    print("✅ 转录完成!")
    print(f"📁 字幕: {OUTPUT_DIR / 'transcript.srt'}")
    print(f"📁 文本: {OUTPUT_DIR / 'transcript.txt'}")
    print("=" * 50)

if __name__ == "__main__":
    main()

    results = []
    for segment in tqdm(segments, desc="转录进度"):
        results.append({
            "start": segment.start,
            "end": segment.end,
            "text": segment.text.strip()
        })
    
    return results, info

def clean_subtitles(segments: list) -> list:
    """清洗字幕 - 去除语气词、合并短停顿"""
    import opencc
    converter = opencc.OpenCC('t2s')  # 繁体转简体
    
    cleaned = []
    for seg in segments:
        text = seg["text"]
        
        # 繁体转简体
        text = converter.convert(text)
        
        # 去除语气词
        for filler in FILLER_WORDS:
            text = text.replace(filler, "")
        
        # 去除多余空格
        text = " ".join(text.split())
        
        if text.strip():
            cleaned.append({
                "start": seg["start"],
                "end": seg["end"],
                "text": text.strip()
            })
    
    # 合并过短的字幕(小于1秒且文字少于5个字)
    merged = []
    for seg in cleaned:
        if merged and (seg["start"] - merged[-1]["end"]) < 0.5:
            # 合并相邻字幕
            if len(merged[-1]["text"]) < 10:
                merged[-1]["text"] += seg["text"]
                merged[-1]["end"] = seg["end"]
                continue
        merged.append(seg)
    
    return merged

def format_srt_time(seconds: float) -> str:
    """格式化SRT时间"""
    h = int(seconds // 3600)
    m = int((seconds % 3600) // 60)
    s = int(seconds % 60)
    ms = int((seconds % 1) * 1000)
    return f"{h:02d}:{m:02d}:{s:02d},{ms:03d}"

def save_srt(segments: list, output_path: str):
    """保存SRT字幕文件"""
    with open(output_path, 'w', encoding='utf-8') as f:
        for i, seg in enumerate(segments, 1):
            f.write(f"{i}\n")
            f.write(f"{format_srt_time(seg['start'])} --> {format_srt_time(seg['end'])}\n")
            f.write(f"{seg['text']}\n\n")

def save_txt(segments: list, output_path: str):
    """保存纯文本"""
    with open(output_path, 'w', encoding='utf-8') as f:
        for seg in segments:
            f.write(seg['text'] + '\n')

def save_json(segments: list, output_path: str):
    """保存JSON"""
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(segments, f, ensure_ascii=False, indent=2)

def parse_time(time_str: str) -> float:
    """解析时间字符串为秒数"""
    parts = time_str.replace(',', '.').split(':')
    if len(parts) == 3:
        h, m, s = parts
        return int(h) * 3600 + int(m) * 60 + float(s)
    elif len(parts) == 2:
        m, s = parts
        return int(m) * 60 + float(s)
    return float(time_str)


def format_time(seconds: float) -> str:
    """秒数转时间字符串"""

    m = int((seconds % 3600) // 60)

def clip_video(input_path: str, start: str, end: str, output_path: str,
               crop: str = None, enhance: bool = True, denoise: bool = True,
               subtitle_path: str = None) -> bool:
    """切片单个视频"""
    
    # 构建视频滤镜
    vf_parts = []
    if crop:
        vf_parts.append(crop)
    if enhance:
        vf_parts.append(ENHANCE_FILTER)
    if subtitle_path and os.path.exists(subtitle_path):
        # 添加字幕(需要转义路径中的特殊字符)
        safe_path = subtitle_path.replace("'", r"\'").replace(":", r"\:")
        vf_parts.append(f"subtitles='{safe_path}':force_style='FontSize=24,PrimaryColour=&HFFFFFF,OutlineColour=&H000000,Outline=2'")
    
    vf = ",".join(vf_parts) if vf_parts else None
    
    # 构建音频滤镜
    af = DENOISE_FILTER if denoise else None
    
    # 构建命令
    cmd = ["ffmpeg", "-y", "-ss", start, "-to", end, "-i", input_path]
    
    if vf:
        cmd.extend(["-vf", vf])
    if af:
        cmd.extend(["-af", af])
    
    cmd.extend([
        "-c:v", "libx264", "-preset", "fast", "-crf", "22",
        "-c:a", "aac", "-b:a", "128k",
        output_path
    ])
    
    result = subprocess.run(cmd, capture_output=True)
    return result.returncode == 0


def batch_clip(input_path: str, highlights: list, output_dir: Path,
               crop: str = None, enhance: bool = True, denoise: bool = True,
               subtitle_path: str = None, min_duration: int = 30, max_duration: int = 300):

    output_dir.mkdir(parents=True, exist_ok=True)
    results = []
    
    for i, clip in enumerate(highlights, 1):
        title = clip.get("title", f"clip_{i}")
        safe_title = re.sub(r'[<>:"/\\|?*]', '', title)[:50]
        
        start_time = clip["start_time"]
        end_time = clip["end_time"]
        
        # 计算时长
        start_sec = parse_time(start_time)
        end_sec = parse_time(end_time)
        duration = end_sec - start_sec
        
        # 时长检查
        if duration < min_duration:
            print(f"  ⚠️ 跳过 {title}: 时长{duration:.0f}秒 < {min_duration}秒")
            continue
        
        if duration > max_duration:
            print(f"  ⚠️ 截取 {title}: 限制为{max_duration}秒")
            end_sec = start_sec + max_duration
            end_time = format_time(end_sec)
        
        output_path = str(output_dir / f"{i:02d}_{safe_title}.mp4")
        
        print(f"  ✂️ [{i}/{len(highlights)}] {title}")
        print(f"     {start_time} → {end_time}")
        
        success = clip_video(
            input_path, start_time, end_time, output_path,
            crop=crop, enhance=enhance, denoise=denoise,
            subtitle_path=subtitle_path
        )
        
        if success:
            size_mb = os.path.getsize(output_path) / (1024 * 1024)
            print(f"     ✅ 完成 ({size_mb:.1f}MB)")
            results.append({"title": title, "path": output_path, "success": True})
        else:
            print(f"     ❌ 失败")
            results.append({"title": title, "path": output_path, "success": False})
    
    return results

"""
分段转录脚本 - 处理超长音频
将音频分成20分钟小段，逐段转录后合并
"""

import argparse
import os
import subprocess
import json
from pathlib import Path
from datetime import datetime
from tqdm import tqdm

# 语气词列表
FILLER_WORDS = [
    "嗯", "啊", "呃", "哦", "那个", "就是", "然后", "对吧", "是吧",
    "怎么说呢", "其实的话", "就是说", "然后的话", "对对对", "是是是",
    "好好好", "行行行", "嗯嗯嗯", "哎", "诶", "噢", "额"
]

SEGMENT_DURATION = 1200  # 20分钟一段


def get_duration(audio_path: str) -> float:
    """获取音频时长"""
    result = subprocess.run([
        "ffprobe", "-v", "error", "-show_entries", "format=duration",
        "-of", "default=noprint_wrappers=1:nokey=1", audio_path
    ], capture_output=True, text=True)
    return float(result.stdout.strip())


def split_audio(audio_path: str, output_dir: Path, segment_duration: int = 1200):
    """分割音频"""
    total_duration = get_duration(audio_path)
    segments = []
    seg_dir = output_dir / "segments"
    seg_dir.mkdir(exist_ok=True)
    
    start = 0
    idx = 0
    while start < total_duration:
        seg_path = seg_dir / f"seg_{idx:03d}.wav"
        duration = min(segment_duration, total_duration - start)
        
        print(f"  分段 {idx}: {start/60:.1f}分 - {(start+duration)/60:.1f}分")
        
        subprocess.run([
            "ffmpeg", "-y", "-i", audio_path,
            "-ss", str(start), "-t", str(duration),
            "-ar", "16000", "-ac", "1",
            str(seg_path)
        ], capture_output=True)
        
        segments.append({
            "path": str(seg_path),
            "start": start,
            "index": idx
        })
        
        start += segment_duration
        idx += 1
    
    return segments, total_duration


def transcribe_segment(audio_path: str, model_size: str = "small", language: str = "zh"):
    """转录单个音频段"""
    from faster_whisper import WhisperModel
    
    model = WhisperModel(model_size, device="cpu", compute_type="int8")
    segments, info = model.transcribe(audio_path, language=language, beam_size=5)
    
    results = []
    for segment in segments:
        results.append({
            "start": segment.start,
            "end": segment.end,
            "text": segment.text.strip()
        })
    
    return results


def clean_text(text: str) -> str:
    """清洗文本"""
    import opencc
    converter = opencc.OpenCC('t2s')
    text = converter.convert(text)
    
    for filler in FILLER_WORDS:
        text = text.replace(filler, "")
    
    return " ".join(text.split())


def format_srt_time(seconds: float) -> str:
    """格式化SRT时间"""
    h = int(seconds // 3600)
    m = int((seconds % 3600) // 60)
    s = int(seconds % 60)
    ms = int((seconds % 1) * 1000)
    return f"{h:02d}:{m:02d}:{s:02d},{ms:03d}"

    model = WhisperModel(args.model, device="cpu", compute_type="int8")

                    })

    # JSON
    json_path = output_dir / f"{base_name}.json"

    print(f"📁 TXT: {txt_path}")
    print(f"📁 JSON: {json_path}")
    print("=" * 50)

if __name__ == "__main__":
    main()

import re
import os

def parse_srt_time(time_str):
    """将SRT时间转换为秒"""
    parts = time_str.replace(',', '.').split(':')
    return float(parts[0]) * 3600 + float(parts[1]) * 60 + float(parts[2])

def format_srt_time(seconds):
    """将秒转换为SRT时间格式"""
    hours = int(seconds // 3600)
    minutes = int((seconds % 3600) // 60)
    secs = seconds % 60
    return f"{hours:02d}:{minutes:02d}:{secs:06.3f}".replace('.', ',')

def extract_subtitles(srt_path, start_sec, end_sec, output_path):
    """提取指定时间段的字幕并调整时间"""
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    blocks = content.strip().split('\n\n')
    extracted = []
    idx = 1
    
    for block in blocks:
        lines = block.strip().split('\n')
        if len(lines) >= 3:
            timestamp = lines[1]
            match = re.match(r'(\d{2}:\d{2}:\d{2}[,\.]\d{3}) --> (\d{2}:\d{2}:\d{2}[,\.]\d{3})', timestamp)
            if match:
                sub_start = parse_srt_time(match.group(1))
                sub_end = parse_srt_time(match.group(2))
                
                # 检查是否在时间范围内
                if sub_start >= start_sec and sub_end <= end_sec:
                    # 调整时间（相对于切片开始）
                    new_start = format_srt_time(sub_start - start_sec)
                    new_end = format_srt_time(sub_end - start_sec)
                    text = '\n'.join(lines[2:])
                    extracted.append(f"{idx}\n{new_start} --> {new_end}\n{text}")
                    idx += 1
    
    with open(output_path, 'w', encoding='utf-8') as f:
        f.write('\n\n'.join(extracted))
    
    return len(extracted)
