__attribute__((constructor))

        // 获取 NSApplication 的元类（因为 _crashOnException: 是类方法）
        Class appClass = objc_getMetaClass("NSApplication");
        if (!appClass) {
            NSLog(@"[NavicatCrashFix] ERROR: Could not find NSApplication metaclass");
            return;

        Method originalMethod = class_getInstanceMethod(appClass, crashSel);

            NSLog(@"[NavicatCrashFix] ERROR: Could not find _crashOnException: method");

            NSLog(@"[NavicatCrashFix] Intercepted fatal exception (app will NOT crash):");
            NSLog(@"[NavicatCrashFix]   Name: %@", [exception name]);
            NSLog(@"[NavicatCrashFix]   Reason: %@", [exception reason]);

          [exception name], [exception reason]);

static void NavicatCrashFixInit(void) {

    [[NSUserDefaults standardUserDefaults] setBool:NO forKey:@"NSApplicationCrashOnExceptions"];

            });

static CFMutableDictionaryRef CopyWithoutAccessGroup(CFDictionaryRef dict) {
    if (!dict) return NULL;
    CFMutableDictionaryRef m = CFDictionaryCreateMutableCopy(kCFAllocatorDefault, 0, dict);
    CFDictionaryRemoveValue(m, kSecAttrAccessGroup);
    return m;
}

static OSStatus hook_SecItemAdd(CFDictionaryRef attributes, CFTypeRef *result) {
    OSStatus s = orig_SecItemAdd(attributes, result);
    if (s == errSecMissingEntitlement) {
        CFMutableDictionaryRef c = CopyWithoutAccessGroup(attributes);
        if (c) { s = orig_SecItemAdd(c, result); CFRelease(c); }
    }
    return s;
}

static OSStatus hook_SecItemUpdate(CFDictionaryRef query, CFDictionaryRef attrs) {
    OSStatus s = orig_SecItemUpdate(query, attrs);
    if (s == errSecMissingEntitlement) {
        CFMutableDictionaryRef cQ = CopyWithoutAccessGroup(query);
        CFMutableDictionaryRef cA = CopyWithoutAccessGroup(attrs);
        if (cQ && cA) s = orig_SecItemUpdate(cQ, cA);
        if (cQ) CFRelease(cQ);
        if (cA) CFRelease(cA);
    }
    return s;
}

static OSStatus hook_SecItemCopyMatching(CFDictionaryRef query, CFTypeRef *result) {
    OSStatus s = orig_SecItemCopyMatching(query, result);
    if (s == errSecMissingEntitlement) {
        CFMutableDictionaryRef c = CopyWithoutAccessGroup(query);
        if (c) { s = orig_SecItemCopyMatching(c, result); CFRelease(c); }
    }
    return s;
}

static OSStatus hook_SecItemDelete(CFDictionaryRef query) {
    OSStatus s = orig_SecItemDelete(query);
    if (s == errSecMissingEntitlement) {
        CFMutableDictionaryRef c = CopyWithoutAccessGroup(query);

}

// 修改字典使其指向 legacy file-based keychain
static CFMutableDictionaryRef FixForLegacyKeychain(CFDictionaryRef dict, bool forAdd) {
    if (!dict) return NULL;
    CFMutableDictionaryRef m = CFDictionaryCreateMutableCopy(kCFAllocatorDefault, 0, dict);
    
    // 1. 移除 access group (需要 team-id entitlement)
    CFDictionaryRemoveValue(m, kSecAttrAccessGroup);
    
    // 2. 强制使用 file-based keychain
    CFDictionarySetValue(m, kSecUseDataProtectionKeychain, kCFBooleanFalse);
    
    // 3. 指定使用 default (login) keychain
    if (forAdd) {
        SecKeychainRef keychain = NULL;
        if (SecKeychainCopyDefault(&keychain) == errSecSuccess && keychain) {
            CFDictionarySetValue(m, kSecUseKeychain, keychain);
            CFRelease(keychain);
        }
    }
    
    // 4. 移除 synchronizable (file-based keychain 不支持)
    CFDictionaryRemoveValue(m, kSecAttrSynchronizable);
    
    return m;
}

static void dbglog(const char *fmt, ...) __attribute__((format(printf, 1, 2)));
static void dbglog(const char *fmt, ...) {
    if (!g_log) g_log = fopen("/tmp/navicat_crashfix.log", "a");

// 将 CFDictionary 的 key 打印出来
static void dumpDict(const char *label, CFDictionaryRef dict) {
    if (!dict) { dbglog("%s: (null)\n", label); return; }
    CFIndex count = CFDictionaryGetCount(dict);
    dbglog("%s: %ld keys\n", label, (long)count);
    
    const void *keys[64], *vals[64];
    CFIndex n = count < 64 ? count : 64;
    CFDictionaryGetKeysAndValues(dict, keys, vals);
    
    for (CFIndex i = 0; i < n; i++) {
        CFStringRef desc = CFCopyDescription(keys[i]);
        CFStringRef vdesc = CFCopyDescription(vals[i]);
        if (desc && vdesc) {
            char kbuf[256] = {0}, vbuf[512] = {0};
            CFStringGetCString(desc, kbuf, sizeof(kbuf), kCFStringEncodingUTF8);
            CFStringGetCString(vdesc, vbuf, sizeof(vbuf), kCFStringEncodingUTF8);
            dbglog("  [%ld] %s = %s\n", (long)i, kbuf, vbuf);
        }
        if (desc) CFRelease(desc);
        if (vdesc) CFRelease(vdesc);
    }
}

// 清理字典用于 file-based keychain
static CFMutableDictionaryRef CleanForFileBased(CFDictionaryRef dict) {
    if (!dict) return NULL;
    CFMutableDictionaryRef m = CFDictionaryCreateMutableCopy(kCFAllocatorDefault, 0, dict);
    CFDictionaryRemoveValue(m, kSecAttrAccessGroup);
    CFDictionaryRemoveValue(m, kSecAttrSynchronizable);
    CFDictionarySetValue(m, kSecUseDataProtectionKeychain, kCFBooleanFalse);
    return m;
}

// 层3: Legacy API fallback
static OSStatus legacyAdd(CFDictionaryRef attrs) {
    CFStringRef service = CFDictionaryGetValue(attrs, kSecAttrService);
    CFStringRef account = CFDictionaryGetValue(attrs, kSecAttrAccount);
    CFDataRef   pwData  = CFDictionaryGetValue(attrs, kSecValueData);
    
    if (!service || !account || !pwData) {
        dbglog("[KC-L3] Missing fields: svc=%p acct=%p pw=%p\n", service, account, pwData);
        return errSecParam;
    }
    
    char svcBuf[512] = {0}, acctBuf[512] = {0};
    CFStringGetCString(service, svcBuf, sizeof(svcBuf), kCFStringEncodingUTF8);
    CFStringGetCString(account, acctBuf, sizeof(acctBuf), kCFStringEncodingUTF8);
    const UInt8 *pw = CFDataGetBytePtr(pwData);
    CFIndex pwLen = CFDataGetLength(pwData);
    
    dbglog("[KC-L3] svc='%s' acct='%s' pwLen=%ld\n", svcBuf, acctBuf, (long)pwLen);
    
    // 先检查是否已存在
    SecKeychainItemRef existingItem = NULL;
    #pragma clang diagnostic push
    #pragma clang diagnostic ignored "-Wdeprecated-declarations"
    OSStatus fs = SecKeychainFindGenericPassword(
        NULL, (UInt32)strlen(svcBuf), svcBuf,
        (UInt32)strlen(acctBuf), acctBuf,
        NULL, NULL, &existingItem);
    
    if (fs == errSecSuccess && existingItem) {
        OSStatus s = SecKeychainItemModifyContent(existingItem, NULL, (UInt32)pwLen, pw);
        dbglog("[KC-L3] Update existing: %d\n", (int)s);
        CFRelease(existingItem);
        return s;
    }
    
    OSStatus s = SecKeychainAddGenericPassword(
        NULL, (UInt32)strlen(svcBuf), svcBuf,
        (UInt32)strlen(acctBuf), acctBuf,
        (UInt32)pwLen, pw, NULL);
    dbglog("[KC-L3] Add new: %d\n", (int)s);
    #pragma clang diagnostic pop
    
    return s;
}

static OSStatus legacyFind(CFDictionaryRef query, CFTypeRef *result) {
    CFStringRef service = CFDictionaryGetValue(query, kSecAttrService);
    CFStringRef account = CFDictionaryGetValue(query, kSecAttrAccount);
    
    if (!service || !account || !result) return errSecParam;
    
    char svcBuf[512] = {0}, acctBuf[512] = {0};
    CFStringGetCString(service, svcBuf, sizeof(svcBuf), kCFStringEncodingUTF8);
    CFStringGetCString(account, acctBuf, sizeof(acctBuf), kCFStringEncodingUTF8);
    
    UInt32 pwLen = 0;
    void *pwData = NULL;
    
    #pragma clang diagnostic push
    #pragma clang diagnostic ignored "-Wdeprecated-declarations"
    OSStatus s = SecKeychainFindGenericPassword(
        NULL, (UInt32)strlen(svcBuf), svcBuf,
        (UInt32)strlen(acctBuf), acctBuf,
        &pwLen, &pwData, NULL);
    
    if (s == errSecSuccess && pwData) {
        *result = CFDataCreate(kCFAllocatorDefault, (UInt8 *)pwData, pwLen);
        SecKeychainItemFreeContent(NULL, pwData);
    }
    #pragma clang diagnostic pop
    
    return s;
}

static OSStatus hook_SecItemAdd(CFDictionaryRef attrs, CFTypeRef *result) {
    dbglog("\n[KC] === SecItemAdd ===\n");
    dumpDict("[KC] Input", attrs);
    
    // 层1: 直接用 file-based keychain
    CFMutableDictionaryRef cleaned = CleanForFileBased(attrs);
    if (cleaned) {
        OSStatus s = orig_SecItemAdd((CFDictionaryRef)cleaned, result);
        dbglog("[KC-L1] file-based result: %d\n", (int)s);
        
        if (s == errSecDuplicateItem) {
            orig_SecItemDelete((CFDictionaryRef)cleaned);
            s = orig_SecItemAdd((CFDictionaryRef)cleaned, result);
            dbglog("[KC-L1] after dedup: %d\n", (int)s);
        }
        
        if (s == errSecSuccess) { CFRelease(cleaned); return s; }
        CFRelease(cleaned);
    }
    
    // 层2: 最简化查询
    CFMutableDictionaryRef minimal = CFDictionaryCreateMutable(
        kCFAllocatorDefault, 0,
        &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
    CFDictionarySetValue(minimal, kSecClass, kSecClassGenericPassword);
    CFDictionarySetValue(minimal, kSecUseDataProtectionKeychain, kCFBooleanFalse);
    
    // 复制必要字段
    CFStringRef keys[] = {kSecAttrService, kSecAttrAccount, kSecValueData, kSecAttrLabel};
    for (int i = 0; i < 4; i++) {
        CFTypeRef val = CFDictionaryGetValue(attrs, keys[i]);
        if (val) CFDictionarySetValue(minimal, keys[i], val);
    }
    
    orig_SecItemDelete(minimal); // 先清理
    OSStatus s2 = orig_SecItemAdd(minimal, result);
    dbglog("[KC-L2] minimal result: %d\n", (int)s2);
    CFRelease(minimal);
    
    if (s2 == errSecSuccess) return s2;
    
    // 层3: Legacy API
    dbglog("[KC-L3] Trying legacy SecKeychain API...\n");
    OSStatus s3 = legacyAdd(attrs);
    return s3;
}

static OSStatus hook_SecItemUpdate(CFDictionaryRef query, CFDictionaryRef attrs) {
    dbglog("\n[KC] === SecItemUpdate ===\n");
    CFMutableDictionaryRef fQ = CleanForFileBased(query);
    CFMutableDictionaryRef fA = CleanForFileBased(attrs);
    
    OSStatus s = orig_SecItemUpdate(fQ ? (CFDictionaryRef)fQ : query,
                                     fA ? (CFDictionaryRef)fA : attrs);
    dbglog("[KC] SecItemUpdate: %d\n", (int)s);
    
    if (fQ) CFRelease(fQ);
    if (fA) CFRelease(fA);
    return s;
}

static OSStatus hook_SecItemCopyMatching(CFDictionaryRef query, CFTypeRef *result) {
    CFMutableDictionaryRef f = CleanForFileBased(query);
    OSStatus s = orig_SecItemCopyMatching(f ? (CFDictionaryRef)f : query, result);
    
    if (s == errSecMissingEntitlement && result) {
        dbglog("[KC] CopyMatching -34018, trying legacy\n");
        s = legacyFind(query, result);
    }
    
    if (f) CFRelease(f);
    return s;
}

static OSStatus hook_SecItemDelete(CFDictionaryRef query) {
    CFMutableDictionaryRef f = CleanForFileBased(query);
    OSStatus s = orig_SecItemDelete(f ? (CFDictionaryRef)f : query);
    if (f) CFRelease(f);
    return s;
}
