# mail.tm API（免费，无需 key）
MAIL_TM_BASE = "https://api.mail.tm"


def get_mailtm_domain() -> str:
    """获取 mail.tm 可用域名"""
    r = requests.get(f"{MAIL_TM_BASE}/domains", timeout=10)
    r.raise_for_status()
    data = r.json()
    domains = data.get("hydra:member", [])
    if not domains:
        raise RuntimeError("mail.tm 无可用域名")
    return domains[0]["domain"]


def create_mailtm_account(domain: str, password: str) -> tuple[str, str]:
    """创建 mail.tm 临时邮箱，返回 (email, token)"""
    local = "".join(random.choices(string.ascii_lowercase + string.digits, k=10))
    email = f"{local}@{domain}"
    payload = {"address": email, "password": password}
    r = requests.post(
        f"{MAIL_TM_BASE}/accounts",
        json=payload,
        headers={"Content-Type": "application/json"},
        timeout=10,
    )
    if r.status_code in (400, 401):
        # 可能已存在，换一个
        local = "".join(random.choices(string.ascii_lowercase + string.digits, k=12))
        email = f"{local}@{domain}"
        payload["address"] = email
        r = requests.post(
            f"{MAIL_TM_BASE}/accounts",
            json=payload,
            headers={"Content-Type": "application/json"},
            timeout=10,
        )
    r.raise_for_status()
    # 获取 token
    tr = requests.post(
        f"{MAIL_TM_BASE}/token",
        json={"address": email, "password": password},
        headers={"Content-Type": "application/json"},
        timeout=10,
    )
    tr.raise_for_status()
    token = tr.json().get("token", "")
    return email, token


def get_verification_link(token: str, max_wait: int = 120) -> str | None:
    """从 mail.tm 收件箱获取验证链接（如 SMS-Activate 发的）"""
    headers = {"Authorization": f"Bearer {token}"}
    for _ in range(max_wait // 5):
        r = requests.get(
            f"{MAIL_TM_BASE}/messages",
            headers=headers,
            timeout=10,
        )
        if r.status_code != 200:
            time.sleep(5)
            continue
        data = r.json()
        msgs = data.get("hydra:member", [])
        for m in msgs:
            mid = m.get("id")
            if not mid:
                continue
            mr = requests.get(
                f"{MAIL_TM_BASE}/messages/{mid}",
                headers=headers,
                timeout=10,
            )
            if mr.status_code != 200:
                continue
            body = mr.json().get("text", "") or mr.json().get("html", "")
            # 匹配常见验证链接
            match = re.search(
                r'https?://[^\s<>"\']+activation[^\s<>"\']*|'
                r'https?://[^\s<>"\']+verify[^\s<>"\']*|'
                r'https?://sms-activate[^\s<>"\']*',
                body,
                re.I,
            )
            if match:
                return match.group(0)
        time.sleep(5)
    return None

MAIL_TM_BASE = "https://api.mail.tm"


def get_mailtm_domain() -> str:
    r = requests.get(f"{MAIL_TM_BASE}/domains", timeout=10)
    r.raise_for_status()
    domains = r.json().get("hydra:member", [])
    if not domains:
        raise RuntimeError("mail.tm 无可用域名")
    return domains[0]["domain"]


def create_mailtm_account(domain: str, password: str) -> tuple[str, str]:
    local = "".join(random.choices(string.ascii_lowercase + string.digits, k=12))
    email = f"{local}@{domain}"
    payload = {"address": email, "password": password}
    r = requests.post(f"{MAIL_TM_BASE}/accounts", json=payload, headers={"Content-Type": "application/json"}, timeout=10)
    r.raise_for_status()
    tr = requests.post(f"{MAIL_TM_BASE}/token", json={"address": email, "password": password}, headers={"Content-Type": "application/json"}, timeout=10)
    tr.raise_for_status()
    return email, tr.json().get("token", "")


def get_verification_link(token: str, max_wait: int = 90) -> str | None:
    headers = {"Authorization": f"Bearer {token}"}
    for _ in range(max_wait // 5):
        r = requests.get(f"{MAIL_TM_BASE}/messages", headers=headers, timeout=10)
        if r.status_code != 200:
            time.sleep(5)
            continue
        for m in r.json().get("hydra:member", []):
            mid = m.get("id")
            if not mid:
                continue
            mr = requests.get(f"{MAIL_TM_BASE}/messages/{mid}", headers=headers, timeout=10)
            if mr.status_code != 200:
                continue
            body = mr.json().get("text", "") or mr.json().get("html", "")
            match = re.search(r'https?://[^\s<>"\']+(?:verify|confirm|activation|signup)[^\s<>"\']*', body, re.I)
            if match:
                return match.group(0)
        time.sleep(5)
    return None


def register_5sim(page, email: str, password: str, mail_token: str | None) -> bool:
    """5sim.net 注册，表单更简单"""
    page.goto("https://5sim.net/zh/signup", wait_until="domcontentloaded")
    time.sleep(4)

    # 接受 cookie
    try:
        page.locator('button:has-text("接受"), button:has-text("Accept")').first.click(timeout=3000)
        time.sleep(1)
    except Exception:
        pass

    # 填表：5sim 有 电子邮箱、密码、重复输入密码
    page.locator('input[type="email"], input[name="email"], input[placeholder*="mail"], input[placeholder*="邮箱"]').first.fill(email)
    time.sleep(0.3)
    pw_inputs = page.locator('input[type="password"]')
    if pw_inputs.count() >= 2:
        pw_inputs.nth(0).fill(password)
        time.sleep(0.2)
        pw_inputs.nth(1).fill(password)
    else:
        pw_inputs.first.fill(password)

    # 勾选协议
    try:
        page.locator('input[type="checkbox"], [role="checkbox"]').first.check(timeout=2000)
    except Exception:
        pass

    time.sleep(0.5)
    page.locator('button:has-text("注册"), button[type="submit"]').first.click(timeout=5000)
    time.sleep(6)

    # 若需邮箱验证
    if mail_token:
        link = get_verification_link(mail_token)
        if link:
            page.goto(link, wait_until="domcontentloaded")
            time.sleep(4)

    content = page.content()
    return "登录" in content and ("余额" in content or "Balance" in content or "退出" in content or "登出" in content or "profile" in content.lower())
