### 手动操作

```python
# Python交互模式
from scripts.brokers.xcsc_web_trader import XCSCWebTrader
import asyncio

async def trade():
    trader = XCSCWebTrader()
    await trader.init_browser(headless=False)
    await trader.login()
    
    # 查询资金
    balance = await trader.get_balance()
    print(f"可用资金: ¥{balance['available']:,.2f}")
    
    # 买入股票
    result = await trader.buy("600519", 1800.00, 100)
    print(result)
    
    # 卖出股票
    result = await trader.sell("600519", 1850.00, 100)
    print(result)
    
    await trader.close()

asyncio.run(trade())
```
