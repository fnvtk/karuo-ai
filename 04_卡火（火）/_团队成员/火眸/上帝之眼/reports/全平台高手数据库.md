### 高手评分算法

```python
def calculate_expert_score(expert):
    """
    高手综合评分算法
    
    评分维度：
    1. 历史收益率 (40%)
    2. 胜率 (30%)
    3. 回撤控制 (20%)
    4. 活跃度 (10%)
    """
    
    # 收益率归一化 (0-1)
    return_score = min(1.0, max(0, expert['annual_return'] / 100))
    
    # 胜率归一化 (0-1)
    win_rate = expert['win_rate'] / 100
    
    # 回撤控制归一化 (回撤越小分越高)
    drawdown_score = 1 - min(1.0, abs(expert['max_drawdown']) / 50)
    
    # 活跃度 (月均交易次数 / 10)
    activity_score = min(1.0, expert['monthly_trades'] / 10)
    
    # 加权计算
    score = (
        return_score * 0.4 +
        win_rate * 0.3 +
        drawdown_score * 0.2 +
        activity_score * 0.1
    )
    
    return round(score, 3)


# 筛选条件
EXPERT_FILTER = {
```

```python
    async def get_top_portfolios(self, top_n=100):
        """获取排行榜前N组合"""
```
