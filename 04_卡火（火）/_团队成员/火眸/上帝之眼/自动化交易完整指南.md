# 二、BigQuant策略部署

## 2.1 策略代码（卡若六步选股法）

登录 https://bigquant.com，创建新策略，粘贴以下代码：

```python
"""
上帝之眼 - 卡若六步选股法
BigQuant模拟交易策略

策略说明：
- 基于5年历史数据选股
- 四维度评分：均线+历史+高手+大盘
- 每板块最多2只，共选3只
- 节假日前2天买入，复盘后第7天卖出
"""

from bigmodule import M
from bigtrader.sdk import *
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# ============ 策略参数 ============
HOLIDAY_STOCKS = {
    '春节': {
        'buy_start': '2026-02-04',
        'sell_date': '2026-02-25',
        'stocks': [
            {'code': '600519.SHA', 'name': '贵州茅台', 'weight': 0.33},
            {'code': '000858.SZA', 'name': '五粮液', 'weight': 0.33},
            {'code': '300760.SZA', 'name': '迈瑞医疗', 'weight': 0.34},
        ]
    },
    '五一': {
        'buy_start': '2026-04-28',
        'sell_date': '2026-05-14',
        'stocks': [
            {'code': '600887.SHA', 'name': '伊利股份', 'weight': 0.33},
            {'code': '600138.SHA', 'name': '中青旅', 'weight': 0.33},
            {'code': '000651.SZA', 'name': '格力电器', 'weight': 0.34},
        ]
    },
    '国庆': {
        'buy_start': '2026-09-28',
        'sell_date': '2026-10-19',
        'stocks': [
            {'code': '300750.SZA', 'name': '宁德时代', 'weight': 0.33},
            {'code': '600893.SHA', 'name': '航发动力', 'weight': 0.33},
            {'code': '002594.SZA', 'name': '比亚迪', 'weight': 0.34},
        ]
    },
}

# 止损止盈参数
STOP_LOSS = -0.05  # 止损线 -5%
TARGET_PROFIT = 0.08  # 目标收益 +8%


def init(context):
    """初始化"""
    context.set_commission(PerOrder(buy_cost=0.0003, sell_cost=0.0013, min_cost=5))
    context.holiday_config = get_current_holiday()
    context.bought = False
    context.buy_prices = {}
    
    print(f"[初始化] 当前节假日配置: {context.holiday_config}")


def get_current_holiday():
    """获取当前应该执行的节假日配置"""
    today = datetime.now().strftime('%Y-%m-%d')
    
    for holiday, config in HOLIDAY_STOCKS.items():
        buy_start = config['buy_start']
        sell_date = config['sell_date']
        
        if buy_start <= today <= sell_date:
            return {'name': holiday, **config}
    
    return None


def check_market_open_status(context):
    """
    检查大盘开盘状态
    返回：'high_high'(高开高走), 'low_high'(低开高走), 'high_low'(高开低走), 'low_low'(低开低走), 'flat'(平开)
    """
    try:
        # 获取上证指数数据
        index_data = context.data.history(
            instruments=['000001.SHA'],
            start_date=context.trading_day,
            end_date=context.trading_day,
            fields=['open', 'close', 'pre_close']
        )
        
        if index_data.empty:
            return 'unknown'
        
        open_price = index_data['open'].iloc[0]
        close_price = index_data['close'].iloc[0]
        pre_close = index_data['pre_close'].iloc[0]
        
        open_pct = (open_price - pre_close) / pre_close
        day_pct = (close_price - open_price) / open_price
        
        if open_pct > 0.005:  # 高开
            if day_pct > 0:
                return 'high_high'  # 高开高走
            else:
                return 'high_low'   # 高开低走
        elif open_pct < -0.005:  # 低开
            if day_pct > 0:
                return 'low_high'   # 低开高走
            else:
                return 'low_low'    # 低开低走
        else:
            return 'flat'  # 平开
            
    except:
        return 'unknown'


def should_buy_now(context):
    """判断当前是否应该买入"""
    market_status = check_market_open_status(context)
    
    # 大盘买入判断
    if market_status in ['high_high', 'low_high', 'flat']:
        return True
    elif market_status in ['high_low', 'low_low']:
        print(f"[大盘判断] 当前大盘状态: {market_status}，不适合买入")
        return False
    
    return True


def handle_data(context, data):
    """每日执行"""
    today = context.trading_day.strftime('%Y-%m-%d')
    config = context.holiday_config
    
    if config is None:
        print(f"[{today}] 不在节假日交易期间")
        return
    
    holiday_name = config['name']
    buy_start = config['buy_start']
    sell_date = config['sell_date']
    stocks = config['stocks']
    
    # ===== 买入逻辑 =====
    if today == buy_start and not context.bought:
        print(f"[{today}] {holiday_name}买入日")
        
        # 检查大盘状态
        if not should_buy_now(context):
            print(f"[{today}] 大盘状态不佳，延迟买入")
            return
        
        # 获取可用资金
        cash = context.portfolio.cash
        
        for stock in stocks:
            code = stock['code']
            name = stock['name']
            weight = stock['weight']
            
            # 计算买入金额
            amount = cash * weight * 0.95  # 留5%余量
            
            # 获取当前价格
            price = data.current(code, 'close')
            
            if price and price > 0:
                # 计算买入股数（100的整数倍）
                shares = int(amount / price / 100) * 100
                
                if shares >= 100:
                    context.order(code, shares)
                    context.buy_prices[code] = price
                    print(f"  买入 {code} {name}: {shares}股 @ ¥{price:.2f}")
        
        context.bought = True
    
    # ===== 止损止盈检查 =====
    if context.bought:
        for code, buy_price in list(context.buy_prices.items()):
            current_price = data.current(code, 'close')
            
            if current_price:
                pct = (current_price - buy_price) / buy_price
                
                # 止损
                if pct <= STOP_LOSS:
                    position = context.portfolio.positions.get(code)
                    if position and position.current_qty > 0:
                        context.order_target(code, 0)
                        print(f"  止损 {code}: 亏损{pct*100:.1f}%")
                        del context.buy_prices[code]
                
                # 止盈
                elif pct >= TARGET_PROFIT:
                    position = context.portfolio.positions.get(code)
                    if position and position.current_qty > 0:
                        context.order_target(code, 0)
                        print(f"  止盈 {code}: 盈利{pct*100:.1f}%")
                        del context.buy_prices[code]
    
    # ===== 卖出逻辑 =====
    if today == sell_date and context.bought:
        print(f"[{today}] {holiday_name}卖出日，清仓")
        
        for code in list(context.buy_prices.keys()):
            position = context.portfolio.positions.get(code)
            if position and position.current_qty > 0:
                context.order_target(code, 0)
                print(f"  卖出 {code}: {position.current_qty}股")
        
        context.bought = False
        context.buy_prices = {}


# BigQuant模块配置
m1 = M.instruments.v2(
    start_date='2026-01-01',
    end_date='2026-12-31',
    market='CN_STOCK_A',
    instrument_list='',
    max_count=0
)

m2 = M.advanced_auto_labeler.v2(
    instruments=m1.data,
    label_expr='shift(close, -5) / close - 1',
    start_date='',
    end_date='',
    benchmark='000300.SHA',
    drop_na_label=True,
    cast_label_int=False
)

m3 = M.input_features.v1(
    features="""
    close
    open
    high
    low
    volume
    """
)

m4 = M.general_feature_extractor.v7(
    instruments=m1.data,
    features=m3.data,
    start_date='',
    end_date='',
    before_start_days=90
)

m5 = M.trade.v4(
    instruments=m1.data,
    options_data=m4.data,
    start_date='',
    end_date='',
    initialize=init,
    handle_data=handle_data,
    capital_base=2570000,  # 总资产257万
    frequency='daily',
    benchmark='000300.SHA',
    price_type='close',
    product_type='stock',
    before_start_days=0,
    volume_limit=0.025,
    order_price_field_buy='open',
    order_price_field_sell='close',
    allow_missing_instrument=True,
    slippage=0.001
)
```

# 五、常见问题

## Q1: 终端无法登录？

```
解决方案：
1. 确认登录时间在 08:45 之后
2. 检查账号密码是否正确
3. 尝试更换节点
4. 联系BigQuant客服
```

## Q2: 交易计划为空？

```
解决方案：
1. 确认模拟交易策略已部署
2. 确认访问凭证已正确绑定
3. 点击「重新生成」刷新
```

## Q3: 委托失败？

```
常见原因：
1. 资金不足 → 检查可用资金
2. 数量不是100的整数倍 → 调整数量
3. 价格超出涨跌停 → 检查价格
4. 无权限 → 联系券商开通
```

## Q4: 终端关闭后信号还会执行吗？

```
不会！
终端关闭后，已确认的信号不会执行。
需要在交易时间保持终端打开。
```
