## 二、雪球数据接入

### 数据内容

- **组合持仓**：高手的股票持仓明细
- **调仓记录**：买入/卖出操作历史
- **收益曲线**：组合的收益走势
- **帖子内容**：高手的分析文章

### 爬虫实现

```python
import aiohttp
import re
from datetime import datetime

class XueqiuCrawler:
    """雪球网数据采集"""
    
    BASE_URL = "https://xueqiu.com"
    
    def __init__(self):
        self.headers = {
            'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_7)',
            'Accept': 'application/json',
            'Referer': 'https://xueqiu.com'
        }
        self.session = None
    
    async def get_top_portfolios(self, top_n: int = 100):
        """获取高手组合排行榜"""
        url = f"{self.BASE_URL}/cubes/discover/rank/cube/list.json"
        params = {
            'category': '1',  # 1=总收益
            'count': top_n
        }
        
        async with aiohttp.ClientSession(headers=self.headers) as session:
            async with session.get(url, params=params) as response:
                data = await response.json()
                
                portfolios = []
                for item in data.get('list', []):
                    portfolios.append({
                        'id': item.get('symbol'),
                        'name': item.get('name'),
                        'owner': item.get('owner', {}).get('screen_name'),
                        'total_gain': item.get('total_gain'),
                        'followers': item.get('followers_count')
                    })
                
                return portfolios
    
    async def get_portfolio_holdings(self, portfolio_id: str):
        """获取组合持仓"""
        url = f"{self.BASE_URL}/cubes/rebalancing/current.json"
        params = {'cube_symbol': portfolio_id}
        
        async with aiohttp.ClientSession(headers=self.headers) as session:
            async with session.get(url, params=params) as response:
                data = await response.json()
                
                holdings = []
                for item in data.get('holdings', []):
                    holdings.append({
                        'stock_code': item.get('stock_symbol', '').replace('SH', '').replace('SZ', ''),
                        'stock_name': item.get('stock_name'),
                        'weight': item.get('weight'),
                        'price': item.get('price')
                    })
                
                return holdings
    
    async def get_portfolio_changes(self, portfolio_id: str, count: int = 20):
        """获取调仓记录"""
        url = f"{self.BASE_URL}/cubes/rebalancing/history.json"
        params = {
            'cube_symbol': portfolio_id,
            'count': count
        }
        
        async with aiohttp.ClientSession(headers=self.headers) as session:
            async with session.get(url, params=params) as response:
                data = await response.json()
                
                changes = []
                for item in data.get('list', []):
                    prev_weight = item.get('prev_weight_adjusted', 0)
                    target_weight = item.get('target_weight', 0)
                    
                    changes.append({
                        'stock_code': item.get('stock_symbol', '').replace('SH', '').replace('SZ', ''),
                        'stock_name': item.get('stock_name'),
                        'action': 'buy' if target_weight > prev_weight else 'sell',
                        'prev_weight': prev_weight,
                        'target_weight': target_weight,
                        'price': item.get('price'),
                        'created_at': datetime.fromtimestamp(item.get('created_at', 0) / 1000)
                    })
                
                return changes
    
    async def get_hot_posts(self, count: int = 20):
        """获取热门帖子"""
        url = f"{self.BASE_URL}/statuses/hot/listV2.json"
        params = {
            'category': '-1',
            'count': count,
            'source': 'all'
        }
        
        async with aiohttp.ClientSession(headers=self.headers) as session:
            async with session.get(url, params=params) as response:
                data = await response.json()
                
                posts = []
                for item in data.get('list', []):
                    # 提取股票代码
                    text = item.get('text', '') + item.get('description', '')
                    stocks = self._extract_stocks(text)
                    
                    posts.append({
                        'id': item.get('id'),
                        'author': item.get('user', {}).get('screen_name'),
                        'title': item.get('title'),
                        'text': item.get('text'),
                        'stocks': stocks,
                        'likes': item.get('like_count'),
                        'comments': item.get('reply_count')
                    })
                
                return posts
    
    def _extract_stocks(self, text: str):
        """提取股票代码"""
        stocks = []
        
        # $SH600519$ 格式
        pattern1 = r'\$([A-Z]{2}\d{6})\$'
        matches = re.findall(pattern1, text)
        stocks.extend([m[2:] for m in matches])
        
        # (SH600519) 格式
        pattern2 = r'\(([A-Z]{2}\d{6})\)'
        matches = re.findall(pattern2, text)
        stocks.extend([m[2:] for m in matches])
        
        return list(set(stocks))
```

### 使用示例

```python
crawler = XueqiuCrawler()

# 获取高手排行

# 获取调仓记录
```

## 四、东方财富数据接入

### 数据内容

- **板块数据**：行业板块、概念板块涨跌
- **资金流向**：主力资金、北向资金
- **行情数据**：股票实时价格

### API接口

```python
import aiohttp

class EastMoneyCrawler:
    """东方财富数据采集"""
    
    async def get_sector_list(self, top_n: int = 10):
        """获取板块涨幅排行"""
        url = "http://push2.eastmoney.com/api/qt/clist/get"
        params = {
            'pn': 1,
            'pz': top_n,
            'po': 1,
            'np': 1,
            'fltt': 2,
            'invt': 2,
            'fid': 'f3',  # 按涨幅排序
            'fs': 'm:90+t:2',  # 行业板块
            'fields': 'f1,f2,f3,f4,f12,f14'
        }
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url, params=params) as response:
                data = await response.json()
                
                sectors = []
                for item in data.get('data', {}).get('diff', []):
                    sectors.append({
                        'code': item.get('f12'),
                        'name': item.get('f14'),
                        'price': item.get('f2'),
                        'change_pct': item.get('f3') / 100,
                        'change_amount': item.get('f4')
                    })
                
                return sectors
    
    async def get_money_flow(self, stock_code: str):
        """获取资金流向"""
        url = "http://push2.eastmoney.com/api/qt/stock/fflow/kline/get"
        params = {
            'secid': f'1.{stock_code}' if stock_code.startswith('6') else f'0.{stock_code}',
            'fields1': 'f1,f2,f3',
            'fields2': 'f51,f52,f53,f54,f55'
        }
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url, params=params) as response:
                data = await response.json()
                return data.get('data', {}).get('klines', [])
```

## 六、新浪财经数据接入

### 实时行情

```python
import aiohttp
import re

class SinaCrawler:
    """新浪财经数据采集"""
    
    async def get_quote(self, stock_code: str):
        """获取实时行情"""
        # 转换代码格式
        if stock_code.startswith('6'):
            sina_code = f'sh{stock_code}'
        else:
            sina_code = f'sz{stock_code}'
        
        url = f'http://hq.sinajs.cn/list={sina_code}'
        
        async with aiohttp.ClientSession() as session:
            headers = {'Referer': 'http://finance.sina.com.cn'}
            async with session.get(url, headers=headers) as response:
                text = await response.text()
                
                # 解析: var hq_str_sh600519="贵州茅台,开盘,昨收,现价,..."
                match = re.search(r'"(.+)"', text)
                if match:
                    parts = match.group(1).split(',')
                    return {
                        'name': parts[0],
                        'open': float(parts[1]),
                        'close': float(parts[2]),
                        'price': float(parts[3]),
                        'high': float(parts[4]),
                        'low': float(parts[5]),
                        'volume': int(float(parts[8])),
                        'amount': float(parts[9])
                    }
        
        return None
    
    async def get_index(self, index_code: str = 'sh000001'):
        """获取指数行情"""
        return await self.get_quote(index_code.replace('sh', '').replace('sz', ''))
```

## 七、统一数据接口

### 数据采集器

```python
class DataCollector:
    """统一数据采集接口"""
    
    def __init__(self, config):
        self.config = config
        self.xueqiu = XueqiuCrawler()
        self.ths = THSCrawler()
        self.eastmoney = EastMoneyCrawler()
        self.sina = SinaCrawler()
    
    async def get_experts(self, platform: str = 'all', top_n: int = 50):
        """获取各平台高手"""
        experts = []
        
        if platform in ['all', 'xueqiu']:
            xueqiu_experts = await self.xueqiu.get_top_portfolios(top_n)
            experts.extend([{**e, 'platform': 'xueqiu'} for e in xueqiu_experts])
        
        if platform in ['all', 'ths']:
            ths_experts = await self.ths.get_top_traders(top_n)
            experts.extend([{**e, 'platform': 'ths'} for e in ths_experts])
        
        return experts
    
    async def get_sectors(self, top_n: int = 10):
        """获取板块数据"""
        return await self.eastmoney.get_sector_list(top_n)
    
    async def get_quote(self, stock_code: str):
        """获取实时行情"""
        return await self.sina.get_quote(stock_code)
```

## 八、注意事项

### 反爬虫处理

1. **请求频率**：控制在每秒1-2次
2. **User-Agent**：模拟浏览器
3. **Cookie**：部分接口需要登录Cookie
4. **IP代理**：大量采集时使用代理池

### 数据合规

1. 仅用于个人投资研究
2. 不得商业化出售数据
3. 遵守各平台服务条款
